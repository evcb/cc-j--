5   /**
6    * An enum of token kinds. Each entry in this enum represents the kind of a
7    * token along with its image (string representation).
8    * <p>
9    * When you add a new token to the scanner, you must also add an entry to this
10   * enum specifying the kind and image of the new token.
11   */
12  
13  enum TokenKind {
14      EOF("<EOF>"), 
15  
16      // Reserved words
17      ABSTRACT("abstract"), BOOLEAN("boolean"), CHAR("char"), CLASS("class"), 
18      ELSE("else"), EXTENDS("extends"), IF("if"), IMPORT("import"), 
19      INSTANCEOF("instanceof"), INT("int"), NEW("new"), PACKAGE("package"), 
20      PRIVATE("private"), PROTECTED("protected"), PUBLIC("public"), 
21      RETURN("return"), STATIC("static"), SUPER("super"), THIS("this"), 
22      VOID("void"), WHILE("while"), 
23  
24      // Operators
25      ASSIGN("="), DEC("--"), EQUAL("=="), GT(">"), INC("++"), LAND("&&"), 
26      LE("<="), LNOT("!"), MINUS("-"), PLUS("+"), PLUS_ASSIGN("+="), STAR("*"),
27  
28      // Separators
29      LPAREN("("), RPAREN(")"), LCURLY("{"), RCURLY("}"), LBRACK("["), 
30      RBRACK("]"), SEMI(";"), COMMA(","), DOT("."), 
31  
32      // Identifiers
33      IDENTIFIER("<IDENTIFIER>"), 
34  
35      // Literals
36      NULL("null"), FALSE("false"), TRUE("true"),
37      INT_LITERAL("<INT_LITERAL>"), CHAR_LITERAL("<CHAR_LITERAL>"), 
38      STRING_LITERAL("<STRING_LITERAL>");
39  
40      /** The token's string representation. */
41      private String image;
42  
43      /**
44       * Constructs an instance TokenKind given its string representation.
45       * 
46       * @param image
47       *            string representation of the token.
48       */
49  
50      private TokenKind(String image) {
51          this.image = image;
52      }
53  
54      /**
55       * Returns the image of the token.
56       * 
57       * @return the token's image.
58       */
59  
60      public String image() {
61          return image;
62      }
63  
64      /**
65       * Returns the string representation of the token.
66       * 
67       * @return the token's string representation.
68       */
69  
70      public String toString() {
71          return image;
72      }
73  
74  }
75  
76  /**
77   * A representation of tokens returned by the lexical analyzer method,
78   * {@link Scanner#getNextToken() getNextToken}. A token has a kind identifying 
79   * what kind of token it is, an image for providing any semantic text, and the 
80   * line in which it occurred in the source file.
81   */
82  
83  class TokenInfo {
84  
85      /** Token kind. */
86      private TokenKind kind;
87  
88      /**
89       * Semantic text (if any). For example, the identifier name when the token
90       * kind is IDENTIFIER. For tokens without a semantic text, it is simply its
91       * string representation. For example, "+=" when the token kind is
92       * PLUS_ASSIGN.
93       */
94      private String image;
95  
96      /** Line in which the token occurs in the source file. */
97      private int line;
98  
99      /**
100      * Constructs a TokenInfo given its kind, the semantic text forming the token,
101      * and its line number.
102      * 
103      * @param kind
104      *            the token's kind.
105      * @param image
106      *            the semantic text comprising the token.
107      * @param line
108      *            the line in which the token occurs in the source file.
109      */
110 
111     public TokenInfo(TokenKind kind, String image, int line) {
112         this.kind = kind;
113         this.image = image;
114         this.line = line;
115     }
116 
117     /**
118      * Constructs a TokenInfo given its kind and its line number. Its image is
119      * simply its string representation.
120      * 
121      * @param kind
122      *            the token's identifying number.
123      * @param line
124      *            identifying the line on which the token was found.
125      */
126 
127     public TokenInfo(TokenKind kind, int line) {
128         this(kind, kind.toString(), line);
129     }
130 
131     /**
132      * Returns the token's string representation.
133      * 
134      * @return the string representation.
135      */
136 
137     public String tokenRep() {
138         return kind.toString();
139     }
140 
141     /**
142      * Returns the semantic text associated with the token.
143      * 
144      * @return the semantic text.
145      */
146 
147     public String image() {
148         return image;
149     }
150 
151     /**
152      * Returns the line number associated with the token.
153      * 
154      * @return the line number.
155      */
156 
157     public int line() {
158         return line;
159     }
160 
161     /**
162      * Returns the token's kind.
163      * 
164      * @return the kind.
165      */
166 
167     public TokenKind kind() {
168         return kind;
169     }
170 
171     /**
172      * Returns the token's image.
173      * 
174      * @return the image.
175      */
176 
177     public String toString() {
178         return image;
179     }
180 
181 }
