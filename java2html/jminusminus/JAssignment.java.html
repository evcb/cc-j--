7   /**
8    * This abstract base class is the AST node for an assignment statement. 
9    */
10  
11  abstract class JAssignment extends JBinaryExpression {
12  
13      /**
14       * Constructs an AST node for an assignment operation.
15       * 
16       * @param line
17       *            line in which the assignment operation occurs in the source
18       *            file.
19       * @param operator
20       *            the actual assignment operator.
21       * @param lhs
22       *            the lhs operand.
23       * @param rhs
24       *            the rhs operand.
25       */
26  
27      public JAssignment(int line, String operator, JExpression lhs,
28              JExpression rhs) {
29          super(line, operator, lhs, rhs);
30      }
31  
32  }
33  
34  /**
35   * The AST node for an assignment (=) expression. The = operator has two
36   * operands: a lhs and a rhs.
37   */
38  
39  class JAssignOp extends JAssignment {
40  
41      /**
42       * Constructs the AST node for an assignment (=) expression given the lhs and
43       * rhs operands.
44       * 
45       * @param line
46       *            line in which the assignment expression occurs in the source
47       *            file.
48       * @param lhs
49       *            lhs operand.
50       * @param rhs
51       *            rhs operand.
52       */
53  
54      public JAssignOp(int line, JExpression lhs, JExpression rhs) {
55          super(line, "=", lhs, rhs);
56      }
57  
58      /**
59       * Analyzes the lhs and rhs, checking that types match, and sets the result
60       * type.
61       * 
62       * @param context
63       *            context in which names are resolved.
64       * @return the analyzed (and possibly rewritten) AST subtree.
65       */
66  
67      public JExpression analyze(Context context) {
68          if (!(lhs instanceof JLhs)) {
69              JAST.compilationUnit.reportSemanticError(line(),
70                      "Illegal lhs for assignment");
71          } else {
72              lhs = (JExpression) ((JLhs) lhs).analyzeLhs(context);
73          }
74          rhs = (JExpression) rhs.analyze(context);
75          rhs.type().mustMatchExpected(line(), lhs.type());
76          type = rhs.type();
77          if (lhs instanceof JVariable) {
78              IDefn defn = ((JVariable) lhs).iDefn();
79              if (defn != null) {
80                  // Local variable; consider it to be initialized now.
81                  ((LocalVariableDefn) defn).initialize();
82              }
83          }
84          return this;
85      }
86  
87      /**
88       * Code generation for an assignment involves, generating code for loading
89       * any necessary Lvalue onto the stack, for loading the Rvalue, for (unless
90       * a statement) copying the Rvalue to its proper place on the stack, and for
91       * doing the store.
92       * 
93       * @param output
94       *            the code emitter (basically an abstraction for producing the
95       *            .class file).
96       */
97  
98      public void codegen(CLEmitter output) {
99          ((JLhs) lhs).codegenLoadLhsLvalue(output);
100         rhs.codegen(output);
101         if (!isStatementExpression) {
102             // Generate code to leave the Rvalue atop stack
103             ((JLhs) lhs).codegenDuplicateRvalue(output);
104         }
105         ((JLhs) lhs).codegenStore(output);
106     }
107 
108 }

110 /**
111  * The AST node for a += expression. A += expression has two operands: a lhs and
112  * a rhs.
113  */
114 
115 class JPlusAssignOp extends JAssignment {
116 
117     /**
118      * Constructs the AST node for a += expression given its lhs and rhs
119      * operands.
120      * 
121      * @param line
122      *            line in which the assignment expression occurs in the source
123      *            file.
124      * @param lhs
125      *            the lhs operand.
126      * @param rhs
127      *            the rhs operand.
128      */
129 
130     public JPlusAssignOp(int line, JExpression lhs, JExpression rhs) {
131         super(line, "+=", lhs, rhs);
132     }
133 
134     /**
135      * Analyzes the lhs and rhs, rewrites rhs as lhs + rhs (string concatenation)
136      * if lhs is of type {@code String}, and sets the result type.
137      * 
138      * @param context
139      *            context in which names are resolved.
140      * @return the analyzed (and possibly rewritten) AST subtree.
141      */
142 
143     public JExpression analyze(Context context) {
144         if (!(lhs instanceof JLhs)) {
145             JAST.compilationUnit.reportSemanticError(line(),
146                     "Illegal lhs for assignment");
147             return this;
148         } else {
149             lhs = (JExpression) ((JLhs) lhs).analyzeLhs(context);
150         }
151         rhs = (JExpression) rhs.analyze(context);
152         if (lhs.type().equals(Type.INT)) {
153             rhs.type().mustMatchExpected(line(), Type.INT);
154             type = Type.INT;
155         } else if (lhs.type().equals(Type.STRING)) {
156             rhs = (new JStringConcatenationOp(line, lhs, rhs)).analyze(context);
157             type = Type.STRING;
158         } else {
159             JAST.compilationUnit.reportSemanticError(line(),
160                     "Invalid lhs type for +=: " + lhs.type());
161         }
162         return this;
163     }
164 
165     /**
166      * Code generation for += involves, generating code for loading any
167      * necessary l-value onto the stack, for (unless a string concatenation)
168      * loading the r-value, for (unless a statement) copying the r-value to its
169      * proper place on the stack, and for doing the store.
170      * 
171      * @param output
172      *            the code emitter (basically an abstraction for producing the
173      *            .class file).
174      */
175 
176     public void codegen(CLEmitter output) {
177         ((JLhs) lhs).codegenLoadLhsLvalue(output);
178         if (lhs.type().equals(Type.STRING)) {
179             rhs.codegen(output);
180         } else {
181             ((JLhs) lhs).codegenLoadLhsRvalue(output);
182             rhs.codegen(output);
183             output.addNoArgInstruction(IADD);
184         }
185         if (!isStatementExpression) {
186             // Generate code to leave the r-value atop stack
187             ((JLhs) lhs).codegenDuplicateRvalue(output);
188         }
189         ((JLhs) lhs).codegenStore(output);
190     }
191 
192 }
