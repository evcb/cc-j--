3   package jminusminus;
4   
5   /**
6    * JAST is the abstract superclass of all nodes in the abstract syntax tree
7    * (AST).
8    */
9   
10  abstract class JAST {
11  
12      /** Current compilation unit (set in JCompilationUnit()). */
13      public static JCompilationUnit compilationUnit;
14  
15      /** Line in which the source for the AST was found. */
16      protected int line;
17  
18      /**
19       * Constructs an AST node the given its line number in the source file.
20       * 
21       * @param line
22       *            line in which the source for the AST was found.
23       */
24  
25      protected JAST(int line) {
26          this.line = line;
27      }
28  
29      /**
30       * Returns the line in which the source for the AST was found.
31       * 
32       * @return the line number.
33       */
34  
35      public int line() {
36          return line;
37      }
38  
39      /**
40       * Performs semantic analysis on this AST. In some instances a new returned
41       * AST reflects surgery.
42       * 
43       * @param context
44       *            the environment (scope) in which code is analyzed.
45       * @return a (rarely modified) AST.
46       */
47  
48      public abstract JAST analyze(Context context);
49  
50      /**
51       * Generates a partial class for this type, reflecting only the member
52       * information required to do analysis.
53       * 
54       * @param context
55       *            the parent (class) context.
56       * @param partial
57       *            the code emitter (basically an abstraction for producing the
58       *            partial class).
59       */
60  
61      public void partialCodegen(Context context, CLEmitter partial) {
62          // A dummy -- redefined where necessary.
63      }
64  
65      /**
66       * Performs code generation for this AST.
67       * 
68       * @param output
69       *            the code emitter (basically an abstraction for producing the
70       *            .class file).
71       */
72  
73      public abstract void codegen(CLEmitter output);
74  
75      /**
76       * Writes the information pertaining to this AST to STDOUT.
77       * 
78       * @param p
79       *            for pretty printing with indentation.
80       */
81  
82      public abstract void writeToStdOut(PrettyPrinter p);
83  
84  }
