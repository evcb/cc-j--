5   /**
6    * The AST node for an expression. The syntax says all expressions are
7    * statements, but a semantic check throws some (those without a side-effect)
8    * out.
9    * <p>
10   * Every expression has a type and a flag saying whether or not it's a
11   * statement-expression.
12   */
13  
14  abstract class JExpression extends JStatement {
15  
16      /** Expression type. */
17      protected Type type;
18  
19      /** Whether or not this expression is a statement. */
20      protected boolean isStatementExpression;
21  
22      /**
23       * Constructs an AST node for an expression given its line number.
24       * 
25       * @param line
26       *            line in which the expression occurs in the source file.
27       */
28  
29      protected JExpression(int line) {
30          super(line);
31          isStatementExpression = false; // by default
32      }
33  
34      /**
35       * Returns the expression type.
36       * 
37       * @return the expression type.
38       */
39  
40      public Type type() {
41          return type;
42      }
43  
44      /**
45       * Is this a statement expression?
46       * 
47       * @return {@code true} if this is being used as a statement; 
48       *         {@code false} otherwise.
49       */
50  
51      public boolean isStatementExpression() {
52          return isStatementExpression;
53      }
54  
55      /**
56       * The analysis of any JExpression returns a JExpression. That's all this
57       * (re-)declaration of {@code analyze} method says.
58       * 
59       * @param context
60       *            context in which names are resolved.
61       * @return the analyzed (and possibly rewritten) AST subtree.
62       */
63  
64      public abstract JExpression analyze(Context context);
65  
66      /**
67       * Performs (short-circuit) code generation for a boolean expression, given
68       * the code emitter, a target label, and whether we branch to that label on
69       * true or on false.
70       * 
71       * @param output
72       *            the code emitter (basically an abstraction for producing the
73       *            .class file).
74       * @param targetLabel
75       *            the label to which we should branch.
76       * @param onTrue
77       *            do we branch on true?
78       */
79  
80      public void codegen(CLEmitter output, String targetLabel, boolean onTrue) {
81          // We should never reach here, i.e., all boolean
82          // (including
83          // identifier) expressions must override this method.
84          System.err.println("Error in code generation");
85      }
86  
87  }
