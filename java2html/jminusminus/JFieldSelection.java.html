5   import static jminusminus.CLConstants.*;

7   /**
8    * The AST node for a field selection operation. It has a target object, a field
9    * name, and the {@link Field} it defines.
10   */
11  
12  class JFieldSelection extends JExpression implements JLhs {
13  
14      /** The target expression. */
15      protected JExpression target;
16  
17      /** The ambiguous part that is reclassified in analyze(). */
18      private AmbiguousName ambiguousPart;
19  
20      /** The field name. */
21      private String fieldName;
22  
23      /** The Field representing this field. */
24      private Field field;
25  
26      /**
27       * Constructs an AST node for a field selection without an ambiguous part.
28       * 
29       * @param line
30       *            the line number of the selection.
31       * @param target
32       *            the target of the selection.
33       * @param fieldName
34       *            the field name.
35       */
36  
37      public JFieldSelection(int line, JExpression target, String fieldName) {
38          this(line, null, target, fieldName);
39      }
40  
41      /**
42       * Construct an AST node for a field selection having an ambiguous part.
43       * 
44       * @param line
45       *            line in which the field selection occurs in the source file.
46       * @param ambiguousPart
47       *            the ambiguous part.
48       * @param target
49       *            the target of the selection.
50       * @param fieldName
51       *            the field name.
52       */
53  
54      public JFieldSelection(int line, AmbiguousName ambiguousPart,
55              JExpression target, String fieldName) {
56          super(line);
57          this.ambiguousPart = ambiguousPart;
58          this.target = target;
59          this.fieldName = fieldName;
60      }
61  
62      /**
63       * Analyzing a field selection expression involves, (1) reclassifying any
64       * ambiguous part, (2) analyzing the target, (3) treating "length" field of
65       * arrays specially, or computing the Field object, (4) checking the access
66       * rules, and (5) computing the resultant type.
67       * 
68       * @param context
69       *            context in which names are resolved.
70       * @return the analyzed (and possibly rewritten) AST subtree.
71       */
72  
73      public JExpression analyze(Context context) {
74          // Reclassify the ambiguous part.
75          if (ambiguousPart != null) {
76              JExpression expr = ambiguousPart.reclassify(context);
77              if (expr != null) {
78                  if (target == null)
79                      target = expr;
80                  else {
81                      // Can't even happen syntactically
82                      JAST.compilationUnit.reportSemanticError(line(),
83                              "Badly formed suffix");
84                  }
85              }
86          }
87          target = (JExpression) target.analyze(context);
88          Type targetType = target.type();
89  
90          // We use a workaround for the "length" field of arrays.
91          if ((targetType.isArray()) && fieldName.equals("length")) {
92              type = Type.INT;
93          } else {
94              // Other than that, targetType has to be a
95              // ReferenceType
96              if (targetType.isPrimitive()) {
97                  JAST.compilationUnit.reportSemanticError(line(),
98                          "Target of a field selection must "
99                                  + "be a defined type");
100                 type = Type.ANY;
101                 return this;
102             }
103             field = targetType.fieldFor(fieldName);
104             if (field == null) {
105                 JAST.compilationUnit.reportSemanticError(line(),
106                         "Cannot find a field: " + fieldName);
107                 type = Type.ANY;
108             } else {
109                 context.definingType().checkAccess(line, (Member) field);
110                 type = field.type();
111 
112                 // Non-static field cannot be referenced from a static context.
113                 if (!field.isStatic()) {
114                     if (target instanceof JVariable && 
115                        ((JVariable) target).iDefn() instanceof TypeNameDefn) {
116                         JAST.compilationUnit.reportSemanticError(line(),
117                                     "Non-static field " + fieldName +
118                                 " cannot be referenced from a static context");
119                     }
120                 }
121             }
122         }
123         return this;
124     }
125 
126     /**
127      * Analyzes the field selection expression for use on the lhs of an
128      * assignment. Although the {@code final} keyword is not in j--, we do 
129      * make use of the Java API and so must respect its constraints.
130      * 
131      * @param context
132      *            context in which names are resolved.
133      * @return the analyzed (and possibly rewritten) AST subtree.
134      */
135 
136     public JExpression analyzeLhs(Context context) {
137         JExpression result = analyze(context);
138         if (field.isFinal()) {
139             JAST.compilationUnit.reportSemanticError(line, "The field "
140                     + fieldName + " in type " + target.type.toString()
141                     + " is declared final.");
142         }
143         return result;
144     }
145 
146     /**
147      * Generates the code necessary to load the Rvalue for this field selection.
148      * 
149      * @param output
150      *            the code emitter (basically an abstraction for producing the
151      *            .class file).
152      */
153 
154     public void codegen(CLEmitter output) {
155         target.codegen(output);
156 
157         // We use a workaround for the "length" field of arrays
158         if ((target.type().isArray()) && fieldName.equals("length")) {
159             output.addNoArgInstruction(ARRAYLENGTH);
160         } else {
161             int mnemonic = field.isStatic() ? GETSTATIC : GETFIELD;
162             output.addMemberAccessInstruction(mnemonic,
163                     target.type().jvmName(), fieldName, type.toDescriptor());
164         }
165     }
166 
167     /**
168      * The semantics of j-- requires that we implement short-circuit branching
169      * in implementing field selections.
170      * 
171      * @param output
172      *            the code emitter (basically an abstraction for producing the
173      *            .class file).
174      * @param targetLabel
175      *            the label to which we should branch.
176      * @param onTrue
177      *            do we branch on true?
178      */
179 
180     public void codegen(CLEmitter output, String targetLabel, boolean onTrue) {
181         // Push the value
182         codegen(output);
183 
184         if (onTrue) {
185             // Branch on true
186             output.addBranchInstruction(IFNE, targetLabel);
187         } else {
188             // Branch on false
189             output.addBranchInstruction(IFEQ, targetLabel);
190         }
191     }
192 
193     /**
194      * Generates the code required for setting up an Lvalue, for example, for use
195      * in an assignment.
196      * 
197      * @param output
198      *            the code emitter (basically an abstraction for producing the
199      *            .class file).
200      */
201 
202     public void codegenLoadLhsLvalue(CLEmitter output) {
203         // Nothing to do for static fields.
204         if (!field.isStatic()) {
205             // Just load the target
206             target.codegen(output);
207         }
208     }
209 
210     /**
211      * Generates the code required for loading an Rvalue for this variable, for 
212      * example, for use in a {@code +=}. Here, this requires either a 
213      * {@code getstatic} or {@code getfield}.
214      * 
215      * @param output
216      *            the code emitter (basically an abstraction for producing the
217      *            .class file).
218      */
219 
220     public void codegenLoadLhsRvalue(CLEmitter output) {
221         String descriptor = field.type().toDescriptor();
222         if (field.isStatic()) {
223             output.addMemberAccessInstruction(GETSTATIC, target.type()
224                     .jvmName(), fieldName, descriptor);
225         } else {
226             output.addNoArgInstruction(type == Type.STRING ? DUP_X1 : DUP);
227             output.addMemberAccessInstruction(GETFIELD,
228                     target.type().jvmName(), fieldName, descriptor);
229         }
230     }
231 
232     /**
233      * Generates the code required for duplicating the Rvalue that is on the
234      * stack because it is to be used in a surrounding expression, as in 
235      * {@code a[i] = x =} &lt;expr&gt; or {@code x = y--}. 
236      * Here this means copying it down.
237      * 
238      * @param output
239      *            the code emitter (basically an abstraction for producing the
240      *            .class file).
241      */
242 
243     public void codegenDuplicateRvalue(CLEmitter output) {
244         if (field.isStatic()) {
245             output.addNoArgInstruction(DUP);
246         } else {
247             output.addNoArgInstruction(DUP_X1);
248         }
249     }
250 
251     /**
252      * Generates the code required for doing the actual assignment.
253      * 
254      * @param output
255      *            the code emitter (basically an abstraction for producing the
256      *            .class file).
257      */
258 
259     public void codegenStore(CLEmitter output) {
260         String descriptor = field.type().toDescriptor();
261         if (field.isStatic()) {
262             output.addMemberAccessInstruction(PUTSTATIC, target.type()
263                     .jvmName(), fieldName, descriptor);
264         } else {
265             output.addMemberAccessInstruction(PUTFIELD,
266                     target.type().jvmName(), fieldName, descriptor);
267         }
268     }
269 
270     /**
271      * {@inheritDoc}
272      */
273 
274     public void writeToStdOut(PrettyPrinter p) {
275         p.printf("<JFieldSelection line=\"%d\" name=\"%s\"/>\n", line(),
276                 fieldName);
277         p.indentRight();
278         if (target != null) {
279             p.println("<Target>");
280             p.indentRight();
281             target.writeToStdOut(p);
282             p.indentLeft();
283             p.println("</Target>");
284         }
285         p.indentLeft();
286         p.println("</JFieldSelection>");
287     }
