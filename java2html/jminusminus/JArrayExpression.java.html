3   package jminusminus;
4   
5   import static jminusminus.CLConstants.*;
6   
7   /**
8    * The AST for an array indexing operation. It has an expression
9    * denoting an array object and an expression denoting an integer
10   * index.
11   */
12  
13  class JArrayExpression extends JExpression implements JLhs {
14  
15      /** The array. */
16      private JExpression theArray;
17  
18      /** The array index expression. */
19      private JExpression indexExpr;
20  
21      /**
22       * Constructs an AST node for an array indexing operation.
23       * 
24       * @param line
25       *            line in which the operation occurs in the
26       *            source file.
27       * @param theArray
28       *            the array we're indexing.
29       * @param indexExpr
30       *            the index expression.
31       */
32  
33      public JArrayExpression(int line, JExpression theArray,
34          JExpression indexExpr) {
35          super(line);
36          this.theArray = theArray;
37          this.indexExpr = indexExpr;
38      }
39  
40      /**
41       * Performs semantic analysis on an array indexing expression
42       * such as {@code A[i]}.
43       * 
44       * @param context
45       *            context in which names are resolved.
46       * @return the analyzed (and possibly rewritten) AST subtree.
47       */
48  
49      public JExpression analyze(Context context) {
50          theArray = (JExpression) theArray.analyze(context);
51          indexExpr = (JExpression) indexExpr.analyze(context);
52          if (!(theArray.type().isArray())) {
53              JAST.compilationUnit.reportSemanticError(line(),
54                  "attempt to index a non-array object");
55              this.type = Type.ANY;
56          } else {
57              this.type = theArray.type().componentType();
58          }
59          indexExpr.type().mustMatchExpected(line(), Type.INT);
60          return this;
61      }
62  
63      /**
64       * Analyzing the array expression as an Lvalue is like
65       * analyzing it for its Rvalue.
66       * 
67       * @param context
68       *            context in which names are resolved.
69       */
70  
71      public JExpression analyzeLhs(Context context) {
72          analyze(context);
73          return this;
74      }
75  
76      /**
77       * Performs code generation for this {@code JArrayExpression} using
78       * the specified code emitter. Generates the code necessary for 
79       * loading the Rvalue.
80       * 
81       * @param output
82       *            the code emitter (basically an abstraction
83       *            for producing the .class file).
84       */
85  
86      public void codegen(CLEmitter output) {
87          theArray.codegen(output);
88          indexExpr.codegen(output);
89          if (type == Type.INT) {
90              output.addNoArgInstruction(IALOAD);
91          } else if (type == Type.BOOLEAN) {
92              output.addNoArgInstruction(BALOAD);
93          } else if (type == Type.CHAR) {
94              output.addNoArgInstruction(CALOAD);
95          } else if (!type.isPrimitive()) {
96              output.addNoArgInstruction(AALOAD);
97          }
98      }
99  
100     /**
101      * Generates the code required for setting up an Lvalue, for example,
102      * for use in an assignment. Here, this requires loading the
103      * array and the index.
104      * 
105      * @param output
106      *            the code emitter (basically an abstraction
107      *            for producing the .class file).
108      */
109 
110     public void codegenLoadLhsLvalue(CLEmitter output) {
111         // Load the lvalue onto the stack: the array and the
112         // index.
113         theArray.codegen(output);
114         indexExpr.codegen(output);
115     }
116 
117     /**
118      * Generates the code required for loading an Rvalue for this
119      * variable, for example, for use in a {@code +=}. Here, this 
120      * requires duplicating the array and the index on the stack and 
121      * doing an array load.
122      * 
123      * @param output
124      *                the code emitter (basically an abstraction
125      *                for producing the .class file).
126      */
127 
128     public void codegenLoadLhsRvalue(CLEmitter output) {
129         // Load rvalue onto stack, by duplicating the lvalue,
130         // and fetching it's content
131         if (type == Type.STRING) {
132             output.addNoArgInstruction(DUP2_X1);
133         } else {
134             output.addNoArgInstruction(DUP2);
135         }
136         if (type == Type.INT) {
137             output.addNoArgInstruction(IALOAD);
138         } else if (type == Type.BOOLEAN) {
139             output.addNoArgInstruction(BALOAD);
140         } else if (type == Type.CHAR) {
141             output.addNoArgInstruction(CALOAD);
142         } else if (!type.isPrimitive()) {
143             output.addNoArgInstruction(AALOAD);
144         }
145     }
146 
147     /**
148      * Generates the code required for duplicating the Rvalue that
149      * is on the stack because it is to be used in a surrounding
150      * expression, as in {@code a[i] = x =} &lt;expr&gt; or 
151      * {@code x = y--}. Here this means copying it down two locations 
152      * (beneath the array and index).
153      * 
154      * @param output
155      *            the code emitter (basically an abstraction
156      *            for producing the .class file).
157      */
158 
159     public void codegenDuplicateRvalue(CLEmitter output) {
160         // It's copied down below the array and index
161         output.addNoArgInstruction(DUP_X2);
162     }
163 
164     /**
165      * Generates the code required for doing the actual
166      * assignment. Here, this requires an array store.
167      * 
168      * @param output
169      *            the code emitter (basically an abstraction
170      *            for producing the .class file).
171      */
172 
173     public void codegenStore(CLEmitter output) {
174         if (type == Type.INT) {
175             output.addNoArgInstruction(IASTORE);
176         } else if (type == Type.BOOLEAN) {
177             output.addNoArgInstruction(BASTORE);
178         } else if (type == Type.CHAR) {
179             output.addNoArgInstruction(CASTORE);
180         } else if (!type.isPrimitive()) {
181             output.addNoArgInstruction(AASTORE);
182         }
183 
184     }
185 
186     /**
187      * {@inheritDoc}
188      */
189 
190     public void writeToStdOut(PrettyPrinter p) {
191         p.println("<JArrayExpression>");
192         p.indentRight();
193         if (theArray != null) {
194             p.println("<TheArray>");
195             p.indentRight();
196             theArray.writeToStdOut(p);
197             p.indentLeft();
198             p.println("</TheArray>");
199         }
200         if (indexExpr != null) {
201             p.println("<IndexExpression>");
202             p.indentRight();
203             indexExpr.writeToStdOut(p);
204             p.indentLeft();
205             p.println("</IndexExpression>");
206         }
207         p.indentLeft();
208         p.println("</JArrayExpression>");
209     }
210 }
