9   /**
10   * A recursive descent parser that, given a lexical analyzer (a
11   * {@link LookaheadScanner}), parses a Java compilation unit (program file),
12   * taking tokens from the LookaheadScanner, and produces an abstract syntax 
13   * tree (AST) for it.
14   * <p>
15   * See Appendix C.2.2 in the textbook or the
16   * <a href="https://docs.oracle.com/javase/specs/jls/se8/html/index.html">Java Language Specifications</a>
17   * for the full syntactic grammar.
18   */
19  
20  public class Parser {
21  
22      /** The lexical analyzer with which tokens are scanned. */
23      private LookaheadScanner scanner;
24  
25      /** Whether a parser error has been found. */
26      private boolean isInError;
27  
28      /** Whether we have recovered from a parser error. */
29      private boolean isRecovered;
30  
31      /**
32       * Constructs a parser from the given lexical analyzer.
33       * 
34       * @param scanner
35       *            the lexical analyzer with which tokens are scanned.
36       */
37  
38      public Parser(LookaheadScanner scanner) {
39          this.scanner = scanner;
40          isInError = false;
41          isRecovered = true;
42          scanner.next(); // Prime the pump
43      }
44  
45      /**
46       * Has a parser error occurred up to now?
47       * 
48       * @return {@code true} if a parser error occurred; {@code false} otherwise.
49       */
50  
51      public boolean errorHasOccurred() {
52          return isInError;
53      }

59      /**
60       * Is the current token this one?
61       * 
62       * @param sought
63       *            the token we're looking for.
64       * @return true iff they match; false otherwise.
65       */
66  
67      private boolean see(TokenKind sought) {
68          return (sought == scanner.token().kind());
69      }
70  
71      /**
72       * Look at the current (unscanned) token to see if it's one we're looking
73       * for. If so, scan it and return true; otherwise return false (without
74       * scanning a thing).
75       * 
76       * @param sought
77       *            the token we're looking for.
78       * @return true iff they match; false otherwise.
79       */
80  
81      private boolean have(TokenKind sought) {
82          if (see(sought)) {
83              scanner.next();
84              return true;
85          } else {
86              return false;
87          }
88      }
89  
90      /**
91       * Attempt to match a token we're looking for with the current input token.
92       * If we succeed, scan the token and go into a "isRecovered" state. If we
93       * fail, then what we do next depends on whether or not we're currently in a
94       * "isRecovered" state: if so, we report the error and go into an
95       * "Unrecovered" state; if not, we repeatedly scan tokens until we find the
96       * one we're looking for (or EOF) and then return to a "isRecovered" state.
97       * This gives us a kind of poor man's syntactic error recovery. The strategy
98       * is due to David Turner and Ron Morrison.
99       * 
100      * @param sought
101      *            the token we're looking for.
102      */
103 
104     private void mustBe(TokenKind sought) {
105         if (scanner.token().kind() == sought) {
106             scanner.next();
107             isRecovered = true;
108         } else if (isRecovered) {
109             isRecovered = false;
110             reportParserError("%s found where %s sought", scanner.token()
111                     .image(), sought.image());
112         } else {
113             // Do not report the (possibly spurious) error,
114             // but rather attempt to recover by forcing a match.
115             while (!see(sought) && !see(EOF)) {
116                 scanner.next();
117             }
118             if (see(sought)) {
119                 scanner.next();
120                 isRecovered = true;
121             }
122         }
123     }
124 
125     /**
126      * Pull out the ambiguous part of a name and return it.
127      * 
128      * @param name
129      *            with an ambiguos part (possibly).
130      * @return ambiguous part or null.
131      */
132 
133     private AmbiguousName ambiguousPart(TypeName name) {
134         String qualifiedName = name.toString();
135         int lastDotIndex = qualifiedName.lastIndexOf('.');
136         return lastDotIndex == -1 ? null // It was a simple
137                 // name
138                 : new AmbiguousName(name.line(), qualifiedName.substring(0,
139                         lastDotIndex));
140     }

142     /**
143      * Report a syntax error.
144      * 
145      * @param message
146      *            message identifying the error.
147      * @param args
148      *            related values.
149      */
150 
151     private void reportParserError(String message, Object... args) {
152         isInError = true;
153         isRecovered = false;
154         System.err
155                 .printf("%s:%d: ", scanner.fileName(), scanner.token().line());
156         System.err.printf(message, args);
157         System.err.println();
158     }

164     /**
165      * Are we looking at an IDENTIFIER followed by a LPAREN? Look ahead to find
166      * out.
167      * 
168      * @return true iff we're looking at IDENTIFIER LPAREN; false otherwise.
169      */
170 
171     private boolean seeIdentLParen() {
172         scanner.recordPosition();
173         boolean result = have(IDENTIFIER) && see(LPAREN);
174         scanner.returnToPosition();
175         return result;
176     }
177 
178     /**
179      * Are we looking at a cast? ie.
180      * 
181      * <pre>
182      *   LPAREN type RPAREN ...
183      * </pre>
184      * 
185      * Look ahead to find out.
186      * 
187      * @return true iff we're looking at a cast; false otherwise.
188      */
189 
190     private boolean seeCast() {
191         scanner.recordPosition();
192         if (!have(LPAREN)) {
193             scanner.returnToPosition();
194             return false;
195         }
196         if (seeBasicType()) {
197             scanner.returnToPosition();
198             return true;
199         }
200         if (!see(IDENTIFIER)) {
201             scanner.returnToPosition();
202             return false;
203         } else {
204             scanner.next(); // Scan the IDENTIFIER
205             // A qualified identifier is ok
206             while (have(DOT)) {
207                 if (!have(IDENTIFIER)) {
208                     scanner.returnToPosition();
209                     return false;
210                 }
211             }
212         }
213         while (have(LBRACK)) {
214             if (!have(RBRACK)) {
215                 scanner.returnToPosition();
216                 return false;
217             }
218         }
219         if (!have(RPAREN)) {
220             scanner.returnToPosition();
221             return false;
222         }
223         scanner.returnToPosition();
224         return true;
225     }
226 
227     /**
228      * Are we looking at a local variable declaration? ie.
229      * 
230      * <pre>
231      *   type IDENTIFIER {LBRACK RBRACK} ...
232      * </pre>
233      * 
234      * Look ahead to determine.
235      * 
236      * @return true iff we are looking at local variable declaration; false
237      *         otherwise.
238      */
239 
240     private boolean seeLocalVariableDeclaration() {
241         scanner.recordPosition();
242         if (have(IDENTIFIER)) {
243             // A qualified identifier is ok
244             while (have(DOT)) {
245                 if (!have(IDENTIFIER)) {
246                     scanner.returnToPosition();
247                     return false;
248                 }
249             }
250         } else if (seeBasicType()) {
251             scanner.next();
252         } else {
253             scanner.returnToPosition();
254             return false;
255         }
256         while (have(LBRACK)) {
257             if (!have(RBRACK)) {
258                 scanner.returnToPosition();
259                 return false;
260             }
261         }
262         if (!have(IDENTIFIER)) {
263             scanner.returnToPosition();
264             return false;
265         }
266         while (have(LBRACK)) {
267             if (!have(RBRACK)) {
268                 scanner.returnToPosition();
269                 return false;
270             }
271         }
272         scanner.returnToPosition();
273         return true;
274     }
275 
276     /**
277      * Are we looking at a basic type? ie.
278      * 
279      * <pre>
280      * BOOLEAN | CHAR | INT
281      * </pre>
282      * 
283      * @return true iff we're looking at a basic type; false otherwise.
284      */
285 
286     private boolean seeBasicType() {
287         return (see(BOOLEAN) || see(CHAR) || see(INT));
288     }
289 
290     /**
291      * Are we looking at a reference type? ie.
292      * 
293      * <pre>
294      *   referenceType ::= basicType LBRACK RBRACK {LBRACK RBRACK}
295      *                   | qualifiedIdentifier {LBRACK RBRACK}
296      * </pre>
297      * 
298      * @return true iff we're looking at a reference type; false otherwise.
299      */
300 
301     private boolean seeReferenceType() {
302         if (see(IDENTIFIER)) {
303             return true;
304         } else {
305             scanner.recordPosition();
306             if (have(BOOLEAN) || have(CHAR) || have(INT)) {
307                 if (have(LBRACK) && see(RBRACK)) {
308                     scanner.returnToPosition();
309                     return true;
310                 }
311             }
312             scanner.returnToPosition();
313         }
314         return false;
315     }
316 
317     /**
318      * Are we looking at []?
319      * 
320      * @return true iff we're looking at a [] pair; false otherwise.
321      */
322 
323     private boolean seeDims() {
324         scanner.recordPosition();
325         boolean result = have(LBRACK) && see(RBRACK);
326         scanner.returnToPosition();
327         return result;
328     }

334     /**
335      * Parses a compilation unit (a program file) and constructs an AST for it.
336      * After constructing the Parser, this is its entry point.
337      * 
338      * <pre>
339      *   compilationUnit ::= [PACKAGE qualifiedIdentifier SEMI]
340      *                       {IMPORT  qualifiedIdentifier SEMI}
341      *                       {typeDeclaration}
342      *                       EOF
343      * </pre>
344      * 
345      * @return an AST for a compilationUnit.
346      */
347 
348     public JCompilationUnit compilationUnit() {
349         int line = scanner.token().line();
350         TypeName packageName = null; // Default
351         if (have(PACKAGE)) {
352             packageName = qualifiedIdentifier();
353             mustBe(SEMI);
354         }
355         ArrayList<TypeName> imports = new ArrayList<TypeName>();
356         while (have(IMPORT)) {
357             imports.add(qualifiedIdentifier());
358             mustBe(SEMI);
359         }
360         ArrayList<JAST> typeDeclarations = new ArrayList<JAST>();
361         while (!see(EOF)) {
362             JAST typeDeclaration = typeDeclaration();
363             if (typeDeclaration != null) {
364                 typeDeclarations.add(typeDeclaration);
365             }
366         }
367         mustBe(EOF);
368         return new JCompilationUnit(scanner.fileName(),
369                                     line, packageName,
370                                     imports, typeDeclarations);
371     }
372 
373     /**
374      * Parse a qualified identifier.
375      * 
376      * <pre>
377      *   qualifiedIdentifier ::= IDENTIFIER {DOT IDENTIFIER}
378      * </pre>
379      * 
380      * @return an instance of TypeName.
381      */
382 
383     private TypeName qualifiedIdentifier() {
384         int line = scanner.token().line();
385         mustBe(IDENTIFIER);
386         String qualifiedIdentifier = scanner.previousToken().image();
387         while (have(DOT)) {
388             mustBe(IDENTIFIER);
389             qualifiedIdentifier += "." + scanner.previousToken().image();
390         }
391         return new TypeName(line, qualifiedIdentifier);
392     }

394     /**
395      * Parse a type declaration.
396      * 
397      * <pre>
398      *   typeDeclaration ::= modifiers classDeclaration
399      * </pre>
400      * 
401      * @return an AST for a typeDeclaration.
402      */
403 
404     private JAST typeDeclaration() {
405         ArrayList<String> mods = modifiers();
406         return classDeclaration(mods);
407     }
408 
409     /**
410      * Parse modifiers.
411      * 
412      * <pre>
413      *   modifiers ::= {PUBLIC | PROTECTED | PRIVATE | STATIC | 
414      *                  ABSTRACT}
415      * </pre>
416      * 
417      * Check for duplicates, and conflicts among access modifiers (public,
418      * protected, and private). Otherwise, no checks.
419      * 
420      * @return a list of modifiers.
421      */
422 
423     private ArrayList<String> modifiers() {
424         ArrayList<String> mods = new ArrayList<String>();
425         boolean scannedPUBLIC = false;
426         boolean scannedPROTECTED = false;
427         boolean scannedPRIVATE = false;
428         boolean scannedSTATIC = false;
429         boolean scannedABSTRACT = false;
430         boolean more = true;
431         while (more)
432             if (have(PUBLIC)) {
433                 mods.add("public");
434                 if (scannedPUBLIC) {
435                     reportParserError("Repeated modifier: public");
436                 }
437                 if (scannedPROTECTED || scannedPRIVATE) {
438                     reportParserError("Access conflict in modifiers");
439                 }
440                 scannedPUBLIC = true;
441             } else if (have(PROTECTED)) {
442                 mods.add("protected");
443                 if (scannedPROTECTED) {
444                     reportParserError("Repeated modifier: protected");
445                 }
446                 if (scannedPUBLIC || scannedPRIVATE) {
447                     reportParserError("Access conflict in modifiers");
448                 }
449                 scannedPROTECTED = true;
450             } else if (have(PRIVATE)) {
451                 mods.add("private");
452                 if (scannedPRIVATE) {
453                     reportParserError("Repeated modifier: private");
454                 }
455                 if (scannedPUBLIC || scannedPROTECTED) {
456                     reportParserError("Access conflict in modifiers");
457                 }
458                 scannedPRIVATE = true;
459             } else if (have(STATIC)) {
460                 mods.add("static");
461                 if (scannedSTATIC) {
462                     reportParserError("Repeated modifier: static");
463                 }
464                 scannedSTATIC = true;
465             } else if (have(ABSTRACT)) {
466                 mods.add("abstract");
467                 if (scannedABSTRACT) {
468                     reportParserError("Repeated modifier: abstract");
469                 }
470                 scannedABSTRACT = true;
471             } else {
472                 more = false;
473             }
474         return mods;
475     }

477     /**
478      * Parse a class declaration.
479      * 
480      * <pre>
481      *   classDeclaration ::= CLASS IDENTIFIER 
482      *                        [EXTENDS qualifiedIdentifier] 
483      *                        classBody
484      * </pre>
485      * 
486      * A class which doesn't explicitly extend another (super) class implicitly
487      * extends the superclass java.lang.Object.
488      * 
489      * @param mods
490      *            the class modifiers.
491      * @return an AST for a classDeclaration.
492      */
493 
494     private JClassDeclaration classDeclaration(ArrayList<String> mods) {
495         int line = scanner.token().line();
496         mustBe(CLASS);
497         mustBe(IDENTIFIER);
498         String name = scanner.previousToken().image();
499         Type superClass;
500         if (have(EXTENDS)) {
501             superClass = qualifiedIdentifier();
502         } else {
503             superClass = Type.OBJECT;
504         }
505         return new JClassDeclaration(line, mods, name, superClass, classBody());
506     }
507 
508     /**
509      * Parse a class body.
510      * 
511      * <pre>
512      *   classBody ::= LCURLY
513      *                   {modifiers memberDecl}
514      *                 RCURLY
515      * </pre>
516      * 
517      * @return list of members in the class body.
518      */
519 
520     private ArrayList<JMember> classBody() {
521         ArrayList<JMember> members = new ArrayList<JMember>();
522         mustBe(LCURLY);
523         while (!see(RCURLY) && !see(EOF)) {
524             members.add(memberDecl(modifiers()));
525         }
526         mustBe(RCURLY);
527         return members;
528     }

530     /**
531      * Parse a member declaration.
532      * 
533      * <pre>
534      *   memberDecl ::= IDENTIFIER            // constructor
535      *                    formalParameters
536      *                    block
537      *                | (VOID | type) IDENTIFIER  // method
538      *                    formalParameters
539      *                    (block | SEMI)
540      *                | type variableDeclarators SEMI
541      * </pre>
542      * 
543      * @param mods
544      *            the class member modifiers.
545      * @return an AST for a memberDecl.
546      */
547 
548     private JMember memberDecl(ArrayList<String> mods) {
549         int line = scanner.token().line();
550         JMember memberDecl = null;
551         if (seeIdentLParen()) {
552             // A constructor
553             mustBe(IDENTIFIER);
554             String name = scanner.previousToken().image();
555             ArrayList<JFormalParameter> params = formalParameters();
556             JBlock body = block();
557             memberDecl = new JConstructorDeclaration(line, mods, name, params,
558                     body);
559         } else {
560             Type type = null;
561             if (have(VOID)) {
562                 // void method
563                 type = Type.VOID;
564                 mustBe(IDENTIFIER);
565                 String name = scanner.previousToken().image();
566                 ArrayList<JFormalParameter> params = formalParameters();
567                 JBlock body = have(SEMI) ? null : block();
568                 memberDecl = new JMethodDeclaration(line, mods, name, type,
569                         params, body);
570             } else {
571                 type = type();
572                 if (seeIdentLParen()) {
573                     // Non void method
574                     mustBe(IDENTIFIER);
575                     String name = scanner.previousToken().image();
576                     ArrayList<JFormalParameter> params = formalParameters();
577                     JBlock body = have(SEMI) ? null : block();
578                     memberDecl = new JMethodDeclaration(line, mods, name, type,
579                             params, body);
580                 } else {
581                     // Field
582                     memberDecl = new JFieldDeclaration(line, mods,
583                             variableDeclarators(type));
584                     mustBe(SEMI);
585                 }
586             }
587         }
588         return memberDecl;
589     }

591     /**
592      * Parse a block.
593      * 
594      * <pre>
595      *   block ::= LCURLY {blockStatement} RCURLY
596      * </pre>
597      * 
598      * @return an AST for a block.
599      */
600 
601     private JBlock block() {
602         int line = scanner.token().line();
603         ArrayList<JStatement> statements = new ArrayList<JStatement>();
604         mustBe(LCURLY);
605         while (!see(RCURLY) && !see(EOF)) {
606             statements.add(blockStatement());
607         }
608         mustBe(RCURLY);
609         return new JBlock(line, statements);
610     }

612     /**
613      * Parse a block statement.
614      * 
615      * <pre>
616      *   blockStatement ::= localVariableDeclarationStatement
617      *                    | statement
618      * </pre>
619      * 
620      * @return an AST for a blockStatement.
621      */
622 
623     private JStatement blockStatement() {
624         if (seeLocalVariableDeclaration()) {
625             return localVariableDeclarationStatement();
626         } else {
627             return statement();
628         }
629     }

631     /**
632      * Parse a statement.
633      * 
634      * <pre>
635      *   statement ::= block
636      *               | IF parExpression statement [ELSE statement]
637      *               | WHILE parExpression statement 
638      *               | RETURN [expression] SEMI
639      *               | SEMI 
640      *               | statementExpression SEMI
641      * </pre>
642      * 
643      * @return an AST for a statement.
644      */
645 
646     private JStatement statement() {
647         int line = scanner.token().line();
648         if (see(LCURLY)) {
649             return block();
650         } else if (have(IF)) {
651             JExpression test = parExpression();
652             JStatement consequent = statement();
653             JStatement alternate = have(ELSE) ? statement() : null;
654             return new JIfStatement(line, test, consequent, alternate);
655         } else if (have(WHILE)) {
656             JExpression test = parExpression();
657             JStatement statement = statement();
658             return new JWhileStatement(line, test, statement);
659         } else if (have(RETURN)) {
660             if (have(SEMI)) {
661                 return new JReturnStatement(line, null);
662             } else {
663                 JExpression expr = expression();
664                 mustBe(SEMI);
665                 return new JReturnStatement(line, expr);
666             }
667         } else if (have(SEMI)) {
668             return new JEmptyStatement(line);
669         } else { // Must be a statementExpression
670             JStatement statement = statementExpression();
671             mustBe(SEMI);
672             return statement;
673         }
674     }

676     /**
677      * Parse formal parameters.
678      * 
679      * <pre>
680      *   formalParameters ::= LPAREN 
681      *                          [formalParameter 
682      *                            {COMMA  formalParameter}]
683      *                        RPAREN
684      * </pre>
685      * 
686      * @return a list of formal parameters.
687      */
688 
689     private ArrayList<JFormalParameter> formalParameters() {
690         ArrayList<JFormalParameter> parameters = new ArrayList<JFormalParameter>();
691         mustBe(LPAREN);
692         if (have(RPAREN))
693             return parameters; // ()
694         do {
695             parameters.add(formalParameter());
696         } while (have(COMMA));
697         mustBe(RPAREN);
698         return parameters;
699     }

701     /**
702      * Parse a formal parameter.
703      * 
704      * <pre>
705      *   formalParameter ::= type IDENTIFIER
706      * </pre>
707      * 
708      * @return an AST for a formalParameter.
709      */
710 
711     private JFormalParameter formalParameter() {
712         int line = scanner.token().line();
713         Type type = type();
714         mustBe(IDENTIFIER);
715         String name = scanner.previousToken().image();
716         return new JFormalParameter(line, name, type);
717     }

719     /**
720      * Parse a parenthesized expression.
721      * 
722      * <pre>
723      *   parExpression ::= LPAREN expression RPAREN
724      * </pre>
725      * 
726      * @return an AST for a parExpression.
727      */
728 
729     private JExpression parExpression() {
730         mustBe(LPAREN);
731         JExpression expr = expression();
732         mustBe(RPAREN);
733         return expr;

736     /**
737      * Parse a local variable declaration statement.
738      * 
739      * <pre>
740      *   localVariableDeclarationStatement ::= type 
741      *                                           variableDeclarators 
742      *                                             SEMI
743      * </pre>
744      * 
745      * @return an AST for a variableDeclaration.
746      */
747 
748     private JVariableDeclaration localVariableDeclarationStatement() {
749         int line = scanner.token().line();
750         ArrayList<String> mods = new ArrayList<String>();
751         ArrayList<JVariableDeclarator> vdecls = variableDeclarators(type());
752         mustBe(SEMI);
753         return new JVariableDeclaration(line, mods, vdecls);
754     }

756     /**
757      * Parse variable declarators.
758      * 
759      * <pre>
760      *   variableDeclarators ::= variableDeclarator 
761      *                             {COMMA variableDeclarator}
762      * </pre>
763      * 
764      * @param type
765      *            type of the variables.
766      * @return a list of variable declarators.
767      */
768 
769     private ArrayList<JVariableDeclarator> variableDeclarators(Type type) {
770         ArrayList<JVariableDeclarator> variableDeclarators = 
771                                        new ArrayList<JVariableDeclarator>();
772         do {
773             variableDeclarators.add(variableDeclarator(type));
774         } while (have(COMMA));
775         return variableDeclarators;
776     }

778     /**
779      * Parse a variable declarator.
780      * 
781      * <pre>
782      *   variableDeclarator ::= IDENTIFIER
783      *                          [ASSIGN variableInitializer]
784      * </pre>
785      * 
786      * @param type
787      *            type of the variable.
788      * @return an AST for a variableDeclarator.
789      */
790 
791     private JVariableDeclarator variableDeclarator(Type type) {
792         int line = scanner.token().line();
793         mustBe(IDENTIFIER);
794         String name = scanner.previousToken().image();
795         JExpression initial = have(ASSIGN) ? variableInitializer(type) : null;
796         return new JVariableDeclarator(line, name, type, initial);
797     }

800      * Parse a variable initializer.
801      * 
802      * <pre>
803      *   variableInitializer ::= arrayInitializer
804      *                         | expression
805      * </pre>
806      * 
807      * @param type
808      *            type of the variable.
809      * @return an AST for a variableInitializer.
810      */
811 
812     private JExpression variableInitializer(Type type) {
813         if (see(LCURLY)) {
814             return arrayInitializer(type);
815         }
816         return expression();
817     }

819     /**
820      * Parse an array initializer.
821      * 
822      * <pre>
823      *   arrayInitializer ::= LCURLY 
824      *                          [variableInitializer 
825      *                            {COMMA variableInitializer} [COMMA]]
826      *                        RCURLY
827      * </pre>
828      * 
829      * @param type
830      *            type of the array.
831      * @return an AST for an arrayInitializer.
832      */
833 
834     private JArrayInitializer arrayInitializer(Type type) {
835         int line = scanner.token().line();
836         ArrayList<JExpression> initials = new ArrayList<JExpression>();
837         mustBe(LCURLY);
838         if (have(RCURLY)) {
839             return new JArrayInitializer(line, type, initials);
840         }
841         initials.add(variableInitializer(type.componentType()));
842         while (have(COMMA)) {
843             initials.add(see(RCURLY) ? null : variableInitializer(type
844                     .componentType()));
845         }
846         mustBe(RCURLY);
847         return new JArrayInitializer(line, type, initials);
848     }

850     /**
851      * Parse arguments.
852      * 
853      * <pre>
854      *   arguments ::= LPAREN [expression {COMMA expression}] RPAREN
855      * </pre>
856      * 
857      * @return a list of expressions.
858      */
859 
860     private ArrayList<JExpression> arguments() {
861         ArrayList<JExpression> args = new ArrayList<JExpression>();
862         mustBe(LPAREN);
863         if (have(RPAREN)) {
864             return args;
865         }
866         do {
867             args.add(expression());
868         } while (have(COMMA));
869         mustBe(RPAREN);
870         return args;
871     }

873     /**
874      * Parse a type.
875      * 
876      * <pre>
877      *   type ::= referenceType 
878      *          | basicType
879      * </pre>
880      * 
881      * @return an instance of Type.
882      */
883 
884     private Type type() {
885         if (seeReferenceType()) {
886             return referenceType();
887         }
888         return basicType();
889     }

891     /**
892      * Parse a basic type.
893      * 
894      * <pre>
895      *   basicType ::= BOOLEAN | CHAR | INT
896      * </pre>
897      * 
898      * @return an instance of Type.
899      */
900 
901     private Type basicType() {
902         if (have(BOOLEAN)) {
903             return Type.BOOLEAN;
904         } else if (have(CHAR)) {
905             return Type.CHAR;
906         } else if (have(INT)) {
907             return Type.INT;
908         } else {
909             reportParserError("Type sought where %s found", scanner.token()
910                     .image());
911             return Type.ANY;
912         }
913     }

915     /**
916      * Parse a reference type.
917      * 
918      * <pre>
919      *   referenceType ::= basicType LBRACK RBRACK {LBRACK RBRACK}
920      *                   | qualifiedIdentifier {LBRACK RBRACK}
921      * </pre>
922      * 
923      * @return an instance of Type.
924      */
925 
926     private Type referenceType() {
927         Type type = null;
928         if (!see(IDENTIFIER)) {
929             type = basicType();
930             mustBe(LBRACK);
931             mustBe(RBRACK);
932             type = new ArrayTypeName(type);
933         } else {
934             type = qualifiedIdentifier();
935         }
936         while (seeDims()) {
937             mustBe(LBRACK);
938             mustBe(RBRACK);
939             type = new ArrayTypeName(type);
940         }
941         return type;
942     }

944     /**
945      * Parse a statement expression.
946      * 
947      * <pre>
948      *   statementExpression ::= expression // but must have 
949      *                                      // side-effect, eg i++
950      * </pre>
951      * 
952      * @return an AST for a statementExpression.
953      */
954 
955     private JStatement statementExpression() {
956         int line = scanner.token().line();
957         JExpression expr = expression();
958         if (expr instanceof JAssignment || expr instanceof JPreIncrementOp
959                                         || expr instanceof JPostDecrementOp
960                                         || expr instanceof JMessageExpression
961                                         || expr instanceof JSuperConstruction
962                                         || expr instanceof JThisConstruction 
963                                         || expr instanceof JNewOp
964                                         || expr instanceof JNewArrayOp) {
965             // So as not to save on stack
966             expr.isStatementExpression = true;
967         } else {
968             reportParserError("Invalid statement expression; "
969                     + "it does not have a side-effect");
970         }
971         return new JStatementExpression(line, expr);
972     }

975      * An expression.
976      * 
977      * <pre>
978      *   expression ::= assignmentExpression
979      * </pre>
980      * 
981      * @return an AST for an expression.
982      */
983 
984     private JExpression expression() {
985         return assignmentExpression();
986     }

988     /**
989      * Parse an assignment expression.
990      * 
991      * <pre>
992      *   assignmentExpression ::= 
993      *       conditionalAndExpression // level 13
994      *           [( ASSIGN  // conditionalExpression
995      *            | PLUS_ASSIGN // must be valid lhs
996      *            )
997      *            assignmentExpression]
998      * </pre>
999      * 
1000     * @return an AST for an assignmentExpression.
1001     */
1002
1003    private JExpression assignmentExpression() {
1004        int line = scanner.token().line();
1005        JExpression lhs = conditionalAndExpression();
1006        if (have(ASSIGN)) {
1007            return new JAssignOp(line, lhs, assignmentExpression());
1008        } else if (have(PLUS_ASSIGN)) {
1009            return new JPlusAssignOp(line, lhs, assignmentExpression());
1010        } else {
1011            return lhs;
1012        }
1013    }

1015    /**
1016     * Parse a conditional-and expression.
1017     * 
1018     * <pre>
1019     *   conditionalAndExpression ::= equalityExpression // level 10
1020     *                                  {LAND equalityExpression}
1021     * </pre>
1022     * 
1023     * @return an AST for a conditionalExpression.
1024     */
1025
1026    private JExpression conditionalAndExpression() {
1027        int line = scanner.token().line();
1028        boolean more = true;
1029        JExpression lhs = equalityExpression();
1030        while (more) {
1031            if (have(LAND)) {
1032                lhs = new JLogicalAndOp(line, lhs, equalityExpression());
1033            } else {
1034                more = false;
1035            }
1036        }
1037        return lhs;
1038    }

1041     * Parse an equality expression.
1042     * 
1043     * <pre>
1044     *   equalityExpression ::= relationalExpression  // level 6
1045     *                            {EQUAL relationalExpression}
1046     * </pre>
1047     * 
1048     * @return an AST for an equalityExpression.
1049     */
1050
1051    private JExpression equalityExpression() {
1052        int line = scanner.token().line();
1053        boolean more = true;
1054        JExpression lhs = relationalExpression();
1055        while (more) {
1056            if (have(EQUAL)) {
1057                lhs = new JEqualOp(line, lhs, relationalExpression());
1058            } else {
1059                more = false;
1060            }
1061        }
1062        return lhs;
1063    }

1065    /**
1066     * Parse a relational expression.
1067     * 
1068     * <pre>
1069     *   relationalExpression ::= additiveExpression  // level 5
1070     *                              [(GT | LE) additiveExpression 
1071     *                              | INSTANCEOF referenceType]
1072     * </pre>
1073     * 
1074     * @return an AST for a relationalExpression.
1075     */
1076
1077    private JExpression relationalExpression() {
1078        int line = scanner.token().line();
1079        JExpression lhs = additiveExpression();
1080        if (have(GT)) {
1081            return new JGreaterThanOp(line, lhs, additiveExpression());
1082        } else if (have(LE)) {
1083            return new JLessEqualOp(line, lhs, additiveExpression());
1084        } else if (have(INSTANCEOF)) {
1085            return new JInstanceOfOp(line, lhs, referenceType());
1086        } else {
1087            return lhs;
1088        }
1089    }

1091    /**
1092     * Parse an additive expression.
1093     * 
1094     * <pre>
1095     *   additiveExpression ::= multiplicativeExpression // level 3
1096     *                            {MINUS multiplicativeExpression}
1097     * </pre>
1098     * 
1099     * @return an AST for an additiveExpression.
1100     */
1101
1102    private JExpression additiveExpression() {
1103        int line = scanner.token().line();
1104        boolean more = true;
1105        JExpression lhs = multiplicativeExpression();
1106        while (more) {
1107            if (have(MINUS)) {
1108                lhs = new JSubtractOp(line, lhs, multiplicativeExpression());
1109            } else if (have(PLUS)) {
1110                lhs = new JPlusOp(line, lhs, multiplicativeExpression());
1111            } else {
1112                more = false;
1113            }
1114        }
1115        return lhs;
1116    }

1118    /**
1119     * Parse a multiplicative expression.
1120     * 
1121     * <pre>
1122     *   multiplicativeExpression ::= unaryExpression  // level 2
1123     *                                  {STAR unaryExpression}
1124     * </pre>
1125     * 
1126     * @return an AST for a multiplicativeExpression.
1127     */
1128
1129    private JExpression multiplicativeExpression() {
1130        int line = scanner.token().line();
1131        boolean more = true;
1132        JExpression lhs = unaryExpression();
1133        while (more) {
1134            if (have(STAR)) {
1135                lhs = new JMultiplyOp(line, lhs, unaryExpression());
1136            } else {
1137                more = false;
1138            }
1139        }
1140        return lhs;
1141    }

1143    /**
1144     * Parse an unary expression.
1145     * 
1146     * <pre>
1147     *   unaryExpression ::= INC unaryExpression // level 1
1148     *                     | MINUS unaryExpression
1149     *                     | simpleUnaryExpression
1150     * </pre>
1151     * 
1152     * @return an AST for an unaryExpression.
1153     */
1154
1155    private JExpression unaryExpression() {
1156        int line = scanner.token().line();
1157        if (have(INC)) {
1158            return new JPreIncrementOp(line, unaryExpression());
1159        } else if (have(MINUS)) {
1160            return new JNegateOp(line, unaryExpression());
1161        } else {
1162            return simpleUnaryExpression();
1163        }
1164    }

1166    /**
1167     * Parse a simple unary expression.
1168     * 
1169     * <pre>
1170     *   simpleUnaryExpression ::= LNOT unaryExpression
1171     *                           | LPAREN basicType RPAREN 
1172     *                               unaryExpression
1173     *                           | LPAREN         
1174     *                               referenceType
1175     *                             RPAREN simpleUnaryExpression
1176     *                           | postfixExpression
1177     * </pre>
1178     * 
1179     * @return an AST for a simpleUnaryExpression.
1180     */
1181
1182    private JExpression simpleUnaryExpression() {
1183        int line = scanner.token().line();
1184        if (have(LNOT)) {
1185            return new JLogicalNotOp(line, unaryExpression());
1186        } else if (seeCast()) {
1187            mustBe(LPAREN);
1188            boolean isBasicType = seeBasicType();
1189            Type type = type();
1190            mustBe(RPAREN);
1191            JExpression expr = isBasicType ? unaryExpression()
1192                    : simpleUnaryExpression();
1193            return new JCastOp(line, type, expr);
1194        } else {
1195            return postfixExpression();
1196        }
1197    }

1199    /**
1200     * Parse a postfix expression.
1201     * 
1202     * <pre>
1203     *   postfixExpression ::= primary {selector} {DEC}
1204     * </pre>
1205     * 
1206     * @return an AST for a postfixExpression.
1207     */
1208
1209    private JExpression postfixExpression() {
1210        int line = scanner.token().line();
1211        JExpression primaryExpr = primary();
1212        while (see(DOT) || see(LBRACK)) {
1213            primaryExpr = selector(primaryExpr);
1214        }
1215        while (have(DEC)) {
1216            primaryExpr = new JPostDecrementOp(line, primaryExpr);
1217        }
1218        return primaryExpr;
1219    }

1221    /**
1222     * Parse a selector.
1223     * 
1224     * <pre>
1225     *   selector ::= DOT qualifiedIdentifier [arguments]
1226     *              | LBRACK expression RBRACK
1227     * </pre>
1228     * 
1229     * @param target
1230     *            the target expression for this selector.
1231     * @return an AST for a selector.
1232     */
1233
1234    private JExpression selector(JExpression target) {
1235        int line = scanner.token().line();
1236        if (have(DOT)) {
1237            // Target . selector
1238            mustBe(IDENTIFIER);
1239            String name = scanner.previousToken().image();
1240            if (see(LPAREN)) {
1241                ArrayList<JExpression> args = arguments();
1242                return new JMessageExpression(line, target, name, args);
1243            } else {
1244                return new JFieldSelection(line, target, name);
1245            }
1246        } else {
1247            mustBe(LBRACK);
1248            JExpression index = expression();
1249            mustBe(RBRACK);
1250            return new JArrayExpression(line, target, index);
1251        }
1252    }

1254    /**
1255     * Parse a primary expression.
1256     * 
1257     * <pre>
1258     *   primary ::= parExpression
1259     *             | THIS [arguments]
1260     *             | SUPER ( arguments 
1261     *                     | DOT IDENTIFIER [arguments] 
1262     *                     )
1263     *             | literal
1264     *             | NEW creator
1265     *             | qualifiedIdentifier [arguments]
1266     * </pre>
1267     * 
1268     * @return an AST for a primary.
1269     */
1270
1271    private JExpression primary() {
1272        int line = scanner.token().line();
1273        if (see(LPAREN)) {
1274            return parExpression();
1275        } else if (have(THIS)) {
1276            if (see(LPAREN)) {
1277                return new JThisConstruction(line, arguments());
1278            } else {
1279                return new JThis(line);
1280            }
1281        } else if (have(SUPER)) {
1282            if (!have(DOT)) {
1283                return new JSuperConstruction(line, arguments());
1284            } else {
1285                mustBe(IDENTIFIER);
1286                String name = scanner.previousToken().image();
1287                JExpression newTarget = new JSuper(line);
1288                if (see(LPAREN)) {
1289                    return new JMessageExpression(line, newTarget, null, name,
1290                            arguments());
1291                } else {
1292                    return new JFieldSelection(line, newTarget, name);
1293                }
1294            }
1295        } else if (have(NEW)) {
1296            return creator();
1297        } else if (see(IDENTIFIER)) {
1298            TypeName id = qualifiedIdentifier();
1299            if (see(LPAREN)) {
1300                return new JMessageExpression(line, null, ambiguousPart(id), id
1301                        .simpleName(), arguments());
1302            } else if (ambiguousPart(id) == null) {
1303                // A simple name
1304                return new JVariable(line, id.simpleName());
1305            } else {
1306                // ambiguousPart.fieldName
1307                return new JFieldSelection(line, ambiguousPart(id), null, id
1308                        .simpleName());
1309            }
1310        } else {
1311            return literal();
1312        }
1313    }

1315    /**
1316     * Parse a creator.
1317     * 
1318     * <pre>
1319     *   creator ::= (basicType | qualifiedIdentifier) 
1320     *                 ( arguments
1321     *                 | LBRACK RBRACK {LBRACK RBRACK} 
1322     *                     [arrayInitializer]
1323     *                 | newArrayDeclarator
1324     *                 )
1325     * </pre>
1326     * 
1327     * @return an AST for a creator.
1328     */
1329
1330    private JExpression creator() {
1331        int line = scanner.token().line();
1332        Type type = seeBasicType() ? basicType() : qualifiedIdentifier();
1333        if (see(LPAREN)) {
1334            ArrayList<JExpression> args = arguments();
1335            return new JNewOp(line, type, args);
1336        } else if (see(LBRACK)) {
1337            if (seeDims()) {
1338                Type expected = type;
1339                while (have(LBRACK)) {
1340                    mustBe(RBRACK);
1341                    expected = new ArrayTypeName(expected);
1342                }
1343                return arrayInitializer(expected);
1344            } else
1345                return newArrayDeclarator(line, type);
1346        } else {
1347            reportParserError("( or [ sought where %s found", scanner.token()
1348                    .image());
1349            return new JWildExpression(line);
1350        }
1351    }

1353    /**
1354     * Parse a new array declarator.
1355     * 
1356     * <pre>
1357     *   newArrayDeclarator ::= LBRACK expression RBRACK 
1358     *                            {LBRACK expression RBRACK}
1359     *                            {LBRACK RBRACK}
1360     * </pre>
1361     * 
1362     * @param line
1363     *            line in which the declarator occurred.
1364     * @param type
1365     *            type of the array.
1366     * @return an AST for a newArrayDeclarator.
1367     */
1368
1369    private JNewArrayOp newArrayDeclarator(int line, Type type) {
1370        ArrayList<JExpression> dimensions = new ArrayList<JExpression>();
1371        mustBe(LBRACK);
1372        dimensions.add(expression());
1373        mustBe(RBRACK);
1374        type = new ArrayTypeName(type);
1375        while (have(LBRACK)) {
1376            if (have(RBRACK)) {
1377                // We're done with dimension expressions
1378                type = new ArrayTypeName(type);
1379                while (have(LBRACK)) {
1380                    mustBe(RBRACK);
1381                    type = new ArrayTypeName(type);
1382                }
1383                return new JNewArrayOp(line, type, dimensions);
1384            } else {
1385                dimensions.add(expression());
1386                type = new ArrayTypeName(type);
1387                mustBe(RBRACK);
1388            }
1389        }
1390        return new JNewArrayOp(line, type, dimensions);
1391    }

1393    /**
1394     * Parse a literal.
1395     * 
1396     * <pre>
1397     *   literal ::= INT_LITERAL | CHAR_LITERAL | STRING_LITERAL
1398     *             | TRUE        | FALSE        | NULL
1399     * </pre>
1400     * 
1401     * @return an AST for a literal.
1402     */
1403
1404    private JExpression literal() {
1405        int line = scanner.token().line();
1406        if (have(INT_LITERAL)) {
1407            return new JLiteralInt(line, scanner.previousToken().image());
1408        } else if (have(CHAR_LITERAL)) {
1409            return new JLiteralChar(line, scanner.previousToken().image());
1410        } else if (have(STRING_LITERAL)) {
1411            return new JLiteralString(line, scanner.previousToken().image());
1412        } else if (have(TRUE)) {
1413            return new JLiteralTrue(line);
1414        } else if (have(FALSE)) {
1415            return new JLiteralFalse(line);
1416        } else if (have(NULL)) {
1417            return new JLiteralNull(line);
1418        } else {
1419            reportParserError("Literal sought where %s found", scanner.token()
1420                    .image());
1421            return new JWildExpression(line);
1422        }
1423    }
