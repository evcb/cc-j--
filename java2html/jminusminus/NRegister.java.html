8   /**
9    * This abstract base class is for both a physical and virtual
10   * register.
11   */
12  
13  abstract class NRegister {
14  
15      /** Register number. */
16      protected int number;
17  
18      /** Register name. */
19      protected String name;
20  
21      /**
22       * Constructs a {@code NRegister} given the register number
23       * and register name.
24       * 
25       * @param number
26       *            register number.
27       * @param name
28       *            register name.
29       */
30  
31      protected NRegister(int number, String name) {
32          this.number = number;
33          this.name = name;
34      }
35  
36      /**
37       * Returns the number of this register.
38       * 
39       * @return the register number.
40       */
41  
42      public int number() {
43          return number;
44      }
45  
46      /**
47       * Returns the name of this register.
48       * 
49       * @return the register name.
50       */
51  
52      public String name() {
53          return name;
54      }
55  
56  }
57  
58  /**
59   * An abstraction for a virtual register.
60   */
61  
62  class NVirtualRegister extends NRegister {
63  
64      /** Type (short name) of value in register. */
65      private String sType;
66  
67      /** Type (long name) of value in register. */
68      private String lType;
69  
70      /**
71       * Constructs a {@code NVirutalRegister} given the register number, the
72       * short type name, and the long type name of the value in the register.
73       * 
74       * @param number
75       *            register number.
76       * @param sType
77       *            type (short name) of value in register.
78       * @param lType
79       *            type (long name) of value in register.
80       */
81  
82      public NVirtualRegister(int number, String sType, String lType) {
83          super(number, "V" + number);
84          this.sType = sType;
85          this.lType = lType;
86      }
87  
88      /**
89       * Returns a string representation of this virtual register.
90       * 
91       * @return string representation.
92       */
93  
94      public String toString() {
95          return "[" + name + "|" + sType + "]";
96      }
97  
98  }
