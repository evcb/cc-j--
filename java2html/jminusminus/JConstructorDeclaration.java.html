3   package jminusminus;
4   
5   import java.util.ArrayList;
6   import static jminusminus.CLConstants.*;
7   
8   /**
9    * The AST node for a constructor declaration. A constructor looks very much
10   * like a method.
11   *
12   * @see JMethodDeclaration
13   */
14  
15  class JConstructorDeclaration extends JMethodDeclaration implements JMember {
16  
17      /** Does this constructor invoke this(...) or super(...)? */
18      private boolean invokesConstructor;
19  
20      /** Defining class */
21      JClassDeclaration definingClass;
22  
23      /**
24       * Constructs an AST node for a constructor declaration given the line
25       * number, modifiers, constructor name, formal parameters, and the
26       * constructor body.
27       * 
28       * @param line
29       *            line in which the constructor declaration occurs in the source
30       *            file.
31       * @param mods
32       *            modifiers.
33       * @param name
34       *            constructor name.
35       * @param params
36       *            the formal parameters.
37       * @param body
38       *            constructor body.
39       */
40  
41      public JConstructorDeclaration(int line, ArrayList<String> mods,
42              String name, ArrayList<JFormalParameter> params, JBlock body)
43  
44      {
45          super(line, mods, name, Type.CONSTRUCTOR, params, body);
46      }
47  
48      /**
49       * Declares this constructor in the parent (class) context.
50       * 
51       * @param context
52       *            the parent (class) context.
53       * @param partial
54       *            the code emitter (basically an abstraction for producing the
55       *            partial class).
56       */
57  
58      public void preAnalyze(Context context, CLEmitter partial) {
59          super.preAnalyze(context, partial);
60          if (isStatic) {
61              JAST.compilationUnit.reportSemanticError(line(),
62                      "Constructor cannot be declared static");
63          } else if (isAbstract) {
64              JAST.compilationUnit.reportSemanticError(line(),
65                      "Constructor cannot be declared abstract");
66          }
67          if (body.statements().size() > 0
68                  && body.statements().get(0) instanceof JStatementExpression) {
69              JStatementExpression first = (JStatementExpression) body
70                      .statements().get(0);
71              if (first.expr instanceof JSuperConstruction) {
72                  ((JSuperConstruction) first.expr).markProperUseOfConstructor();
73                  invokesConstructor = true;
74              } else if (first.expr instanceof JThisConstruction) {
75                  ((JThisConstruction) first.expr).markProperUseOfConstructor();
76                  invokesConstructor = true;
77              }
78          }
79      }
80  
81      /**
82       * Analysis for a constructor declaration is very much like that for a
83       * method declaration.
84       * 
85       * @param context
86       *            context in which names are resolved.
87       * @return the analyzed (and possibly rewritten) AST subtree.
88       */
89  
90      public JAST analyze(Context context) {
91          // Record the defining class declaration.
92          definingClass = (JClassDeclaration) (context.classContext()
93                                                      .definition());
94          MethodContext methodContext =
95              new MethodContext(context, isStatic, returnType);
96          this.context = methodContext;
97  
98          if (!isStatic) {
99              // Offset 0 is used to address "this"
100             this.context.nextOffset();
101         }
102 
103         // Declare the parameters. We consider a formal parameter
104         // to be always initialized, via a function call. 
105         for (JFormalParameter param : params) {
106             LocalVariableDefn defn = new LocalVariableDefn(param.type(),
107                                              this.context.nextOffset());
108             defn.initialize();
109             this.context.addEntry(param.line(), param.name(), defn);
110         }
111         if (body != null) {
112             body = body.analyze(this.context);
113         }
114         return this;
115 
116     }
117 
118     /**
119      * Adds this constructor declaration to the partial class.
120      * 
121      * @param context
122      *            the parent (class) context.
123      * @param partial
124      *            the code emitter (basically an abstraction for producing the
125      *            partial class).
126      */
127 
128     public void partialCodegen(Context context, CLEmitter partial) {
129         partial.addMethod(mods, "<init>", descriptor, null, false);
130         if (!invokesConstructor) {
131             partial.addNoArgInstruction(ALOAD_0);
132             partial.addMemberAccessInstruction(INVOKESPECIAL,
133                     ((JTypeDecl) context.classContext().definition())
134                             .superType().jvmName(), "<init>", "()V");
135         }
136         partial.addNoArgInstruction(RETURN);
137     }
138 
139     /**
140      * Generates code for the constructor declaration.
141      * 
142      * @param output
143      *            the code emitter (basically an abstraction for producing the
144      *            .class file).
145      */
146 
147     public void codegen(CLEmitter output) {
148         output.addMethod(mods, "<init>", descriptor, null, false);
149         if (!invokesConstructor) {
150             output.addNoArgInstruction(ALOAD_0);
151             output.addMemberAccessInstruction(INVOKESPECIAL,
152                     ((JTypeDecl) context.classContext().definition())
153                             .superType().jvmName(), "<init>", "()V");
154         }
155         // Field initializations
156         for (JFieldDeclaration field : definingClass
157                 .instanceFieldInitializations()) {
158             field.codegenInitializations(output);
159         }
160         // And then the body
161         body.codegen(output);
162         output.addNoArgInstruction(RETURN);
163     }
164 
165     /**
166      * {@inheritDoc}
167      */
168 
169     public void writeToStdOut(PrettyPrinter p) {
170         p.printf("<JConstructorDeclaration line=\"%d\" " + "name=\"%s\">\n",
171                 line(), name);
172         p.indentRight();
173         if (context != null) {
174             context.writeToStdOut(p);
175         }
176         if (mods != null) {
177             p.println("<Modifiers>");
178             p.indentRight();
179             for (String mod : mods) {
180                 p.printf("<Modifier name=\"%s\"/>\n", mod);
181             }
182             p.indentLeft();
183             p.println("</Modifiers>");
184         }
185         if (params != null) {
186             p.println("<FormalParameters>");
187             for (JFormalParameter param : params) {
188                 p.indentRight();
189                 param.writeToStdOut(p);
190                 p.indentLeft();
191             }
192             p.println("</FormalParameters>");
193         }
194         if (body != null) {
195             p.println("<Body>");
196             p.indentRight();
197             body.writeToStdOut(p);
198             p.indentLeft();
199             p.println("</Body>");
200         }
201         p.indentLeft();
202         p.println("</JConstructorDeclaration>");
203     }
