3   package jminusminus;
4   
5   import java.util.ArrayList;
6   import java.util.BitSet;
7   
8   import static jminusminus.NPhysicalRegister.*;
9   
10  /**
11   * The abstract base class for a register allocator that maps virtual registers 
12   * (from LIR code) to physical registers on the target machine.
13   */
14  
15  public abstract class NRegisterAllocator {
16  
17      /** The control flow graph for a method. */
18      protected NControlFlowGraph cfg;
19  
20      /**
21       * Constructs a {@code NRegisterAllocator} object given the control flow 
22       * graph for method.
23       * 
24       * @param cfg
25       *            control flow graph for a method.
26       */
27  
28      protected NRegisterAllocator(NControlFlowGraph cfg) {
29          this.cfg = cfg;
30          this.cfg.intervals = new ArrayList<NInterval>();
31          for (int i = 0; i < cfg.registers.size(); i++) {
32              this.cfg.intervals.add(new NInterval(i, cfg));
33          }
34          this.cfg.maxIntervals = this.cfg.intervals.size();
35      }
36  
37      /**
38       * The work horse that does the allocation, implemented in the concrete
39       * sub-classes of NRegisterAllocator.
40       */
41  
42      public abstract void allocation();
43  
44      /**
45       * Builds the intervals for a control flow graph.
46       */
47  
48      protected void buildIntervals() {
49          this.computeLocalLiveSets();
50          this.computeGlobalLiveSets();
51          for (int i = cfg.basicBlocks.size() - 1; i >= 0; i--) {
52              NBasicBlock currBlock = cfg.basicBlocks.get(i);
53              if (currBlock.lir.size() == 0) {
54                  continue;
55              }
56              int blockStart = currBlock.lir.get(0).id;
57              int blockEnd   = currBlock.lir.get(currBlock.lir.size() - 1).id;
58              BitSet liveOut = currBlock.liveOut;
59              for (int idx = liveOut.nextSetBit(0); idx >= 0; idx = liveOut
60                      .nextSetBit(idx + 1)) {
61                  cfg.intervals.get(idx)
62                               .addOrExtendNRange(new NRange(blockStart,
63                                                             blockEnd)
64                                                 );
65              }
66              for (int j = currBlock.lir.size() - 1; j >= 0; j--) {
67                  int currLIRid = currBlock.lir.get(j).id;
68                  NRegister output = currBlock.lir.get(j).write;
69                  if (output != null) {
70                      cfg.intervals.get(output.number).newFirstRangeStart(
71                              currLIRid);
72                      cfg.intervals.get(output.number)
73                                   .addUsePosition(currLIRid,
74                                                   InstructionType.write);
75                  }
76                  ArrayList<NRegister> inputs = currBlock.lir.get(j).reads;
77                  for (NRegister reg : inputs) {
78                      cfg.intervals.get(reg.number)
79                                   .addOrExtendNRange(new NRange(blockStart,
80                                                                 currLIRid)
81                                                     );
82                      cfg.intervals.get(reg.number)
83                                   .addUsePosition(currLIRid,
84                                                   InstructionType.read);
85                  }
86              }
87          }
88      }
89  
90      /**
91       * Preprocesses information needed for naive, linear, and graph register
92       * allocation schemes. More information provided in method comments.
93       */
94  
95      protected void preprocess() {
96          // Allocate any fixed registers (a0, ..., a3 and v0) that were
97          // assigned during generation phase to the appropriate
98          // interval.
99          for (int i = 0; i < 32; i++) {
100             if (cfg.registers.get(i) != null) {
101                 cfg.intervals.get(i).pRegister = ((NPhysicalRegister) 
102                                                    cfg.registers.get(i)
103                                                  );
104             }
105         }
106 
107         // Assign stack offset (relative to fp) for formal parameters
108         // fourth and above, and stack offset (relative to sp) for
109         // arguments fourth or above.
110         for (NBasicBlock block : cfg.basicBlocks) {
111             for (NLIRInstruction lir : block.lir) {
112                 if (lir instanceof NLIRLoadLocal) {
113                     NLIRLoadLocal loadLocal = (NLIRLoadLocal) lir;
114                     if (loadLocal.local >= 4) {
115                         NInterval interval = cfg.intervals.get(
116                                 ((NVirtualRegister) loadLocal.write).number());
117                         
118                         interval.spill      = true;
119                         interval.offset     = loadLocal.local - 3;
120                         interval.offsetFrom = OffsetFrom.FP;
121                     }
122                 }
123             }
124         }
125     }
126 
127     /**
128      * Iterate through a list of basic blocks in order, and sets their liveUse
129      * and liveDef BitSet fields to represent the appropriate virtual registers
130      * that are locally defined to each block. It works internally with the
131      * cfg's basicBlock structure.
132      */
133 
134     private void computeLocalLiveSets() {
135         for (NBasicBlock block : cfg.basicBlocks) {
136             block.liveUse = new BitSet(cfg.registers.size());
137             block.liveDef = new BitSet(cfg.registers.size());
138             for (NLIRInstruction inst : block.lir) {
139                 for (NRegister reg : inst.reads) {
140                     if (!(block.liveDef.get(reg.number()))) {
141                         block.liveUse.set(reg.number());
142                     }
143                 }
144                 if (inst.write != null) {
145                     block.liveDef.set(inst.write.number());
146                 }
147             }
148         }
149     }
150 
151     /**
152      * Iterate through a list of basic blocks in reverse order, and sets their
153      * lliveIn and liveOut bit sets to reflect global use-def information. It
154      * works internally with the cfg's basicBlock structure.
155      */
156 
157     private void computeGlobalLiveSets() {
158         boolean changed = false;
159         for (NBasicBlock b : cfg.basicBlocks) {
160             b.liveOut = new BitSet(cfg.registers.size());
161         }
162 
163         // note: we only check for changes in liveOut.
164         do {
165             changed = false;
166             for (int i = cfg.basicBlocks.size() - 1; i >= 0; i--) {
167                 NBasicBlock currBlock = cfg.basicBlocks.get(i);
168                 BitSet newLiveOut = new BitSet(cfg.registers.size());
169                 for (NBasicBlock successor : currBlock.successors) {
170                     newLiveOut.or(successor.liveIn);
171                 }
172                 if (!currBlock.liveOut.equals(newLiveOut)) {
173                     currBlock.liveOut = newLiveOut;
174                     changed = true;
175                 }
176                 currBlock.liveIn = (BitSet) currBlock.liveOut.clone();
177                 currBlock.liveIn.andNot(currBlock.liveDef);
178                 currBlock.liveIn.or(currBlock.liveUse);
179             }
180         } while (changed);
181     }
182 
183     /**
184      * Prints the local and global live set informations to STDOUT.
185      *
186      * @param p
187      *          pretty printing with indentation.
188      */
189     protected void writeSetsToStdOut(PrettyPrinter p) {
190         p.println("========== Live set informations ==========\n");
191         p.println("------- Local live sets -------");
192         for (NBasicBlock block : cfg.basicBlocks) {
193             p.println(block.id());
194 
195             p.printf("   liveUse: ");
196             BitSet use = block.liveUse;
197             
198             if (use.nextSetBit(0) < 0) p.println();
199 
200             for (int i = use.nextSetBit(0); i >= 0; i = use.nextSetBit(i + 1)) {
201                 if (i < 32) {
202                     p.printf(regInfo[i] + 
203                             (use.nextSetBit(i + 1) >= 0 ? ", " : "\n"));
204                 } else {
205                     p.printf("V" + i + 
206                             (use.nextSetBit(i + 1) >= 0 ? ", " : "\n"));
207                 }
208             }
209             p.printf("   liveDef: ");
210             BitSet def = block.liveDef;
211 
212             if (def.nextSetBit(0) < 0) p.println("\n");
213 
214             for (int i = def.nextSetBit(0); i >= 0; i = def.nextSetBit(i + 1)) {
215                 if (i < 32) {
216                     p.printf(regInfo[i] + 
217                             (def.nextSetBit(i + 1) >= 0 ? ", " : "\n\n"));
218                 } else {
219                     p.printf("V" + i + 
220                             (def.nextSetBit(i + 1) >= 0 ? ", " : "\n\n"));
221                 }
222             }
223         }
224 
225         p.println("------- Global live sets -------");
226         for (int idx = cfg.basicBlocks.size() - 1; idx >= 0; idx--) {
227             p.println(cfg.basicBlocks.get(idx).id());
228             
229             p.printf("   liveIn:  ");
230             BitSet in = cfg.basicBlocks.get(idx).liveIn;
231 
232             if (in.nextSetBit(0) < 0) p.println();
233 
234             for (int i = in.nextSetBit(0); i >= 0; i = in.nextSetBit(i + 1)) {
235                 if (i < 32) {
236                     p.printf(regInfo[i] + 
237                             (in.nextSetBit(i + 1) >= 0 ? ", " : "\n"));
238                 } else {
239                     p.printf("V" + i + 
240                             (in.nextSetBit(i + 1) >= 0 ? ", " : "\n"));
241                 }
242             }
243 
244             p.printf("   liveOut: ");
245             BitSet out = cfg.basicBlocks.get(idx).liveOut;
246 
247             if (out.nextSetBit(0) < 0) p.println("\n");
248 
249             for (int i = out.nextSetBit(0); i >= 0; i = out.nextSetBit(i + 1)) {
250                 if (i < 32) {
251                     p.printf(regInfo[i] + 
252                             (out.nextSetBit(i + 1) >= 0 ? ", " : "\n\n"));
253                 } else {
254                     p.printf("V" + i + 
255                             (out.nextSetBit(i + 1) >= 0 ? ", " : "\n\n"));
256                 }
257             }
258         }
259 
260     }
