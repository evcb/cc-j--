3   package jminusminus;
4   
5   import java.util.ArrayList;
6   import java.util.LinkedList;
7   import java.util.Queue;
8   
9   import static jminusminus.NPhysicalRegister.*;
10  
11  /**
12   * Implements a naive register allocation method. Each interval is considered
13   * live for the entire cfg. Intervals are assigned physical registers on a first
14   * come basis. When we run out of registers, we reuse the ones already assigned
15   * and spill.
16   */
17  
18  public class NNaiveRegisterAllocator extends NRegisterAllocator {
19  
20      /**
21       * Constructs a {@code NNaiveRegisterAllocator} given a control flow
22       * graph.
23       * 
24       * @param cfg
25       *            an instance of a control flow graph.
26       */
27  
28      public NNaiveRegisterAllocator(NControlFlowGraph cfg) {
29          super(cfg);
30      }
31  
32      /**
33       * Builds intervals with (naive) register allocation information in them.
34       */
35  
36      public void allocation() {
37          // In this allocation scheme, each interval just has a single
38          // range spanning the entire cfg.
39          for (NInterval interval : cfg.intervals) {
40              NBasicBlock lastBlock = cfg.basicBlocks
41                                         .get(cfg.basicBlocks.size() - 1);
42              NLIRInstruction lastLir = lastBlock.lir
43                                                 .get(lastBlock.lir.size() - 1);
44              interval.ranges.add(new NRange(0, lastLir.id));
45          }
46  
47          this.preprocess();
48  
49          // Allocate registers.
50          Queue<NInterval> assigned = new LinkedList<NInterval>();
51          for (int i = 32, j = 0; i < cfg.intervals.size(); i++) {
52              NInterval interval = cfg.intervals.get(i);
53              if (interval.pRegister == null) {
54                  if (j >= MAX_COUNT) {
55                      // Pull out (from a queue) a register that's
56                      // already assigned to another interval and
57                      // re-assign it to this interval. But then
58                      // we have a spill situation, so
59                      // create an offset for the spill.
60                      NInterval spilled = assigned.remove();
61                      spilled.spill = true;
62                      if (spilled.offset == -1) {
63                          spilled.offset = cfg.offset++;
64                          spilled.offsetFrom = OffsetFrom.SP;
65                      }
66                      interval.pRegister = spilled.pRegister;
67                      interval.spill = true;
68                      if (interval.offset == -1) {
69                          interval.offset = cfg.offset++;
70                          interval.offsetFrom = OffsetFrom.SP;
71                      }
72                  } else {
73                      // Allocate free register to interval.
74                      NPhysicalRegister pRegister = regInfo[T0 + j++];
75                      interval.pRegister = pRegister;
76                      cfg.pRegisters.add(pRegister);
77                  }
78                  assigned.add(interval);
79              }
80          }
81  
82          // Make sure that inputs of LIR instructions are not all
83          //   assigned the same register. Also, handle spills 
84          //   (i.e., generate loads and stores where needed).
85          for (int i = 1; i < cfg.basicBlocks.size(); i++) { 
86              // We ignore block B0
87              NBasicBlock block = cfg.basicBlocks.get(i);
88              ArrayList<NLIRInstruction> newLir = new ArrayList<NLIRInstruction>();
89              for (NLIRInstruction lir : block.lir) {
90                  newLir.add(lir);
91              }
92              for (NLIRInstruction lir : block.lir) {
93                  int id = lir.id;
94  
95                  if (lir.reads.size() == 2) {
96                      NInterval input1 = cfg.intervals.get(lir.reads.get(0)
97                                                              .number()).childAt(id);
98                      NInterval input2 = cfg.intervals.get(lir.reads.get(1)
99                                                              .number()).childAt(id);
100                     if (input1.pRegister == input2.pRegister) {
101                         input2.pRegister = regInfo[T0 + 
102                                                   (input2.pRegister.number() + 1)
103                                                  % MAX_COUNT];
104                     }
105                 }
106 
107                 // Loads.
108                 for (int j = 0; j < lir.reads.size(); j++) {
109                     NInterval input = cfg.intervals.get(
110                             lir.reads.get(j).number()).childAt(id);
111                     if (input.spill) {
112                         NLIRLoad load = new NLIRLoad(block, 
113                                                      id - lir.reads.size() + j, 
114                                                      input.offset,
115                                                      input.offsetFrom, 
116                                                      input.pRegister);
117                         
118                         newLir.add(newLir.indexOf(lir), load);
119                     }
120                 }
121 
122                 // Stores.
123                 if (lir.write != null) {
124                     NInterval output = cfg.intervals.get(lir.write.number());
125                     if (output.spill) {
126                         NLIRStore store = new NLIRStore(block, 
127                                                         id + 1,
128                                                         output.offset, 
129                                                         output.offsetFrom, 
130                                                         lir.write);
131                         
132                         newLir.add(newLir.indexOf(lir) + 1, store);
133                     }
134                 }
135             }
136             block.lir = newLir;
137         }
138     }
139 
140 }
