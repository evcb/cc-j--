7   /**
8    * This abstract base class is the AST node for an unary expression.
9    * A unary expression has a single operand.
10   */
11  
12  abstract class JUnaryExpression extends JExpression {
13  
14      /** The operator. */
15      private String operator;
16  
17      /** The operand. */
18      protected JExpression arg;
19  
20      /**
21       * Constructs an AST node for an unary expression given its line number, the
22       * unary operator, and the operand.
23       * 
24       * @param line
25       *            line in which the unary expression occurs in the source file.
26       * @param operator
27       *            the unary operator.
28       * @param arg
29       *            the operand.
30       */
31  
32      protected JUnaryExpression(int line, String operator, JExpression arg) {
33          super(line);
34          this.operator = operator;
35          this.arg = arg;
36      }
37  
38      /**
39       * {@inheritDoc}
40       */
41  
42      public void writeToStdOut(PrettyPrinter p) {
43          p.printf("<JUnaryExpression line=\"%d\" type=\"%s\" "
44                  + "operator=\"%s\">\n", line(), ((type == null) ? "" : type
45                  .toString()), Util.escapeSpecialXMLChars(operator));
46          p.indentRight();
47          p.printf("<Operand>\n");
48          p.indentRight();
49          arg.writeToStdOut(p);
50          p.indentLeft();
51          p.printf("</Operand>\n");
52          p.indentLeft();
53          p.printf("</JUnaryExpression>\n");
54      }
55  
56  }

58  /**
59   * The AST node for a unary negation (-) expression.
60   */
61  
62  class JNegateOp extends JUnaryExpression {
63  
64      /**
65       * Constructs an AST node for a negation expression given its line number,
66       * and the operand.
67       * 
68       * @param line
69       *            line in which the negation expression occurs in the source
70       *            file.
71       * @param arg
72       *            the operand.
73       */
74  
75      public JNegateOp(int line, JExpression arg) {
76          super(line, "-", arg);
77      }
78  
79      /**
80       * Analyzing the negation operation involves analyzing its operand, checking
81       * its type and determining the result type.
82       * 
83       * @param context
84       *            context in which names are resolved.
85       * @return the analyzed (and possibly rewritten) AST subtree.
86       */
87  
88      public JExpression analyze(Context context) {
89          arg = arg.analyze(context);
90          arg.type().mustMatchExpected(line(), Type.INT);
91          type = Type.INT;
92          return this;
93      }
94  
95      /**
96       * Generating code for the negation operation involves generating code for
97       * the operand, and then the negation instruction.
98       * 
99       * @param output
100      *            the code emitter (basically an abstraction for producing the
101      *            .class file).
102      */
103 
104     public void codegen(CLEmitter output) {
105         arg.codegen(output);
106         output.addNoArgInstruction(INEG);
107     }
108 
109 }

111 /**
112  * The AST node for a logical NOT (!) expression.
113  */
114 
115 class JLogicalNotOp extends JUnaryExpression {
116 
117     /**
118      * Constructs an AST for a logical NOT expression given its line number, and
119      * the operand.
120      * 
121      * @param line
122      *            line in which the logical NOT expression occurs in the source
123      *            file.
124      * @param arg
125      *            the operand.
126      */
127 
128     public JLogicalNotOp(int line, JExpression arg) {
129         super(line, "!", arg);
130     }
131 
132     /**
133      * Analyzing a logical NOT operation means analyzing its operand, insuring
134      * it's a boolean, and setting the result to boolean.
135      * 
136      * @param context
137      *            context in which names are resolved.
138      */
139 
140     public JExpression analyze(Context context) {
141         arg = (JExpression) arg.analyze(context);
142         arg.type().mustMatchExpected(line(), Type.BOOLEAN);
143         type = Type.BOOLEAN;
144         return this;
145     }
146 
147     /**
148      * Generates code for the case where we actually want a boolean value 
149      * ({@code true} or {@code false}) computed onto the stack. For example,
150      * assignment to a boolean variable.
151      * 
152      * @param output
153      *            the code emitter (basically an abstraction for producing the
154      *            .class file).
155      */
156 
157     public void codegen(CLEmitter output) {
158         String elseLabel = output.createLabel();
159         String endIfLabel = output.createLabel();
160         this.codegen(output, elseLabel, false);
161         output.addNoArgInstruction(ICONST_1); // true
162         output.addBranchInstruction(GOTO, endIfLabel);
163         output.addLabel(elseLabel);
164         output.addNoArgInstruction(ICONST_0); // false
165         output.addLabel(endIfLabel);
166     }
167 
168     /**
169      * The code generation necessary for branching simply flips the condition on
170      * which we branch.
171      * 
172      * @param output
173      *            the code emitter (basically an abstraction for producing the
174      *            .class file).
175      */
176 
177     public void codegen(CLEmitter output, String targetLabel, boolean onTrue) {
178         arg.codegen(output, targetLabel, !onTrue);
179     }
180 
181 }

183 /**
184  * The AST node for an expr--.
185  */
186 
187 class JPostDecrementOp extends JUnaryExpression {
188 
189     /**
190      * Constructs an AST node for an expr-- expression given its line number, and
191      * the operand.
192      * 
193      * @param line
194      *            line in which the expression occurs in the source file.
195      * @param arg
196      *            the operand.
197      */
198 
199     public JPostDecrementOp(int line, JExpression arg) {
200         super(line, "post--", arg);
201     }
202 
203     /**
204      * Analyzes the operand as a lhs (since there is a side effect), checks types
205      * and determines the type of the result.
206      * 
207      * @param context
208      *            context in which names are resolved.
209      * @return the analyzed (and possibly rewritten) AST subtree.
210      */
211 
212     public JExpression analyze(Context context) {
213         if (!(arg instanceof JLhs)) {
214             JAST.compilationUnit.reportSemanticError(line,
215                     "Operand to expr-- must have an LValue.");
216             type = Type.ANY;
217         } else {
218             arg = (JExpression) arg.analyze(context);
219             arg.type().mustMatchExpected(line(), Type.INT);
220             type = Type.INT;
221         }
222         return this;
223     }
224 
225     /**
226      * In generating code for a post-decrement operation, we treat simple
227      * variable ({@link JVariable}) operands specially since the JVM has an 
228      * increment instruction. 
229      * Otherwise, we rely on the {@link JLhs} code generation support for
230      * generating the proper code. Notice that we distinguish between
231      * expressions that are statement expressions and those that are not; we
232      * insure the proper value (before the decrement) is left atop the stack in
233      * the latter case.
234      * 
235      * @param output
236      *            the code emitter (basically an abstraction for producing the
237      *            .class file).
238      */
239 
240     public void codegen(CLEmitter output) {
241         if (arg instanceof JVariable) {
242             // A local variable; otherwise analyze() would
243             // have replaced it with an explicit field selection.
244             int offset = ((LocalVariableDefn) ((JVariable) arg).iDefn())
245                     .offset();
246             if (!isStatementExpression) {
247                 // Loading its original rvalue
248                 arg.codegen(output);
249             }
250             output.addIINCInstruction(offset, -1);
251         } else {
252             ((JLhs) arg).codegenLoadLhsLvalue(output);
253             ((JLhs) arg).codegenLoadLhsRvalue(output);
254             if (!isStatementExpression) {
255                 // Loading its original rvalue
256                 ((JLhs) arg).codegenDuplicateRvalue(output);
257             }
258             output.addNoArgInstruction(ICONST_1);
259             output.addNoArgInstruction(ISUB);
260             ((JLhs) arg).codegenStore(output);
261         }
262     }
263 
264 }

266 /**
267  * The AST node for a ++expr expression.
268  */
269 
270 class JPreIncrementOp extends JUnaryExpression {
271 
272     /**
273      * Constructs an AST node for a ++expr given its line number, and the
274      * operand.
275      * 
276      * @param line
277      *            line in which the expression occurs in the source file.
278      * @param arg
279      *            the operand.
280      */
281 
282     public JPreIncrementOp(int line, JExpression arg) {
283         super(line, "++pre", arg);
284     }
285 
286     /**
287      * Analyzes the operand as a lhs (since there is a side effect), check types
288      * and determine the type of the result.
289      * 
290      * @param context
291      *            context in which names are resolved.
292      * @return the analyzed (and possibly rewritten) AST subtree.
293      */
294 
295     public JExpression analyze(Context context) {
296         if (!(arg instanceof JLhs)) {
297             JAST.compilationUnit.reportSemanticError(line,
298                     "Operand to ++expr must have an LValue.");
299             type = Type.ANY;
300         } else {
301             arg = (JExpression) arg.analyze(context);
302             arg.type().mustMatchExpected(line(), Type.INT);
303             type = Type.INT;
304         }
305         return this;
306     }
307 
308     /**
309      * In generating code for a pre-increment operation, we treat simple
310      * variable ({@link JVariable}) operands specially since the JVM has an 
311      * increment instruction. 
312      * Otherwise, we rely on the {@link JLhs} code generation support for
313      * generating the proper code. Notice that we distinguish between
314      * expressions that are statement expressions and those that are not; we
315      * insure the proper value (after the increment) is left atop the stack in
316      * the latter case.
317      * 
318      * @param output
319      *            the code emitter (basically an abstraction for producing the
320      *            .class file).
321      */
322 
323     public void codegen(CLEmitter output) {
324         if (arg instanceof JVariable) {
325             // A local variable; otherwise analyze() would
326             // have replaced it with an explicit field selection.
327             int offset = ((LocalVariableDefn) ((JVariable) arg).iDefn())
328                     .offset();
329             output.addIINCInstruction(offset, 1);
330             if (!isStatementExpression) {
331                 // Loading its original rvalue
332                 arg.codegen(output);
333             }
334         } else {
335             ((JLhs) arg).codegenLoadLhsLvalue(output);
336             ((JLhs) arg).codegenLoadLhsRvalue(output);
337             output.addNoArgInstruction(ICONST_1);
338             output.addNoArgInstruction(IADD);
339             if (!isStatementExpression) {
340                 // Loading its original rvalue
341                 ((JLhs) arg).codegenDuplicateRvalue(output);
342             }
343             ((JLhs) arg).codegenStore(output);
344         }
345     }
