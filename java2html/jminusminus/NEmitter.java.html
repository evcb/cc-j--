18  /**
19   * A class for generating native SPIM code.
20   */
21  
22  public class NEmitter {
23  
24      /** Source program file name. */
25      private String sourceFile;
26  
27      /**
28       * Map of maps, one per class in the compilation unit. Each one of them maps
29       * methods in a class to their control flow graph.
30       */
31      private HashMap<CLFile, HashMap<CLMethodInfo, NControlFlowGraph>> classes;
32  
33      /** Destination directory for the native SPIM code. */
34      private String destDir;
35  
36      /**
37       * Whether an error occurred while creating/writing SPIM code.
38       */
39      private boolean errorHasOccurred;
40  
41      /**
42       * Report any error that occurs while creating/writing the spim file, to
43       * STDERR.
44       * 
45       * @param message
46       *            message identifying the error.
47       * @param args
48       *            related values.
49       */
50  
51      private void reportEmitterError(String message, Object... args) {
52          System.err.printf(message, args);
53          System.err.println();
54          errorHasOccurred = true;
55      }
56  
57      /**
58       * Emits SPIM code to setup a stack frame for the procedure denoted by cfg.
59       * This involves saving the return address (ra), saving the frame pointer
60       * (fp), saving any physical registers (t0, ..., t9, s0, ..., s7) used by
61       * the procedure, and setting up the new value for fp (i.e. pushing a stack
62       * frame).
63       * 
64       * @param cfg
65       *            the control flow graph instance.
66       * @param out
67       *            output stream for SPIM code.
68       */
69  
70      private void pushStackFrame(NControlFlowGraph cfg, PrintWriter out) {
71          int frameSize = cfg.pRegisters.size() * 4 + cfg.offset * 4 + 8;
72          out.printf(
73                  "    subu    $sp,$sp,%d \t # Stack frame is %d bytes long\n",
74                  frameSize, frameSize);
75          out.printf("    sw      $ra,%d($sp) \t # Save return address\n",
76                  frameSize - 4);
77          out.printf("    sw      $fp,%d($sp) \t # Save frame pointer\n",
78                  frameSize - 8);
79          int i = 12;
80          for (NPhysicalRegister pRegister : cfg.pRegisters) {
81              out.printf("    sw      %s,%d($sp) \t # Save register %s\n",
82                      pRegister, frameSize - i, pRegister);
83              i += 4;
84          }
85          out.printf("    addiu   $fp,$sp,%d \t # Save frame pointer\n",
86                  frameSize - 4);
87          out.println();
88      }
89  
90      /**
91       * Emits SPIM code to pop the stack frame that was setup for the procedure
92       * denoted by cfg. This involves restoring the return address (ra), the
93       * frame pointer (fp), any physical registers (t0, ..., t9, s0, ..., s7)
94       * used by the procedure, setting fp to the restored value (i.e. popping the
95       * stack frame), and finally jumping to ra (the caller).
96       * 
97       * @param cfg
98       *            the control flow graph instance.
99       * @param out
100      *            output stream for SPIM code.
101      */
102 
103     private void popStackFrame(NControlFlowGraph cfg, PrintWriter out) {
104         int frameSize = cfg.pRegisters.size() * 4 + cfg.offset * 4 + 8;
105         out.printf("%s.restore:\n", cfg.labelPrefix);
106         out.printf("    lw      $ra,%d($sp) \t # Restore return address\n",
107                 frameSize - 4);
108         out.printf("    lw      $fp,%d($sp) \t # Restore frame pointer\n",
109                 frameSize - 8);
110         int i = 12;
111         for (NPhysicalRegister pRegister : cfg.pRegisters) {
112             out.printf("    lw      %s,%d($sp) \t # Restore register %s\n",
113                     pRegister, frameSize - i, pRegister);
114             i += 4;
115         }
116         out.printf("    addiu   $sp,$sp,%d \t # Pop stack\n", frameSize);
117         out.printf("    jr      $ra \t # Return to caller\n", frameSize);
118         out.println();
119     }

121     /**
122      * Constructs a NEmitter instance given the source file, list of CLFile 
123      * objects, and the register allocation scheme.
124      * 
125      * @param sourceFile
126      *            the source j-- program file name.
127      * @param clFiles
128      *            list of CLFile objects.
129      * @param ra
130      *            register allocation scheme (naive, linear, or graph).
131      */
132 
133     public NEmitter(String sourceFile, ArrayList<CLFile> clFiles, String ra) {
134         this.sourceFile = sourceFile.substring(sourceFile
135                 .lastIndexOf(File.separator) + 1);
136         classes = new HashMap<CLFile, HashMap<CLMethodInfo, NControlFlowGraph>>();
137         for (CLFile clFile : clFiles) {
138             CLConstantPool cp = clFile.constantPool;
139             HashMap<CLMethodInfo, NControlFlowGraph> methods = 
140                                     new HashMap<CLMethodInfo, NControlFlowGraph>();
141 
142             for (int i = 0; i < clFile.methodsCount; i++) {
143                 CLMethodInfo m = clFile.methods.get(i);
144 
145                 // Build a control flow graph (cfg) for this method.
146                 // Each block in the cfg, at the end of this step,
147                 // has the JVM bytecode translated into tuple
148                 // representation.
149                 NControlFlowGraph cfg = new NControlFlowGraph(cp, m);
150 
151                 // Write the tuples in cfg to STDOUT.
152                 PrettyPrinter p = new PrettyPrinter();
153                 p.printf("%s %s\n", cfg.name, cfg.desc);
154                 cfg.writeTuplesToStdOut(p);
155 
156                 // Identify blocks in cfg that are loop heads and
157                 // loop tails. Also, compute number of backward
158                 // branches to blocks.
159                 cfg.detectLoops(cfg.basicBlocks.get(0), null);
160 
161                 // Remove unreachable blocks from cfg.
162                 cfg.removeUnreachableBlocks();
163 
164                 // Compute the dominator of each block in the cfg.
165                 cfg.computeDominators(cfg.basicBlocks.get(0), null);
166 
167                 // Convert the tuples in each block in the cfg to
168                 // high-level (HIR) instructions.
169                 cfg.tuplesToHir();
170 
171                 // Eliminate redundant phi functions, i.e., replace
172                 // phi functions of the form x = (y, x, x, ..., x)
173                 // with y.
174                 cfg.eliminateRedundantPhiFunctions();
175 
176                 // Perform optimizations on the high-level
177                 // instructions.
178                 cfg.optimize();
179 
180                 // Write the HIR instructions in cfg to STDOUT.
181                 cfg.writeHirToStdOut(p);
182 
183                 // Convert the HIR instructions in each block in the
184                 // cfg to low-level (LIR) instructions.
185                 cfg.hirToLir();
186 
187                 // Resolve phi functions;
188                 cfg.resolvePhiFunctions();
189 
190                 // Compute block order.
191                 cfg.orderBlocks();
192 
193                 // Assign new ids to LIR instructions.
194                 cfg.renumberLirInstructions();
195 
196                 // Write the LIR instructions in cfg to STDOUT.
197                 cfg.writeLirToStdOut(p);
198 
199                 // Save the cfg for the method in a map keyed in by
200                 // the CLMethodInfo object for the method.
201                 methods.put(m, cfg);
202 
203                 // Perform register allocation.
204                 NRegisterAllocator regAllocator;
205                 if (ra.equals("naive")) {
206                     regAllocator = new NNaiveRegisterAllocator(cfg);
207                 } else if (ra.equals("linear")) {
208                     regAllocator = new NLinearRegisterAllocator(cfg);
209                 } else {
210                     regAllocator = new NGraphRegisterAllocator(cfg);
211                 }
212                 regAllocator.allocation();
213 
214                 // Write the intervals in cfg to STDOUT.
215                 cfg.writeIntervalsToStdOut(p);
216 
217                 // Replace references to virtual registers in LIR
218                 // instructions with references to physical registers.
219                 cfg.allocatePhysicalRegisters();
220 
221                 // Write the LIR instructions in cfg to STDOUT.
222                 cfg.writeLirToStdOut(p);
223 
224                 // Lastly write the live sets to STDOUT.
225                 // Only relevant for linear scan and graph coloring
226                 if (regAllocator instanceof NLinearRegisterAllocator ||
227                     regAllocator instanceof NGraphRegisterAllocator) {
228                     
229                     regAllocator.writeSetsToStdOut(p);
230                 }
231             }
232 
233             // Store the cfgs for the methods in this class in a map.
234             classes.put(clFile, methods);
235         }
236     }

259     /**
260      * Writes out SPIM file(s) to the file system. The destination directory for
261      * the files can be set using the {@code destinationDir} method.
262      */
263 
264     public void write() {
265         String file = "";
266         try {
267             file = destDir + File.separator + sourceFile.replace(".java", ".s");
268             PrintWriter out = new PrintWriter(file);
269 
270             // Header.
271             out.printf("# %s\n", file);
272             out.printf("# Source file: %s\n", sourceFile);
273             out.printf("# Compiled: %s\n\n", Calendar.getInstance().getTime()
274                     .toString());
275 
276             // Translate classes and their methods to SPIM.
277             for (CLFile clFile : classes.keySet()) {
278                 HashMap<CLMethodInfo, NControlFlowGraph> aClass = classes
279                         .get(clFile);
280                 CLConstantPool cp = clFile.constantPool;
281                 int nameIndex = ((CLConstantClassInfo) cp
282                         .cpItem(clFile.thisClass)).nameIndex;
283                 String className = new String(((CLConstantUtf8Info) cp
284                         .cpItem(nameIndex)).b);
285                 for (CLMethodInfo m : aClass.keySet()) {
286                     NControlFlowGraph cfg = aClass.get(m);
287                     String methodName = cfg.name;
288                     String methodDesc = cfg.desc;
289                     if (methodName.equals("<init>")) {
290                         continue;
291                     }
292                     out.printf(".text\n\n");
293                     if (methodName.equals("main")
294                             && methodDesc.equals("([Ljava/lang/String;)V")) {
295                         out.printf("%s:\n", methodName);
296                         cfg.labelPrefix = methodName;
297                     } else {
298                         out.printf("%s.%s:\n", className, methodName);
299                         cfg.labelPrefix = className + "." + methodName;
300                     }
301 
302                     // Setup stack frame for this method
303                     pushStackFrame(cfg, out);
304 
305                     for (NBasicBlock block : cfg.basicBlocks) {
306                         out.printf("%s.%d:\n", cfg.labelPrefix, block.id);
307                         for (NLIRInstruction lir : block.lir) {
308                             lir.toSpim(out);
309                         }
310                         out.printf("\n");
311                     }
312 
313                     // Pop the stack frame for this method.
314                     popStackFrame(cfg, out);
315 
316                     // Data segment for this cfg storing string
317                     // literals.
318                     if (cfg.data.size() > 0) {
319                         out.printf(".data\n\n");
320                         for (String line : cfg.data) {
321                             out.printf(line);
322                         }
323                     }
324 
325                     out.printf("\n\n");
326                 }
327             }
328 
329             // Emit SPIM runtime code; just SPIM.s for now.
330             String[] libs = { "SPIM.s" };
331             out.printf("# SPIM Runtime\n\n");
332             for (String lib : libs) {
333                 file = System.getenv("j") + File.separator + "src"
334                         + File.separator + "spim" + File.separator + lib;
335                 BufferedReader in = new BufferedReader(new FileReader(file));
336                 String line;
337                 while ((line = in.readLine()) != null) {
338                     out.printf("%s\n", line);
339                 }
340                 in.close();
341             }
342 
343             out.close();
344         } catch (FileNotFoundException e) {
345             reportEmitterError("File %s not found", file);
346         } catch (IOException e) {
347             reportEmitterError("Cannot write to file %s", file);
348         }
349     }
