3   package jminusminus;
4   
5   import java.util.StringTokenizer;
6   
7   /**
8    * This class is used to encapsulate ambiguous names that the parser can't
9    * distinguish and disambiguates them during the analysis phase. Ambiguous 
10   * names are meant to deal with snippets like:
11   *
12   * <pre>
13   *   x.y.z
14   *   a.b.c()
15   * </pre>
16   *
17   * Clearly, z is a field and c is a method. But what about <em>x.y</em> and 
18   * <em>a.b</em>? x could be a package name and y a type, making for a (static) 
19   * class field selection. Or, x could be a local variable and y an instance 
20   * field. The parser cannot know how to parse these.
21   * <p>
22   * Disambiguating the ambiguity must wait until analysis time. The parser can,
23   * with <em>x.y.z</em>, treat the <em>.z</em> as a field selection, but 
24   * constructs an AmbiguousName object encapsulating the <em>x.y</em>. And it 
25   * can, with <em>a.b.c()</em>, treat <em>.c()</em> as a message expression, 
26   * but constructs an AbiguousName object encapsulating <em>a.b</em>.
27   * <p>
28   * The {@code reclassify} method is called upon in 
29   * {@link JFieldSelection#analyze(Context)} and 
30   * {@link JMessageExpression#analyze(Context)} to reclassify the components and
31   * construct the proper AST, following the rules for names in the Java Language
32   * Specification (Third Edition), section 6.5.2. In j--, both <em>x.y</em> and 
33   * <em>a.b</em> are clearly expressions in these contexts. If inner types were 
34   * to be introduced, their meaning and their reclassification would necessarily
35   * be more complicated.
36   */
37  
38  class AmbiguousName {
39  
40      /**
41       * Line in which the ambiguous name occurs in the source file.
42       */
43      private int line;
44  
45      /** The ambiguous part, for example x.y */
46      private String name;
47  
48      /**
49       * Constructs an encapsulation of the ambiguous portion of a snippet like
50       * <em>x.y.z</em>.
51       * 
52       * @param line
53       *            line in which the ambiguous name occurs in the source file.
54       * @param name
55       *            the ambiguous part.
56       */
57  
58      public AmbiguousName(int line, String name) {
59          this.line = line;
60          this.name = name;
61      }
62  
63      /**
64       * Reclassifies the name according to the rules in the Java Language
65       * Specification.
66       * 
67       * @param context
68       *            context in which we look up the component names.
69       * @return the properly parsed AST.
70       */
71  
72      public JExpression reclassify(Context context) {
73          // Easier because we require all types to be imported.
74          JExpression result = null;
75          StringTokenizer st = new StringTokenizer(name, ".");
76  
77          // Firstly, find a variable or Type.
78          String newName = st.nextToken();
79          IDefn iDefn = null;
80  
81          do {
82              iDefn = context.lookup(newName);
83              if (iDefn != null) {
84                  result = new JVariable(line, newName);
85                  break;
86              } else if (!st.hasMoreTokens()) {
87                  // Nothing found. :(
88                  JAST.compilationUnit.reportSemanticError(line,
89                          "Cannot find name " + newName);
90                  return null;
91              } else {
92                  newName += "." + st.nextToken();
93              }
94          } while (true);
95  
96          // For now we can assume everything else are fields.
97          while (st.hasMoreTokens()) {
98              result = new JFieldSelection(line, result, st.nextToken());
99          }
100         return result;
101     }
102 
103 }
