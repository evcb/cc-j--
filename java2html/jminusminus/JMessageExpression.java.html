5   import java.util.ArrayList;

7   import static jminusminus.CLConstants.*;

9   /**
10   * The AST node for a message expression that has a target, optionally an
11   * ambiguous part, a message name, and zero or more actual arguments.
12   */
13  
14  class JMessageExpression extends JExpression {
15  
16      /** The target expression. */
17      private JExpression target;
18  
19      /** The ambiguous part that is reclassfied in analyze(). */
20      private AmbiguousName ambiguousPart;
21  
22      /** The message name. */
23      private String messageName;
24  
25      /** Message arguments. */
26      private ArrayList<JExpression> arguments;
27  
28      /** Types of arguments. */
29      private Type[] argTypes;
30  
31      /** The Method representing this message. */
32      private Method method;
33  
34      /**
35       * Constructs an AST node for a message expression without an ambiguous part.
36       * 
37       * @param line
38       *            line in which the expression occurs in the source file.
39       * @param target
40       *            the target expression.
41       * @param messageName
42       *            the message name.
43       * @param arguments
44       *            the ambiguousPart arguments.
45       */
46  
47      protected JMessageExpression(int line, JExpression target,
48              String messageName, ArrayList<JExpression> arguments) {
49          this(line, target, null, messageName, arguments);
50      }
51  
52      /**
53       * Constructs an AST node for a message expression having an ambiguous part.
54       * 
55       * @param line
56       *            line in which the expression occurs in the source file.
57       * @param target
58       *            the target expression.
59       * @param ambiguousPart
60       *            the ambiguous part.
61       * @param messageName
62       *            the message name.
63       * @param arguments
64       *            the arguments.
65       */
66  
67      protected JMessageExpression(int line, JExpression target,
68              AmbiguousName ambiguousPart, String messageName,
69              ArrayList<JExpression> arguments) {
70          super(line);
71          this.target = target;
72          this.ambiguousPart = ambiguousPart;
73          this.messageName = messageName;
74          this.arguments = arguments;
75      }
76  
77      /**
78       * Analysis of a message expression involves: (1) reclassifying any
79       * ambiguous part, (2) analyzing and computing the types for the actual
80       * arguments, (3) determining the type we are currently in (for checking
81       * access), (4) analyzing the target and determining its type, (5) finding
82       * the appropriate Method, (6) checking accessibility, and (7) determining
83       * the result type.
84       * 
85       * @param context
86       *            context in which names are resolved.
87       * @return the analyzed (and possibly rewritten) AST subtree.
88       * @see    Method
89       */
90  
91      public JExpression analyze(Context context) {
92          // Reclassify the ambiguous part
93          if (ambiguousPart != null) {
94              JExpression expr = ambiguousPart.reclassify(context);
95              if (expr != null) {
96                  if (target == null) {
97                      target = expr;
98                  } else {
99                      // Can't even happen syntactically
100                     JAST.compilationUnit.reportSemanticError(line(),
101                             "Badly formed suffix");
102                 }
103             }
104         }
105 
106         // Then analyze the arguments, collecting
107         // their types (in Class form) as argTypes
108         argTypes = new Type[arguments.size()];
109         for (int i = 0; i < arguments.size(); i++) {
110             arguments.set(i, (JExpression) arguments.get(i).analyze(context));
111             argTypes[i] = arguments.get(i).type();
112         }
113 
114         // Where are we now? (For access)
115         Type thisType = ((JTypeDecl) context.classContext.definition())
116                 .thisType();
117 
118         // Then analyze the target
119         if (target == null) {
120             // Implied this (or, implied type for statics)
121             if (!context.methodContext().isStatic()) {
122                 target = new JThis(line()).analyze(context);
123             } else {
124                 target = new JVariable(line(), context.definingType()
125                         .toString()).analyze(context);
126             }
127         } else {
128             target = (JExpression) target.analyze(context);
129             if (target.type().isPrimitive()) {
130                 JAST.compilationUnit.reportSemanticError(line(),
131                         "cannot invoke a message on a primitive type:"
132                                 + target.type());
133             }
134         }
135 
136         // Find appropriate Method for this message expression
137         method = target.type().methodFor(messageName, argTypes);
138         if (method == null) {
139             JAST.compilationUnit.reportSemanticError(line(),
140                     "Cannot find method for: "
141                             + Type.signatureFor(messageName, argTypes));
142             type = Type.ANY;
143         } else {
144             context.definingType().checkAccess(line, (Member) method);
145             type = method.returnType();
146 
147             // Non-static method cannot be referenced from a static context.
148             if (!method.isStatic()) {
149                 if (target instanceof JVariable
150                         && ((JVariable) target).iDefn() instanceof TypeNameDefn) {
151                     JAST.compilationUnit
152                             .reportSemanticError(
153                                     line(),
154                                     "Non-static method "
155                                             + Type.signatureFor(messageName,
156                                                     argTypes)
157                                             + "cannot be referenced from a static context");
158                 }
159             }
160         }
161         return this;
162     }
163 
164     /**
165      * Code generation for a message expression involves generating code for
166      * loading the target onto the stack, generating code to load the actual
167      * arguments onto the stack, and then invoking the named Method. Notice that
168      * if this is a statement expression (as marked by a parent
169      * {@link JStatementExpression}) then we also generate code for popping the 
170      * stacked value for any non-void invocation.
171      * 
172      * @param output
173      *            the code emitter (basically an abstraction for producing the
174      *            .class file).
175      * @see   Method
176      */
177 
178     public void codegen(CLEmitter output) {
179         if (!method.isStatic()) {
180             target.codegen(output);
181         }
182         for (JExpression argument : arguments) {
183             argument.codegen(output);
184         }
185         int mnemonic = method.isStatic() ? INVOKESTATIC : target.type()
186                 .isInterface() ? INVOKEINTERFACE : INVOKEVIRTUAL;
187         output.addMemberAccessInstruction(mnemonic, target.type().jvmName(),
188                 messageName, method.toDescriptor());
189         if (isStatementExpression && type != Type.VOID) {
190             // Pop any value left on the stack
191             output.addNoArgInstruction(POP);
192         }
193     }
194 
195     /**
196      * The semantics of j-- require that we implement short-circuiting branching
197      * in implementing message expressions.
198      * 
199      * @param output
200      *            the code emitter (basically an abstraction for producing the
201      *            .class file).
202      * @param targetLabel
203      *            the label to which we should branch.
204      * @param onTrue
205      *            do we branch on true?
206      */
207 
208     public void codegen(CLEmitter output, String targetLabel, boolean onTrue) {
209         // Push the value
210         codegen(output);
211 
212         if (onTrue) {
213             // Branch on true
214             output.addBranchInstruction(IFNE, targetLabel);
215         } else {
216             // Branch on false
217             output.addBranchInstruction(IFEQ, targetLabel);
218         }
219     }
220 
221     /**
222      * {@inheritDoc}
223      */
224 
225     public void writeToStdOut(PrettyPrinter p) {
226         p.printf("<JMessageExpression line=\"%d\" name=\"%s\">\n", line(),
227                 messageName);
228         p.indentRight();
229         if (target != null) {
230             p.println("<Target>");
231             p.indentRight();
232             target.writeToStdOut(p);
233             p.indentLeft();
234             p.println("</Target>");
235         }
236         if (arguments != null) {
237             p.println("<Arguments>");
238             for (JExpression argument : arguments) {
239                 p.indentRight();
240                 p.println("<Argument>");
241                 p.indentRight();
242                 argument.writeToStdOut(p);
243                 p.indentLeft();
244                 p.println("</Argument>");
245                 p.indentLeft();
246             }
247             p.println("</Arguments>");
248         }
249         p.indentLeft();
250         p.println("</JMessageExpression>");
251     }
