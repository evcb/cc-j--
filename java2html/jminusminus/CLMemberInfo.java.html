3   package jminusminus;
4   
5   import java.io.IOException;
6   import java.util.ArrayList;
7   
8   /**
9    * Abstraction of field_info and method_info structures (JVM Spec Section 4.6,
10   * 4.7). {@code CLFieldInfo} and {@code CLMethodInfo} classes in this file 
11   * provide concrete implementations. Instances of these classes are created to 
12   * store information about class members (fields and methods) when a class is 
13   * read using {@link CLAbsorber} or created using {@link CLEmitter}.
14   */
15  
16  abstract class CLMemberInfo {
17  
18      // The fields below represent the members of the field_info and
19      // method_info structures. See JVM Spec Sections 4.6, 4.7 for
20      // details.
21  
22      /** member_info.access_flags item. */
23      public int accessFlags;
24  
25      /** member_info.name_index item. */
26      public int nameIndex;
27  
28      /** member_info.descriptor_index item. */
29      public int descriptorIndex;
30  
31      /** member_info.attributes_count item. */
32      public int attributesCount;
33  
34      /** member_info.attributes item. */
35      public ArrayList<CLAttributeInfo> attributes;
36  
37      /**
38       * Constructs a CLMemberInfo object.
39       * 
40       * @param accessFlags
41       *            member_info.access_flags item.
42       * @param nameIndex
43       *            member_info.name_index item.
44       * @param descriptorIndex
45       *            member_info.descriptor_index item.
46       * @param attributesCount
47       *            member_info.attributes_count item.
48       * @param attributes
49       *            member_info.attributes item.
50       */
51  
52      protected CLMemberInfo(int accessFlags, int nameIndex, int descriptorIndex,
53              int attributesCount, ArrayList<CLAttributeInfo> attributes) {
54          this.accessFlags = accessFlags;
55          this.nameIndex = nameIndex;
56          this.descriptorIndex = descriptorIndex;
57          this.attributesCount = attributesCount;
58          this.attributes = attributes;
59      }
60  
61      /**
62       * Writes the contents of this class member to the specified output stream.
63       * 
64       * @param out
65       *            output stream.
66       * @exception IOException
67       *                if the contents of this class member can't be written
68       *                to the specified output stream.
69       */
70  
71      public void write(CLOutputStream out) throws IOException {
72          out.writeShort(accessFlags);
73          out.writeShort(nameIndex);
74          out.writeShort(descriptorIndex);
75          out.writeShort(attributesCount);
76          for (int i = 0; i < attributes.size(); i++) {
77              CLAttributeInfo attributeInfo = attributes.get(i);
78              attributeInfo.write(out);
79          }
80      }
81  
82      /**
83       * Writes the contents of this class member to STDOUT in a format similar to
84       * that of javap.
85       * 
86       * @param p
87       *            for pretty printing.
88       */
89  
90      public void writeToStdOut(PrettyPrinter p) {
91          p.indentRight();
92          p.printf("Acces Flags: %s\n", CLFile
93                  .fieldAccessFlagsToString(accessFlags));
94          p.printf("Name Index: %d\n", nameIndex);
95          p.printf("Descriptor Index: %d\n", descriptorIndex);
96          p.println();
97          p.printf("// Attributes (%d Items)\n", attributesCount);
98          for (int i = 0; i < attributes.size(); i++) {
99              CLAttributeInfo attributeInfo = attributes.get(i);
100             attributeInfo.writeToStdOut(p);
101         }
102         p.indentLeft();
103     }
104 
105 }

107 /**
108  * Representation of field_info structure (JVM Spec Section 4.6).
109  */
110 
111 class CLFieldInfo extends CLMemberInfo {
112 
113     /**
114      * Constructs a CLFieldInfo object.
115      * 
116      * @param accessFlags
117      *            field_info.access_flags item.
118      * @param nameIndex
119      *            field_info.name_index item.
120      * @param descriptorIndex
121      *            field_info.descriptor_index item.
122      * @param attributesCount
123      *            field_info.attributes_count item.
124      * @param attributes
125      *            field_info.attributes item.
126      */
127 
128     public CLFieldInfo(int accessFlags, int nameIndex, int descriptorIndex,
129             int attributesCount, ArrayList<CLAttributeInfo> attributes) {
130         super(accessFlags, nameIndex, descriptorIndex, attributesCount,
131                 attributes);
132     }
133 
134     /**
135      * {@inheritDoc}
136      */
137 
138     public void writeToStdOut(PrettyPrinter p) {
139         p.printf("Field {\n");
140         super.writeToStdOut(p);
141         p.printf("}\n");
142     }
143 
144 }
