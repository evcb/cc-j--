3   package jminusminus;
4   
5   import java.io.FileInputStream;
6   import java.io.FileNotFoundException;
7   
8   /**
9    * Driver class for j-- compiler using JavaCC front-end. This is the main entry
10   * point for the compiler. The compiler proceeds as follows:
11   * <ol>
12   *   <li>It reads arguments that affects its behavior.</li>
13   * 
14   *   <li>It builds a scanner.</li>
15   * 
16   *   <li>It builds a parser (using the scanner) and parses the input for 
17   *       producing an abstact syntax tree (AST).</li>
18   * 
19   *   <li>It sends the {@code preAnalyze} message to that AST, which recursively
20   *       descends the tree so far as the member headers for declaring types and
21   *       members in the symbol table (represented as a string of contexts).</li>
22   * 
23   *   <li>It sends the {@code analyze} message to that AST for declaring local 
24   *       variables, and checking and assigning types to expressions. Analysis 
25   *       also sometimes rewrites some of the abstract syntax tree for clarifying
26   *       the semantics. Analysis does all of this by recursively descending the 
27   *       AST down to its leaves.</li>
28   * 
29   *   <li>Finally, it sends a {@code codegen} message to the AST for generating 
30   *       code. Again, {@code codegen} recursively descends the tree, down to its
31   *       leaves, generating JVM code for producing a .class or .s (SPIM) file 
32   *       for each defined type (class).</li>
33   * </ol>
34   */
35  
36  public class JavaCCMain {
37  
38      /** Whether an error occurred during compilation. */
39      private static boolean errorHasOccurred;
40  
41      /**
42       * Entry point.
43       *
44       * @param args
45       *            the command-line arguments.
46       */
47  
48      public static void main(String args[]) {
49          String caller = "java jminusminus.JavaCCMain";
50          String sourceFile = "";
51          String debugOption = "";
52          String outputDir = ".";
53          boolean spimOutput = false;
54          String registerAllocation = "";
55          errorHasOccurred = false;
56          for (int i = 0; i < args.length; i++) {
57              if (args[i].equals("javaccj--")) {
58                  caller = "javaccj--";
59              } else if (args[i].endsWith(".java")) {
60                  sourceFile = args[i];
61              } else if (args[i].equals("-t") || args[i].equals("-p")
62                      || args[i].equals("-pa") || args[i].equals("-a")) {
63                  debugOption = args[i];
64              } else if (args[i].endsWith("-d") && (i + 1) < args.length) {
65                  outputDir = args[++i];
66              } else if (args[i].endsWith("-s") && (i + 1) < args.length) {
67                  spimOutput = true;
68                  registerAllocation = args[++i];
69                  if (!registerAllocation.equals("naive") &&
70                      !registerAllocation.equals("linear") &&
71                      !registerAllocation.equals("graph")
72                      || registerAllocation.equals("")) {
73                      printUsage(caller);
74                      return;
75                  }
76              } else if (args[i].endsWith("-r") && (i + 1) < args.length) {
77                  NPhysicalRegister.MAX_COUNT = Math.min(18, Integer
78                          .parseInt(args[++i]));
79                  NPhysicalRegister.MAX_COUNT = Math.max(1,
80                          NPhysicalRegister.MAX_COUNT);
81              } else {
82                  printUsage(caller);
83                  return;
84              }
85          }
86          if (sourceFile.equals("")) {
87              printUsage(caller);
88              return;
89          }
90  
91          JavaCCParserTokenManager javaCCScanner = null;
92          try {
93              javaCCScanner = new JavaCCParserTokenManager(new SimpleCharStream(
94                      new FileInputStream(sourceFile), 1, 1));
95          } catch (FileNotFoundException e) {
96              System.err.println("Error: file " + sourceFile + " not found.");
97          }
98  
99          if (debugOption.equals("-t")) {
100             // Just tokenize input and print the tokens to STDOUT
101             Token token;
102             do {
103                 token = javaCCScanner.getNextToken();
104                 if (token.kind == JavaCCParserConstants.ERROR) {
105                     System.err.printf(
106                             "%s:%d: Unidentified input token: '%s'\n",
107                             sourceFile, token.beginLine, token.image);
108                     errorHasOccurred |= true;
109                 } else {
110                     System.out.printf("%d\t : %s = %s\n", token.beginLine,
111                             JavaCCParserConstants.tokenImage[token.kind],
112                             token.image);
113                 }
114             } while (token.kind != JavaCCParserConstants.EOF);
115             return;
116         }
117 
118         // Parse input
119         JCompilationUnit ast = null;
120         JavaCCParser javaCCParser = new JavaCCParser(javaCCScanner);
121         javaCCParser.fileName(sourceFile);
122         try {
123             ast = javaCCParser.compilationUnit();
124             errorHasOccurred |= javaCCParser.errorHasOccurred();
125         } catch (ParseException e) {
126             System.err.println(e.getMessage());
127         }
128         if (debugOption.equals("-p")) {
129             ast.writeToStdOut(new PrettyPrinter());
130             return;
131         }
132         if (errorHasOccurred) {
133             return;
134         }
135 
136         // Do pre-analysis
137         ast.preAnalyze();
138         errorHasOccurred |= JAST.compilationUnit.errorHasOccurred();
139         if (debugOption.equals("-pa")) {
140             ast.writeToStdOut(new PrettyPrinter());
141             return;
142         }
143         if (errorHasOccurred) {
144             return;
145         }
146 
147         // Do analysis
148         ast.analyze(null);
149         errorHasOccurred |= JAST.compilationUnit.errorHasOccurred();
150         if (debugOption.equals("-a")) {
151             ast.writeToStdOut(new PrettyPrinter());
152             return;
153         }
154         if (errorHasOccurred) {
155             return;
156         }
157 
158         // Generate JVM code
159         CLEmitter clEmitter = new CLEmitter(!spimOutput);
160         clEmitter.destinationDir(outputDir);
161         ast.codegen(clEmitter);
162         errorHasOccurred |= clEmitter.errorHasOccurred();
163         if (errorHasOccurred) {
164             return;
165         }
166 
167         // If SPIM output was asked for, convert the in-memory
168         // JVM instructions to SPIM using the specified register
169         // allocation scheme.
170         if (spimOutput) {
171             NEmitter nEmitter = new NEmitter(sourceFile, ast.clFiles(),
172                     registerAllocation);
173             nEmitter.destinationDir(outputDir);
174             nEmitter.write();
175             errorHasOccurred |= nEmitter.errorHasOccurred();
176         }
177     }
178 
179     /**
180      * Returns true if an error occurred during compilation; false otherwise.
181      * 
182      * @return {@code true} if error has occurred and {@code false} otherwise.
183      */
184 
185     public static boolean errorHasOccurred() {
186         return errorHasOccurred;
187     }
188 
189     /**
190      * Prints command usage to STDOUT.
191      * 
192      * @param caller
193      *            denotes how this class is invoked.
194      */
195 
196     private static void printUsage(String caller) {
197         String usage = "Usage: "
198                 + caller
199                 + " <options> <source file>\n"
200                 + "Where possible options include:\n"
201                 + "  -t  Only tokenize input and print tokens to STDOUT\n"
202                 + "  -p  Only parse input and print AST to STDOUT\n"
203                 + "  -pa Only parse and pre-analyze input and print "
204                 + "AST to STDOUT\n"
205                 + "  -a  Only parse, pre-analyze, and analyze input "
206                 + "and print AST to STDOUT\n"
207                 + "  -s  <naive|linear|graph> Generate SPIM code\n"
208                 + "  -r  <num> Max. physical registers (1-18) available for allocation; default = 8\n"
209                 + "  -d  <dir> Specify where to place output files; default = .";
210         System.out.println(usage);
211     }
