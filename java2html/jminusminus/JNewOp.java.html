3   package jminusminus;
4   
5   import java.util.ArrayList;
6   import static jminusminus.CLConstants.*;
7   
8   /**
9    * The AST node for a {@code new} expression. It keeps track of its type, the
10   * {@link Constructor} representing the expression, its arguments and their 
11   * types.
12   */
13  
14  class JNewOp extends JExpression {
15  
16      /** The constructor representing this "new" expression. */
17      private Constructor constructor;
18  
19      /** The arguments to the constructor. */
20      private ArrayList<JExpression> arguments;
21  
22      /** Types of the arguments. */
23      private Type[] argTypes;
24  
25      /**
26       * Constructs an AST node for a {@code new} expression.
27       * 
28       * @param line
29       *            the line in which the "new" expression occurs in the source
30       *            file.
31       * @param type
32       *            the type being constructed.
33       * @param arguments
34       *            arguments to the constructor.
35       */
36  
37      public JNewOp(int line, Type type, ArrayList<JExpression> arguments) {
38          super(line);
39          this.type = type;
40          this.arguments = arguments;
41      }
42  
43      /**
44       * To analyze the new operation, we (1) resolve the type, (2) analyze its
45       * arguments, (3) check accessibility of the type, (3) find the appropriate
46       * Constructor.
47       * 
48       * @param context
49       *            context in which names are resolved.
50       * @return the analyzed (and possibly rewritten) AST subtree.
51       */
52  
53      public JExpression analyze(Context context) {
54          // First resolve the type
55          type = type.resolve(context);
56  
57          // Analyze the arguments, collecting
58          // their types (in Class form) as argTypes.
59          argTypes = new Type[arguments.size()];
60          for (int i = 0; i < arguments.size(); i++) {
61              arguments.set(i, (JExpression) arguments.get(i).analyze(context));
62              argTypes[i] = arguments.get(i).type();
63          }
64  
65          // Can't instantiate an abstract type
66          if (type.isAbstract()) {
67              JAST.compilationUnit.reportSemanticError(line(),
68                      "Cannot instantiate an abstract type:" + type.toString());
69          }
70  
71          // Where are we now? Check accessability of type
72          // resolve() checks accessibility, so the following two
73          // are commented.
74          // Type thisType = context.definingType();
75          // thisType.checkAccess( line, type );
76  
77          // Then get the proper constructor, given the arguments
78          constructor = type.constructorFor(argTypes);
79  
80          if (constructor == null) {
81              JAST.compilationUnit.reportSemanticError(line(),
82                      "Cannot find constructor: "
83                              + Type.signatureFor(type.toString(), argTypes));
84          }
85          return this;
86      }
87  
88      /**
89       * Generating code for a new operation involves generating the {@code NEW}
90       * instruction for creating the object on the stack, then generating the 
91       * code for the actual arguments, and then code for invoking the constructor 
92       * (the initialization method).
93       * 
94       * @param output
95       *            the code emitter (basically an abstraction for producing the
96       *            .class file).
97       */
98  
99      public void codegen(CLEmitter output) {
100         output.addReferenceInstruction(NEW, type.jvmName());
101         output.addNoArgInstruction(DUP);
102         for (JExpression argument : arguments) {
103             argument.codegen(output);
104         }
105         output.addMemberAccessInstruction(INVOKESPECIAL,
106                                           type.jvmName(),
107                                           "<init>",
108                                           constructor.toDescriptor());
109     }
110 
111     /**
112      * {@inheritDoc}
113      */
114 
115     public void writeToStdOut(PrettyPrinter p) {
116         p.printf("<JNewOp line=\"%d\" type=\"%s\"/>\n", line(),
117                 ((type == null) ? "" : type.toString()));
118         p.indentRight();
119         if (arguments != null) {
120             p.println("<Arguments>");
121             for (JExpression argument : arguments) {
122                 p.indentRight();
123                 p.println("<Argument>");
124                 p.indentRight();
125                 argument.writeToStdOut(p);
126                 p.indentLeft();
127                 p.println("</Argument>");
128                 p.indentLeft();
129             }
130             p.println("</Arguments>");
131         }
132         p.indentLeft();
133         p.println("</JNewOp>");
134     }
135 
136 }
