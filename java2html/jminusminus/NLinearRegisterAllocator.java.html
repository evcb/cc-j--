5   import java.util.ArrayList;

7   import static jminusminus.NPhysicalRegister.*;

9   /**
10   * Implements the Linear Scan register allocation algorithm.
11   */
12  
13  public class NLinearRegisterAllocator extends NRegisterAllocator {
14      /**
15       * Interval queues for tracking the allocation process.
16       */
17      private ArrayList<NInterval> unhandled;
18      private ArrayList<NInterval> active;
19      private ArrayList<NInterval> inactive;
20  
21      /**
22       * Used to keep track of which intervals get assigned to what physical
23       * register. Needed only in allocateBlockedRegFor.
24       */
25      private ArrayList<ArrayList<NInterval>> regIntervals;
26      private int[] freePos, usePos, blockPos;
27  
28      /**
29       * Constructs a linear register allocator for the given control flow graph.
30       * 
31       * @param cfg
32       *            the control flow graph instance.
33       */
34  
35      public NLinearRegisterAllocator(NControlFlowGraph cfg) {
36          super(cfg);
37          unhandled = new ArrayList<NInterval>();
38          active    = new ArrayList<NInterval>();
39          inactive  = new ArrayList<NInterval>();
40  
41          // Instantiate usePositions and freePos to be the size of
42          // the physical registers used.
43          freePos  = new int[MAX_COUNT];
44          usePos   = new int[MAX_COUNT];
45          blockPos = new int[MAX_COUNT];
46          
47          regIntervals = new ArrayList<ArrayList<NInterval>>();
48          for (int i = 0; i < MAX_COUNT; i++) {
49              regIntervals.add(new ArrayList<NInterval>());
50          }
51      }
52  
53      /**
54       * Performs the linear register allocation, assigning physical registers to
55       * virtual registers.
56       */
57  
58      public void allocation() {
59          // Build the intervals for the control flow graph.
60          this.buildIntervals(); // The correct intervals are now in intervals
61  
62          // Add all intervals corresponding to vregs to unhandled list
63          for (int i = 32; i < cfg.intervals.size(); i++) {
64              this.addSortedToUnhandled(cfg.intervals.get(i));
65          }
66  
67          this.preprocess();
68  
69          NInterval currInterval; // the current interval
70          int psi;                // the current interval's first start position
71          ArrayList<NInterval> tmp;
72  
73          // Linear allocation begins; repeat so long as there are
74          //   additional virtual registers to map to physical registers.
75          while (!unhandled.isEmpty()) {
76              currInterval = unhandled.remove(0);
77              psi = currInterval.firstRangeStart();
78              tmp = new ArrayList<NInterval>();
79              for (int i = 0; i < active.size(); i++) {
80                  if (active.get(i).lastNRangeStop() < psi) {
81                      tmp.add(active.get(i));
82                  } else if (!active.get(i).isLiveAt(psi)) {
83                      inactive.add(active.get(i));
84                      tmp.add(active.get(i));
85                  }
86              }
87              for (NInterval nonActive : tmp) {
88                  active.remove(nonActive);
89              }
90              tmp = new ArrayList<NInterval>();
91              for (int i = 0; i < inactive.size(); i++) {
92                  if (inactive.get(i).lastNRangeStop() < psi) {
93                      tmp.add(inactive.get(i));
94                  } else if (inactive.get(i).isLiveAt(psi)) {
95                      active.add(inactive.get(i));
96                      tmp.add(inactive.get(i));
97                  }
98              }
99              for (NInterval nonInActive : tmp) {
100                 inactive.remove(nonInActive);
101             }
102             if (!this.foundFreeRegFor(currInterval)) {    // check
103                 this.allocateBlockedRegFor(currInterval); // never fails
104             }
105             active.add(currInterval);
106         }
107         this.resolveDataFlow();
108     }
109 
110     /**
111      * Adds a given interval onto the unhandled list, maintaining an order based
112      * on the first range start of the {@code NInterval}s.
113      * 
114      * @param newInterval
115      *            the NInterval to sort onto unhandled.
116      */
117 
118     private void addSortedToUnhandled(NInterval newInterval) {
119         if (unhandled.isEmpty()) {
120             unhandled.add(newInterval);
121         } else {
122             int i = 0;
123             while (i < unhandled.size()
124                     && unhandled.get(i).firstRangeStart() <= newInterval
125                             .firstRangeStart()) {
126                 i++;
127             }
128             unhandled.add(i, newInterval);
129         }
130     }
131 
132     /**
133      * Allocates a free physical register for the current interval. Inspects
134      * active and inactive sets. Cannot split or alter the assigned physical
135      * register of any other interval but current.
136      * 
137      * @param currInterval
138      *            the current interval for which a physical register is sought.
139      * @return {@code true} if a free physical register was found and allocated 
140      *         for {@code currInterval}; {@code false} otherwise.
141      */
142 
143     private boolean foundFreeRegFor(NInterval currInterval) {
144         this.initFreePositions(); // must be reset every iteration
145         for (NInterval activeInterval : active) {
146             if (activeInterval.pRegister != null)
147                 freePos[activeInterval.pRegister.number - T0] = 0;
148         }
149         for (NInterval inactiveInterval : inactive) {
150             if (inactiveInterval.nextIntersection(currInterval) >= 0) {
151                 freePos[inactiveInterval.pRegister.number - T0] = Math.min(
152                               freePos[inactiveInterval.pRegister.number - T0],
153                               inactiveInterval.nextIntersection(currInterval));
154             }
155         }
156 
157         // The physical registers available are in NPhysicalRegister.regInfo
158         // static array. This is indexed from 0 to NPhysicalRegister.MAX_COUNT
159         int reg = this.getBestFreeReg();
160         if (freePos[reg] == 0) {
161             return false;
162         } else if (freePos[reg] > currInterval.lastNRangeStop()) {
163             currInterval.pRegister = regInfo[reg + T0];
164             cfg.pRegisters.add(regInfo[reg + T0]);
165             regIntervals.get(reg).add(currInterval);
166             return true;
167         } else {
168             this.addSortedToUnhandled(currInterval.splitAt(freePos[reg]));
169             currInterval.spill();
170             currInterval.pRegister = regInfo[reg + T0];
171             regIntervals.get(reg).add(currInterval);
172             return true;
173         }
174     }
175 
176     /**
177      * Sets all free positions of physical registers available for allocation 
178      * to a really high number.
179      */
180 
181     private void initFreePositions() {
182         for (int i = 0; i < MAX_COUNT; i++) {
183             freePos[i] = Integer.MAX_VALUE;
184         }
185     }
186 
187     /**
188      * The best free physical register number.
189      * 
190      * @return the register number.
191      */
192 
193     private int getBestFreeReg() {
194         int freeRegNumber = 0;
195         for (int i = 0; i < MAX_COUNT; i++) {
196             if (freePos[i] > freePos[freeRegNumber])
197                 freeRegNumber = i;
198         }
199         return freeRegNumber;
200     }

202     /**
203      * Allocates a register based on spilling an interval.
204      * 
205      * @param currInterval
206      *            the current interval.
207      */
208 
209     private void allocateBlockedRegFor(NInterval currInterval) {
210         this.initUseAndBlockPositions(); // must be reset every iteration
211         for (NInterval activeInterval : active) {
212             usePos[activeInterval.pRegister.number - T0] = Math.min(
213                     usePos[activeInterval.pRegister.number - T0], 
214                     activeInterval.nextUsageOverlapping(currInterval));
215         }
216         for (NInterval inactiveInterval : inactive) {
217             if (inactiveInterval.nextIntersection(currInterval) >= 0) {
218                 usePos[inactiveInterval.pRegister.number - T0] = Math.min(
219                         usePos[inactiveInterval.pRegister.number - T0], 
220                         inactiveInterval.nextUsageOverlapping(currInterval));
221             }
222         }
223         int reg = this.getBestBlockedReg(); // This is just an index in the
224                                             //   usePos array
225         if (usePos[reg] < currInterval.firstUsage()) {
226             // best to spill current - no reg assignment.
227             this.addSortedToUnhandled(currInterval.splitAt(currInterval
228                                                            .firstUsage() - 5));
229             currInterval.spill();
230             NInterval splitChild = currInterval.splitAt(currInterval
231                                                         .firstRangeStart());
232             this.addSortedToUnhandled(splitChild);
233             currInterval.spill();
234         } else {
235             // spilling frees reg for all of current
236             currInterval.pRegister = regInfo[reg + T0];
237             for (NInterval i : regIntervals.get(reg)) {
238                 if (currInterval.nextIntersection(i) >= 0) {
239                     NInterval splitChild = i.splitAt(currInterval
240                                                      .firstRangeStart());
241                     this.addSortedToUnhandled(splitChild);
242                     i.spill();
243                 }
244             }
245             regIntervals.get(reg).add(currInterval);
246         }
247     }
248 
249     /**
250      * Initialize use and block positions before processing each virtual
251      * register.
252      */
253 
254     private void initUseAndBlockPositions() {
255         for (int i = 0; i < MAX_COUNT; i++) {
256             usePos[i]   = Integer.MAX_VALUE;
257             blockPos[i] = Integer.MAX_VALUE;
258         }
259     }
260 
261     /**
262      * Get the best blocked physical register number.
263      * 
264      * @return the register number.
265      */
266 
267     private int getBestBlockedReg() {
268         int usableRegNumber = 0;
269         for (int i = 0; i < MAX_COUNT; i++) {
270             if (usePos[i] > usePos[usableRegNumber])
271                 usableRegNumber = i;
272         }
273         return usableRegNumber;
274     }

276     /**
277      * Resolve the data flow after allocating registers, inserting additional
278      * saves and restores for registers to maintain consistency.
279      */
280 
281     private void resolveDataFlow() {
282         // Local data flow construction.
283         // Devised an alternate way of doing this, perhaps with more
284         // clarity, will implement later, but has same effect.
285         for (NInterval i : cfg.intervals) {
286             if (cfg.registers.get(i.vRegId) != null) {
287                 if (i.spill) {
288                     for (int c = 0; c < i.children.size(); c++) {
289                         if (i.endsAtBlock() == i.children.get(c)
290                                                          .startsAtBlock()) {
291                             if (c == 0) {
292                                 addStoreInstruction(i, i.lastNRangeStop());
293                                 addLoadInstruction(i.children.get(c),
294                                                    i.children.get(c)
295                                                     .firstRangeStart());
296                             } else {
297                                 addStoreInstruction(i.children.get(c - 1),
298                                                     i.children.get(c - 1)
299                                                      .lastNRangeStop());
300                                 
301                                 addLoadInstruction(i.children.get(c),
302                                                    i.children.get(c)
303                                                     .firstRangeStart());
304                             }
305                         }
306                     }
307                 }
308             }
309         }
310 
311         // resolution of global data flow
312         for (NBasicBlock b : cfg.basicBlocks) {
313             for (NBasicBlock s : b.successors) {
314                 for (int i = s.liveIn.nextSetBit(0); i >= 0; i = s.liveIn
315                         .nextSetBit(i + 1)) {
316                     NInterval parent = cfg.intervals.get(i);
317                     NInterval from   = parent.childAtOrEndingBefore(b);
318                     NInterval to     = parent.childAtOrStartingAfter(s);
319                     if (!from.equals(to)) {
320                         addStoreInstruction(from, from.usePositions
321                                                       .floorKey(
322                                                         b.getLastLIRInstId())
323                                                                );
324                         to = getSegmentWithNearestUse(to, s.getFirstLIRInstId());
325                         if (to.usePositions.ceilingEntry(s.getFirstLIRInstId())
326                                 .getValue() == InstructionType.read)
327                             // no use loading prior to a write.
328                             addLoadInstruction(to, to.usePositions
329                                                      .ceilingKey(
330                                                       s.getFirstLIRInstId())
331                                                                 );
332                     }
333                 }
334             }
335         }
336     }

338     /**
339      * Get the the interval segment that contains the nearest first use.
340      * 
341      * @param i
342      *            the interval segment (could be a parent or child).
343      * @param id
344      *            the lir id after which a use is sought.
345      * @return the interval segment that contains the first use at or after
346      *         the id position and is associated with the interval i through a
347      *         sibling or child relationship. Returns i if there is a use after
348      *         id within i. Null if no interval exists that is related to i and
349      *         contains a use position at or after id.
350      */
351 
352     private NInterval getSegmentWithNearestUse(NInterval i, int id) {
353         if (i.usePositions.ceilingEntry(id) != null)
354             return i;
355         else {
356             NInterval parent = i;
357             int idx = 0;
358             if (i.isChild()) {
359                 parent = i.parent;
360                 idx = parent.children.indexOf(i) + 1;
361             }
362             for (; idx < parent.children.size(); idx++) {
363                 if (parent.children.get(idx).usePositions.ceilingEntry(id) != null)
364                     return parent.children.get(idx);
365             }
366             return null;
367         }
368     }

370     /**
371      * Adds a store instruction right after a use position specified by id.
372      * 
373      * @param from
374      *            the interval which this use position is a part of.
375      * @param id
376      *            the id of the use position.
377      */
378 
379     private void addStoreInstruction(NInterval from, int id) {
380         NBasicBlock b = cfg.blockAt(id);
381         id++;
382         if (b.idIsFree(id)) { // assumes always same instr
383             b.insertLIRInst(new NLIRStore(b, id, from.offset, from.offsetFrom,
384                                                               from.pRegister));
385         }
386     }
387 
388     /**
389      * Adds a store instruction right before a use position specified by id.
390      * 
391      * @param to
392      *            the interval which this use position is a part of.
393      * @param id
394      *            the id of the use position.
395      */
396 
397     private void addLoadInstruction(NInterval to, int id) {
398         NBasicBlock s = cfg.blockAt(id);
399         id--;
400         if (s.idIsFree(id)) { // assumes always same instr
401             s.insertLIRInst(new NLIRLoad(s, id, to.offset, to.offsetFrom,
402                                                            to.pRegister));
403         }
404     }
