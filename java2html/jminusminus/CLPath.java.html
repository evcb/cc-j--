3   package jminusminus;
4   
5   import java.io.BufferedInputStream;
6   import java.io.File;
7   import java.io.FileInputStream;
8   import java.io.FileNotFoundException;
9   import java.io.IOException;
10  import java.util.ArrayList;
11  import java.util.StringTokenizer;
12  import java.util.zip.ZipEntry;
13  import java.util.zip.ZipFile;
14  
15  /**
16   * This class can be used to locate and load system, extension, and user-defined
17   * class files from directories and zip (jar) files. The code for this class has
18   * been adapted from the Kopi (http://www.dms.at/kopi/) project.
19   */
20  
21  class CLPath {
22  
23      /**
24       * Stores the individual directories, zip, and jar files from the class
25       * path.
26       */
27      private ArrayList<String> dirs;
28  
29      /**
30       * Return a list of conceptual directories defining the class path.
31       * 
32       * @param classPath
33       *            the directory names defining the class path.
34       * @return a list of conceptual directories defining the class path.
35       */
36  
37      private ArrayList<String> loadClassPath(String classPath) {
38          ArrayList<String> container = new ArrayList<String>();
39  
40          // Add directories/jars/zips from the classpath
41          StringTokenizer entries = new StringTokenizer(classPath,
42                  File.pathSeparator);
43          while (entries.hasMoreTokens()) {
44              container.add(entries.nextToken());
45          }
46  
47          // Add system directories
48          if (System.getProperty("sun.boot.class.path") != null) {
49              entries = new StringTokenizer(System
50                      .getProperty("sun.boot.class.path"), File.pathSeparator);
51              while (entries.hasMoreTokens()) {
52                  container.add(entries.nextToken());
53              }
54          } else {
55              float version = Float
56                      .parseFloat(System.getProperty("java.version"));
57              if (version > 1.1) {
58                  String dir = System.getProperty("java.home")
59                          + File.separatorChar + "lib" + File.separatorChar
60                          + "rt.jar";
61                  container.add(dir);
62              }
63          }
64          return container;
65      }
66  
67      /**
68       * Constructs a CLPath object.
69       */
70  
71      public CLPath() {
72          this(null, null);
73      }
74  
75      /**
76       * Constructs a CLPath object given the directory names defining the path and
77       * the directory for the Java extension classes.
78       * 
79       * @param path
80       *            the directory names defining the class path, separated by path
81       *            separator.
82       * @param extdir
83       *            the directory for the Java extension classes.
84       */
85  
86      public CLPath(String path, String extdir) {
87          if (path == null) {
88              // No path specified, use CLASSPATH
89              path = System.getProperty("java.class.path");
90          }
91          if (path == null) {
92              // Last resort, use current directory
93              path = ".";
94          }
95          dirs = loadClassPath(path);
96          if (extdir == null) {
97              // Java extension classes
98              extdir = System.getProperty("java.ext.dirs");
99          }
100         if (extdir != null) {
101             File extDirectory = new File(extdir);
102             if (extDirectory.isDirectory()) {
103                 File[] extFiles = extDirectory.listFiles();
104                 for (int i = 0; i < extFiles.length; i++) {
105                     File file = extFiles[i];
106                     if (file.isFile()
107                             && (file.getName().endsWith(".zip") || file
108                                     .getName().endsWith(".jar"))) {
109                         dirs.add(file.getName());
110                     } else {
111                         // Wrong suffix; ignore
112                     }
113                 }
114             }
115         }
116     }
117 
118     /**
119      * Returns a CLInputStream instance for the class with specified name
120      * (fully-qualified; tokens separated by '/') or {@code null} if the class 
121      * was not found.
122      * 
123      * @param name
124      *            the fully-qualified name of the class -- java/util/ArrayList
125      *            for example.
126      * @return a CLInputStream instance for the class with specified name or
127      *         {@code null} if the class was not found.
128      */
129 
130     public CLInputStream loadClass(String name) {
131         CLInputStream reader = null;
132         for (int i = 0; i < dirs.size(); i++) {
133             String dir = dirs.get(i);
134             File file = new File(dir);
135             if (file.isDirectory()) {
136                 File theClass = new File(dir, name.replace('/',
137                         File.separatorChar)
138                         + ".class");
139                 if (theClass.canRead()) {
140                     try {
141                         reader = new CLInputStream(new BufferedInputStream(
142                                 new FileInputStream(theClass)));
143                     } catch (FileNotFoundException e) {
144                         // Ignore
145                     }
146                 }
147             } else if (file.isFile()) {
148                 try {
149                     ZipFile zip = new ZipFile(dir);
150                     ZipEntry entry = zip.getEntry(name + ".class");
151                     if (entry != null) {
152                         reader = new CLInputStream(zip.getInputStream(entry));
153                     }
154                 } catch (IOException e) {
155                     // Ignore
156                 }
157             } else {
158                 // Bogus entry; ignore
159             }
160         }
161         return reader;
162     }
