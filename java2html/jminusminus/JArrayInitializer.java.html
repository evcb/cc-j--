3   package jminusminus;
4   
5   import java.util.ArrayList;
6   import static jminusminus.CLConstants.*;
7   
8   /**
9    * The AST node for an array initializer. Basically a list of
10   * initializing expressions.
11   */
12  
13  class JArrayInitializer extends JExpression {
14  
15      /** The initializations. */
16      private ArrayList<JExpression> initials;
17  
18      /**
19       * Constructs an AST node for an array initializer given the
20       * (expected) array type and initial values.
21       * 
22       * @param line
23       *            line in which this array initializer occurs
24       *            in the source file.
25       * @param expected
26       *            the type of the array we're initializing.
27       * @param initials
28       *            initializations.
29       */
30  
31      public JArrayInitializer(int line, Type expected,
32          ArrayList<JExpression> initials) {
33          super(line);
34          type = expected;
35          this.initials = initials;
36      }
37  
38      /**
39       * Analysis of this array initializer involves making sure that
40       * that the type of the initials is the same as the component
41       * type.
42       * 
43       * @param context
44       *            context in which names are resolved.
45       * @return the analyzed (and possibly rewritten) AST subtree.
46       */
47  
48      public JExpression analyze(Context context) {
49          type = type.resolve(context);
50          if (!type.isArray()) {
51              JAST.compilationUnit.reportSemanticError(line,
52                  "Cannot initialize a " + type.toString()
53                      + " with an array sequence {...}");
54              return this; // un-analyzed
55          }
56          Type componentType = type.componentType();
57          for (int i = 0; i < initials.size(); i++) {
58              JExpression component = initials.get(i);
59              initials.set(i, component = component.analyze(context));
60              if (!(component instanceof JArrayInitializer)) {
61                  component.type().mustMatchExpected(line,
62                      componentType);
63              }
64          }
65          return this;
66      }
67  
68      /**
69       * Performs code generation necessary to construct the
70       * initializing array and leaves it on top of the stack.
71       * 
72       * @param output
73       *            the code emitter (basically an abstraction
74       *            for producing the .class file).
75       */
76  
77      public void codegen(CLEmitter output) {
78          Type componentType = type.componentType();
79  
80          // Code to push array length.
81          new JLiteralInt(line, String.valueOf(initials.size()))
82              .codegen(output);
83  
84          // Code to create the (empty) array
85          output.addArrayInstruction(componentType.isReference()
86              ? ANEWARRAY
87              : NEWARRAY, componentType.jvmName());
88  
89          // Code to load initial values and store them as
90          // elements in the newly created array.
91          for (int i = 0; i < initials.size(); i++) {
92              JExpression initExpr = initials.get(i);
93  
94              // Duplicate the array for each element store
95              output.addNoArgInstruction(DUP);
96  
97              // Code to push index for store
98              new JLiteralInt(line, String.valueOf(i)).codegen(output);
99  
100             // Code to compute the initial value.
101             initExpr.codegen(output);
102 
103             // Code to store the initial value in the array
104             if (componentType == Type.INT) {
105                 output.addNoArgInstruction(IASTORE);
106             } else if (componentType == Type.BOOLEAN) {
107                 output.addNoArgInstruction(BASTORE);
108             } else if (componentType == Type.CHAR) {
109                 output.addNoArgInstruction(CASTORE);
110             } else if (!componentType.isPrimitive()) {
111                 output.addNoArgInstruction(AASTORE);
112             }
113         }
114     }
115 
116     /**
117      * {@inheritDoc}
118      */
119 
120     public void writeToStdOut(PrettyPrinter p) {
121         p.println("<JArrayInitializer>");
122         if (initials != null) {
123             for (JAST initial : initials) {
124                 p.indentRight();
125                 initial.writeToStdOut(p);
126                 p.indentLeft();
127             }
128         }
129         p.println("</JArrayInitializer>");
130     }
131 }
