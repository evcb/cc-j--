12  /**
13   * For representing j-- types. All types are represented underneath (in the
14   * classRep field) by Java objects of type {@code Class}. These objects 
15   * represent types in Java, so this should ease our interfacing with existing 
16   * Java classes.
17   * <p>
18   * Class types (reference types that are represented by the identifiers
19   * introduced in class declarations) are represented using {@link TypeName}. So 
20   * for now, every TypeName represents a class. In the future, TypeName could be 
21   * extended to represent interfaces or enumerations.
22   * <p>
23   * IdentifierTypes must be "resolved" at some point, so that all Types having
24   * the same name refer to the same Type object. The {@code resolve} method does 
25   * this.
26   */
27

93      /**
94       * Constructs a Type representation for a type from its (Java) Class
95       * representation. Make sure there is a unique Type for each unique type.

97       * @param classRep
98       *            the Java representation.
99       * @return the Type representation of this classRep.

102     public static Type typeFor(Class<?> classRep) {
103         if (types.get(descriptorFor(classRep)) == null) {
104             types.put(descriptorFor(classRep), new Type(classRep));
105         }
106         return types.get(descriptorFor(classRep));

241     /**
242      * Returns a list of this class' abstract methods.
243      * 
244      * It has abstract methods if:
245      * <ol>
246      *   <li>Any method declared in the class is abstract or 
247      *   <li>its superclass has an abstract method which is not overridden here.
248      * </ol>
249      * 
250      * @return a list of abstract methods.
251      */
252 
253     public ArrayList<Method> abstractMethods() {
254         ArrayList<Method> inheritedAbstractMethods = superClass() == null ? new ArrayList<Method>()
255                 : superClass().abstractMethods();
256         ArrayList<Method> abstractMethods = new ArrayList<Method>();
257         ArrayList<Method> declaredConcreteMethods = declaredConcreteMethods();
258         ArrayList<Method> declaredAbstractMethods = declaredAbstractMethods();
259         abstractMethods.addAll(declaredAbstractMethods);
260         for (Method method : inheritedAbstractMethods) {
261             if (!declaredConcreteMethods.contains(method)
262                     && !declaredAbstractMethods.contains(method)) {
263                 abstractMethods.add(method);
264             }
265         }
266         return abstractMethods;
267     }
268 
269     /**
270      * Returns a list of this class' declared abstract methods.
271      * 
272      * @return a list of declared abstract methods.
273      */
274 
275     private ArrayList<Method> declaredAbstractMethods() {
276         ArrayList<Method> declaredAbstractMethods = new ArrayList<Method>();
277         for (java.lang.reflect.Method method : classRep.getDeclaredMethods()) {
278             if (Modifier.isAbstract(method.getModifiers())) {
279                 declaredAbstractMethods.add(new Method(method));
280             }
281         }
282         return declaredAbstractMethods;
283     }

285     /**
286      * Returns a list of this class' declared concrete methods.
287      * 
288      * @return a list of declared concrete methods.
289      */
290 
291     private ArrayList<Method> declaredConcreteMethods() {
292         ArrayList<Method> declaredConcreteMethods = new ArrayList<Method>();
293         for (java.lang.reflect.Method method : classRep.getDeclaredMethods()) {
294             if (!Modifier.isAbstract(method.getModifiers())) {
295                 declaredConcreteMethods.add(new Method(method));
296             }
297         }
298         return declaredConcreteMethods;
299     }

301     /**
302      * An assertion that this type matches one of the specified types. If there
303      * is no match, an error message is returned.
304      * 
305      * @param line
306      *            the line near which the mismatch occurs.
307      * @param expectedTypes
308      *            expected types.
309      */
310 
311     public void mustMatchOneOf(int line, Type... expectedTypes) {
312         if (this == Type.ANY)
313             return;
314         for (int i = 0; i < expectedTypes.length; i++) {
315             if (matchesExpected(expectedTypes[i])) {
316                 return;
317             }
318         }
319         JAST.compilationUnit.reportSemanticError(line,
320                 "Type %s doesn't match any of the expected types %s", this,
321                 Arrays.toString(expectedTypes));
322     }

325      * An assertion that this type matches the specified type. If there is no
326      * match, an error message is written.

328      * @param line
329      *            the line near which the mismatch occurs.
330      * @param expectedType
331      *            type with which to match.
332      */
333 
334     public void mustMatchExpected(int line, Type expectedType) {
335         if (!matchesExpected(expectedType)) {
336             JAST.compilationUnit.reportSemanticError(line,
337                     "Type %s doesn't match type %s", this, expectedType);
338         }
339     }

352                 || (this == Type.NULLTYPE && expected.isReference())
353                 || this.equals(expected);

356     /**
357      * Do argument types match? A helper used for finding candidate methods and
358      * constructors.
359      * 
360      * @param argTypes1
361      *            arguments (classReps) of one method.
362      * @param argTypes2
363      *            arguments (classReps) of another method.
364      * @return {@code true} iff all corresponding types of argTypes1 and 
365      *         argTypes2 match; {@code false} otherwise.
366      */
367 
368     public static boolean argTypesMatch(Class<?>[] argTypes1,
369             Class<?>[] argTypes2) {
370         if (argTypes1.length != argTypes2.length) {
371             return false;
372         }
373         for (int i = 0; i < argTypes1.length; i++) {
374             if (!Type.descriptorFor(argTypes1[i]).equals(
375                     Type.descriptorFor(argTypes2[i]))) {
376                 return false;
377             }
378         }
379         return true;
380     }

416      * A helper translating a type's internal representation to its (JVM)

425         return cls == null ? "V" : cls == void.class ? "V"

428                                 : cls == char.class ? "C"
429                                         : cls == boolean.class ? "Z" : "?")
430                                 : "L" + cls.getName().replace('.', '/') + ";";
431     }

486     /**
487      * Finds an appropriate method in this type, given a message (method) name
488      * and its argument types. This is pretty easy given our (current)
489      * restriction that the types of the actual arguments must exactly match the
490      * types of the formal parameters. Returns null if it cannot find one.
491      * 
492      * @param name
493      *            the method name.
494      * @param argTypes
495      *            the argument types.
496      * @return Method with given name and argument types, or {@code null}.
497      */
498 
499     public Method methodFor(String name, Type[] argTypes) {
500         Class[] classes = new Class[argTypes.length];
501         for (int i = 0; i < argTypes.length; i++) {
502             classes[i] = argTypes[i].classRep;
503         }
504         Class cls = classRep;
505 
506         // Search this class and all superclasses
507         while (cls != null) {
508             java.lang.reflect.Method[] methods = cls.getDeclaredMethods();
509             for (java.lang.reflect.Method method : methods) {
510                 if (method.getName().equals(name)
511                         && Type.argTypesMatch(classes, method
512                                 .getParameterTypes())) {
513                     return new Method(method);
514                 }
515             }
516             cls = cls.getSuperclass();
517         }
518         return null;
519     }
520 
521     /**
522      * Finds an appropriate constructor in this type, given its argument types.
523      * This is pretty easy given our (current) restriction that the types of the
524      * actual arguments must exactly match the types of the formal parameters.
525      * Returns null if it cannot find one.
526      * 
527      * @param argTypes
528      *            the argument types.
529      * @return Constructor with the specified argument types, or {@code null}.
530      */
531 
532     public Constructor constructorFor(Type[] argTypes) {
533         Class[] classes = new Class[argTypes.length];
534         for (int i = 0; i < argTypes.length; i++) {
535             classes[i] = argTypes[i].classRep;
536         }
537 
538         // Search only this class (we don't inherit constructors)
539         java.lang.reflect.Constructor[] constructors = classRep
540                 .getDeclaredConstructors();
541         for (java.lang.reflect.Constructor constructor : constructors) {
542             if (argTypesMatch(classes, constructor.getParameterTypes())) {
543                 return new Constructor(constructor);
544             }
545         }
546         return null;
547     }

549     /**
550      * Returns the {@code Field} having this name.
551      * 
552      * @param name
553      *            the name of the field we want.
554      * @return the Field or {@code null} if it's not there.
555      */
556 
557     public Field fieldFor(String name) {
558         Class<?> cls = classRep;
559         while (cls != null) {
560             java.lang.reflect.Field[] fields = cls.getDeclaredFields();
561             for (java.lang.reflect.Field field : fields) {
562                 if (field.getName().equals(name)) {
563                     return new Field(field);
564                 }
565             }
566             cls = cls.getSuperclass();
567         }
568         return null;
569     }

573      * parenthesized list of the types, for example, (int, boolean,

594     /**
595      * Checks the accessibility of a member from this type (that is, this type 
596      * is the referencing type).
597      * 
598      * @param line
599      *            the line in which the access occurs.
600      * @param member
601      *            the member being accessed.
602      * @return {@code true} if access is valid; {@code false} otherwise.
603      */
604 
605     public boolean checkAccess(int line, Member member) {
606         if (!checkAccess(line, classRep, member.declaringType().classRep)) {
607             return false;
608         }
609 
610         // The member must be either public, protected, or private
611         if (member.isPublic()) {
612             return true;
613         }
614         java.lang.Package p1 = classRep.getPackage();
615         java.lang.Package p2 = member.declaringType().classRep.getPackage();
616         if ((p1 == null ? "" : p1.getName()).equals((p2 == null ? "" : p2
617                 .getName()))) {
618             return true;
619         }
620         if (member.isProtected()) {
621             if (classRep.getPackage().getName().equals(
622                     member.declaringType().classRep.getPackage().getName())
623                     || typeFor(member.getClass().getDeclaringClass())
624                             .isJavaAssignableFrom(this)) {
625                 return true;
626             } else {
627                 JAST.compilationUnit.reportSemanticError(line,
628                         "The protected member, " + member.name()
629                                 + ", is not accessible.");
630                 return false;
631             }
632         }
633         if (member.isPrivate()) {
634             if (descriptorFor(classRep).equals(
635                     descriptorFor(member.member().getDeclaringClass()))) {
636                 return true;
637             } else {
638                 JAST.compilationUnit.reportSemanticError(line,
639                         "The private member, " + member.name()
640                                 + ", is not accessible.");
641                 return false;
642             }
643         }
644 
645         // Otherwise, the member has default access
646         if (packageName().equals(member.declaringType().packageName())) {
647             return true;
648         } else {
649             JAST.compilationUnit.reportSemanticError(line, "The member, "
650                     + member.name()
651                     + ", is not accessible because it's in a different "
652                     + "package.");
653             return false;
654         }
655     }

657     /**
658      * Checks the accesibility of a target type (from this type).
659      * 
660      * @param line
661      *            line in which the access occurs.
662      * @param targetType
663      *            the type being accessed.
664      * @return {@code true} if access is valid; {@code false} otherwise.
665      */
666 
667     public boolean checkAccess(int line, Type targetType) {
668         if (targetType.isPrimitive()) {
669             return true;
670         }
671         if (targetType.isArray()) {
672             return this.checkAccess(line, targetType.componentType());
673         }
674         return checkAccess(line, classRep, targetType.classRep);
675     }
676 
677     /**
678      * Checks the accessibility of a type.
679      * 
680      * @param line
681      *            the line in which the access occurs.
682      * @param referencingType
683      *            the type attempting the access.
684      * @param type
685      *            the type that we want to access.
686      * @return {@code true} if access is valid; {@code false} otherwise.
687      */
688 
689     public static boolean checkAccess(int line, Class referencingType,
690             Class type) {
691         java.lang.Package p1 = referencingType.getPackage();
692         java.lang.Package p2 = type.getPackage();
693         if (Modifier.isPublic(type.getModifiers())
694                 || (p1 == null ? "" : p1.getName()).equals((p2 == null ? ""
695                         : p2.getName()))) {
696             return true;
697         } else {
698             JAST.compilationUnit.reportSemanticError(line, "The type, "
699                     + type.getCanonicalName() + ", is not accessible from "
700                     + referencingType.getCanonicalName());
701             return false;
702         }
703     }

720      * A helper for constructing method signatures for reporting unfound methods
721      * and constructors.
722      * 
723      * @param name
724      *            the message or Type name.
725      * @param argTypes
726      *            the actual argument types.

804     /**
805      * Returns the Java representation of this type. For example, 
806      * java.lang.String.
807      * 
808      * @return the qualified name.
809      */
810 
811     public String toString() {
812         return name;
813     }
814 
815     /**
816      * Returns the simple name for this type. For example, String for 
817      * java.lang.String.
818      * 
819      * @return the simple name.
820      */
821 
822     public String simpleName() {
823         return name.substring(name.lastIndexOf('.') + 1);
824     }
825 
826     /**
827      * Resolves this type in the given context. Notice that this has meaning 
828      * only for TypeName and ArrayTypeName, where names are replaced by real 
829      * types. Names are looked up in the context.
830      * 
831      * @param context
832      *            context in which the names are resolved.
833      * @return the resolved type.
834      */
835 
836     public Type resolve(Context context) {
837         Type resolvedType = context.lookupType(name);
838         if (resolvedType == null) {
839             // Try loading a type with the give fullname
840             try {
841                 resolvedType = typeFor(Class.forName(name));
842                 context.addType(line, resolvedType);
843                 // context.compilationUnitContext().addEntry(line,
844                 // resolvedType.toString(),
845                 // new TypeNameDefn(resolvedType));
846             } catch (Exception e) {
847                 JAST.compilationUnit.reportSemanticError(line,
848                         "Unable to locate a type named %s", name);
849                 resolvedType = Type.ANY;
850             }
851         }
852         if (resolvedType != Type.ANY) {
853             Type referencingType = ((JTypeDecl) (context.classContext
854                     .definition())).thisType();
855             Type.checkAccess(line, referencingType.classRep(), resolvedType
856                     .classRep());
857         }
858         return resolvedType;
859     }
860 }

862 /**
863  * The (temporary) representation of an array's type. It is built by the 
864  * {@link Parser} to stand in for a {@link Type} until the {@code analyze} 
865  * phase, at which point it is resolved to an actual Type object 
866  * (having a Class that identifies it).
867  */
868 
869 class ArrayTypeName extends Type {
870 
871     /** The array's base or component type. */
872     private Type componentType;
873 
874     /**
875      * Constructs an array's type given its component type.
876      * 
877      * @param componentType
878      *            the type of its elements.
879      */
880 
881     public ArrayTypeName(Type componentType) {
882         this.componentType = componentType;
883     }
884 
885     /**
886      * Returns the (component) type of its elements.
887      * 
888      * @return the component type.
889      */
890 
891     public Type componentType() {
892         return componentType;
893     }
894 
895     /**
896      * Returns the JVM descriptor for this type.
897      * 
898      * @return the descriptor.
899      */
900 
901     public String toDescriptor() {
902         return "[" + componentType.toDescriptor();
903     }
904 
905     /**
906      * A string representation of the type in Java form.
907      * 
908      * @return the representation in Java form.
909      */
910 
911     public String toString() {
912         return componentType.toString() + "[]";
913     }
914 
915     /**
916      * Resolves this type in the given context.
917      * 
918      * @param context
919      *            context in which the names are resolved.
920      * @return the resolved type.
921      */
922 
923     public Type resolve(Context context) {
924         componentType = componentType.resolve(context);
925 
926         // The API forces us to make an instance and get its
927         // type.
928         Class classRep = Array.newInstance(componentType().classRep(), 0)
929                 .getClass();
930         return Type.typeFor(classRep);
931     }
