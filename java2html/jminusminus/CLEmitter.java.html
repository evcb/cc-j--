170     /**
171      * Add the method created using addMethod() to the ClassFile structure. This
172      * involves adding an instance of CLMethodInfo to ClassFile.methods for the
173      * method.
174      */
175 
176     private void endOpenMethodIfAny() {
177         if (isMethodOpen) {
178             isMethodOpen = false;
179             if (!mInstructionAfterLabel) {
180                 // Must jump to an instruction
181                 addNoArgInstruction(NOP);
182             }
183 
184             // Resolve jump labels in exception handlers
185             ArrayList<CLExceptionInfo> exceptionTable = new ArrayList<CLExceptionInfo>();
186             for (int i = 0; i < mExceptionHandlers.size(); i++) {
187                 CLException e = mExceptionHandlers.get(i);
188                 if (!e.resolveLabels(mLabels)) {
189                     reportEmitterError(
190                             "%s: Unable to resolve exception handler "
191                                     + "label(s)", eCurrentMethod);
192                 }
193 
194                 // We allow catchType to be null (mapping to index 0),
195                 // implying this exception handler is called for all
196                 // exceptions. This is used to implement "finally"
197                 int catchTypeIndex = (e.catchType == null) ? 0 : constantPool
198                         .constantClassInfo(e.catchType);
199                 CLExceptionInfo c = new CLExceptionInfo(e.startPC, e.endPC,
200                         e.handlerPC, catchTypeIndex);
201                 exceptionTable.add(c);
202             }
203 
204             // Convert Instruction objects to bytes
205             ArrayList<Integer> byteCode = new ArrayList<Integer>();
206             int maxLocals = mArgumentCount;
207             for (int i = 0; i < mCode.size(); i++) {
208                 CLInstruction instr = mCode.get(i);
209 
210                 // Compute maxLocals
211                 int localVariableIndex = instr.localVariableIndex();
212                 switch (instr.opcode()) {
213                 case LLOAD:
214                 case LSTORE:
215                 case DSTORE:
216                 case DLOAD:
217                 case LLOAD_0:
218                 case LLOAD_1:
219                 case LLOAD_2:
220                 case LLOAD_3:
221                 case LSTORE_0:
222                 case LSTORE_1:
223                 case LSTORE_2:
224                 case LSTORE_3:
225                 case DLOAD_0:
226                 case DLOAD_1:
227                 case DLOAD_2:
228                 case DLOAD_3:
229                 case DSTORE_0:
230                 case DSTORE_1:
231                 case DSTORE_2:
232                 case DSTORE_3:
233                     // Each long and double occupies two slots in the
234                     //   local variable table
235                     localVariableIndex++;
236                 }
237                 maxLocals = Math.max(maxLocals, localVariableIndex + 1);
238 
239                 // Resolve jump labels in flow control instructions
240                 if (instr instanceof CLFlowControlInstruction) {
241                     if (!((CLFlowControlInstruction) instr)
242                             .resolveLabels(mLabels)) {
243                         reportEmitterError(
244                                 "%s: Unable to resolve jump label(s)",
245                                 eCurrentMethod);
246                     }
247                 }
248 
249                 byteCode.addAll(instr.toBytes());
250             }
251 
252             // Code attribute; add only if method is neither native nor abstract
253             if (!((mAccessFlags & ACC_NATIVE) == ACC_NATIVE ||
254                   (mAccessFlags & ACC_ABSTRACT) == ACC_ABSTRACT)) {
255                 addMethodAttribute(codeAttribute(byteCode, exceptionTable,
256                         stackDepth(), maxLocals));
257             }
258 
259             methods.add(new CLMethodInfo(mAccessFlags, mNameIndex,
260                     mDescriptorIndex, mAttributes.size(), mAttributes));
261         }
262 
263         // This method could be the last method, so we need
264         // the following wrap up code
265 
266         // Add the InnerClass attribute if this class has inner
267         // classes
268         if (innerClasses.size() > 0) {
269             addClassAttribute(innerClassesAttribute());
270         }
271 
272         // Set the members of the ClassFile structure to their
273         // appropriate values
274         clFile.constantPoolCount = constantPool.size() + 1;
275         clFile.constantPool      = constantPool;
276         clFile.interfacesCount   = interfaces.size();
277         clFile.interfaces        = interfaces;
278         clFile.fieldsCount       = fields.size();
279         clFile.fields            = fields;
280         clFile.methodsCount      = methods.size();
281         clFile.methods           = methods;
282         clFile.attributesCount   = attributes.size();
283         clFile.attributes        = attributes;
284     }

286     /**
287      * Add a field.
288      * 
289      * @param accessFlags
290      *            access flags for the field.
291      * @param name
292      *            name of the field.
293      * @param type
294      *            type descriptor for the field.
295      * @param isSynthetic
296      *            is this a synthetic field?
297      * @param c
298      *            index into the constant pool at which there is a
299      *            ConstantIntegerInfo, ConstantFloatInfo, ConstantLongInfo,
300      *            ConstantDoubleInfo or ConstantStringInfo object. c is -1 if
301      *            the field does not have an initialization.
302      */
303 
304     private void addFieldInfo(ArrayList<String> accessFlags, String name,
305             String type, boolean isSynthetic, int c) {
306         if (!validTypeDescriptor(type)) {
307             reportEmitterError("'%s' is not a valid type descriptor for field",
308                     type);
309         }
310         int flags = 0;
311         int nameIndex = constantPool.constantUtf8Info(name);
312         int descriptorIndex = constantPool.constantUtf8Info(type);
313         fAttributes = new ArrayList<CLAttributeInfo>();
314         if (accessFlags != null) {
315             for (int i = 0; i < accessFlags.size(); i++) {
316                 flags |= CLFile.accessFlagToInt(accessFlags.get(i));
317             }
318         }
319         if (isSynthetic) {
320             addFieldAttribute(syntheticAttribute());
321         }
322         if (c != -1) {
323             addFieldAttribute(constantValueAttribute(c));
324         }
325         fields.add(new CLFieldInfo(flags, nameIndex, descriptorIndex,
326                 fAttributes.size(), fAttributes));
327     }

340     private int typeStackResidue(String descriptor) {
341         int i = 0;
342         char c = descriptor.charAt(0);
343         switch (c) {
344         case 'B':
345         case 'C':
346         case 'I':
347         case 'F':
348         case 'L':
349         case 'S':
350         case 'Z':
351         case '[':
352             i = 1;
353             break;
354         case 'J':
355         case 'D':
356             i = 2;
357             break;
358         }
359         return i;
360     }

362     /**
363      * Return the difference between the number of units consumed from the
364      * operand stack by a method with the specified descriptor and the number of
365      * units produced by the method in the operand stack. 0 is returned if the
366      * descriptor is invalid.
367      * 
368      * @param descriptor
369      *            a method descriptor.
370      * @return number of units consumed from the operand stack - number of units
371      *         produced in the stack.
372      */
373 
374     private int methodStackResidue(String descriptor) {
375         int i = 0;
376 
377         // Extract types of arguments and the return type from
378         // the method descriptor
379         String argTypes = descriptor.substring(1, descriptor.lastIndexOf(")"));
380         String returnType = descriptor
381                 .substring(descriptor.lastIndexOf(")") + 1);
382 
383         // Units consumed
384         for (int j = 0; j < argTypes.length(); j++) {
385             char c = argTypes.charAt(j);
386             switch (c) {
387             case 'B':
388             case 'C':
389             case 'I':
390             case 'F':
391             case 'S':
392             case 'Z':
393                 i -= 1;
394                 break;
395             case '[':
396                 break;
397             case 'J':
398             case 'D':
399                 i -= 2;
400                 break;
401             case 'L':
402                 int k = argTypes.indexOf(";", j);
403                 j = k;
404                 i -= 1;
405                 break;
406             }
407         }
408 
409         // Units produced
410         i += typeStackResidue(returnType);
411         return i;
412     }

458     /**
459      * Return true if the specified name is in the internal form of a fully
460      * qualified class or interface name [JVMS 4.3], false otherwise.
461      * 
462      * @param name
463      *            fully qualified class name in internal form.
464      * @return true if the specified name is in the internal form of a fully
465      *         qualified class or interface name, false otherwise.
466      */
467 
468     private boolean validInternalForm(String name) {
469         if ((name == null) || name.equals("") ||
470              name.startsWith("/") || name.endsWith("/")) {
471             return false;
472         }
473         StringTokenizer t = new StringTokenizer(name, "/");
474         while (t.hasMoreTokens()) {
475             String s = t.nextToken();
476             for (int i = 0; i < s.length(); i++) {
477                 if (i == 0) {
478                     if (!Character.isJavaIdentifierStart(s.charAt(i))) {
479                         return false;
480                     }
481                 } else {
482                     if (!Character.isJavaIdentifierPart(s.charAt(i))) {
483                         return false;
484                     }
485                 }
486             }
487         }
488         return true;
489     }

491     /**
492      * Return true if the specified descriptor is a valid type descriptor [JVMS
493      * 4.4.2], false otherwise.
494      * 
495      * @param descriptor
496      *            type descriptor.
497      * @return true if the specified descriptor is a valid type descriptor,
498      *         false otherwise.
499      */
500 
501     private boolean validTypeDescriptor(String descriptor) {
502         if (descriptor != null) {
503             try {
504                 char c = descriptor.charAt(0);
505                 switch (c) {
506                 case 'B':
507                 case 'C':
508                 case 'I':
509                 case 'F':
510                 case 'S':
511                 case 'Z':
512                 case 'J':
513                 case 'D':
514                     return (descriptor.length() == 1);
515                 case 'L':
516                     if (descriptor.endsWith(";")) {
517                         return validInternalForm(descriptor.substring(1,
518                                 descriptor.length() - 1));
519                     }
520                     return false;
521                 case '[':
522                     return validTypeDescriptor(descriptor.substring(1));
523                 }
524             } catch (IndexOutOfBoundsException e) {
525                 return false;
526             }
527         }
528         return false;
529     }

531     /**
532      * Return true if the specified descriptor is a valid method descriptor
533      * [JVMS 4.4.3], false otherwise.
534      * 
535      * @param descriptor
536      *            method descriptor.
537      * @return true if the specified descriptor is a valid method descriptor,
538      *         false otherwise.
539      */
540 
541     private boolean validMethodDescriptor(String descriptor) {
542         if ((descriptor != null) && (descriptor.length() > 0)) {
543             try {
544                 // Extract types of arguments and the return type
545                 // from the method descriptor
546                 String argTypes = descriptor.substring(1, descriptor
547                         .lastIndexOf(")"));
548                 String returnType = descriptor.substring(descriptor
549                         .lastIndexOf(")") + 1);
550 
551                 // Validate argument type syntax
552                 if (argTypes.endsWith("[")) {
553                     return false;
554                 }
555                 for (int i = 0; i < argTypes.length(); i++) {
556                     char c = argTypes.charAt(i);
557                     switch (c) {
558                     case 'B':
559                     case 'C':
560                     case 'I':
561                     case 'F':
562                     case 'S':
563                     case 'Z':
564                     case 'J':
565                     case 'D':
566                     case '[':
567                         break;
568                     case 'L':
569                         int j = argTypes.indexOf(";", i);
570                         String s = argTypes.substring(i, j + 1);
571                         i = j;
572                         if (!validTypeDescriptor(s)) {
573                             return false;
574                         }
575                         break;
576                     default:
577                         return false;
578                     }
579                 }
580 
581                 // Validate return type syntax
582                 return (returnType.equals("V") || validTypeDescriptor(returnType));
583             } catch (IndexOutOfBoundsException e) {
584                 return false;
585             }
586         }
587         return false;
588     }

629     /**
630      * Compute the maximum depth of the operand stack for the method last added,
631      * and return the value.
632      * 
633      * @return maximum depth of operand stack.
634      */
635 
636     private int stackDepth() {
637         CLBranchStack branchTargets = new CLBranchStack();
638         for (int i = 0; i < mExceptionHandlers.size(); i++) {
639             CLException e = mExceptionHandlers.get(i);
640             CLInstruction h = instruction(e.handlerPC);
641             if (h != null) {
642                 // 1 because the exception that is thrown is
643                 //   pushed on top of the operand stack
644                 branchTargets.push(h, 1);
645             }
646         }
647         int stackDepth = 0, maxStackDepth = 0, c = 0;
648         CLInstruction instr = (mCode.size() == 0) ? null : mCode.get(c);
649         while (instr != null) {
650             int opcode = instr.opcode();
651             int stackUnits = instr.stackUnits();
652             if (stackUnits == EMPTY_STACK) {
653                 stackDepth = 0;
654             } else if (stackUnits == UNIT_SIZE_STACK) {
655                 stackDepth = 1;
656             } else {
657                 stackDepth += stackUnits;
658             }
659             if (stackDepth > maxStackDepth) {
660                 maxStackDepth = stackDepth;
661             }
662 
663             // For tracing purposes
664             // System.out.println( instr.mnemonic() + ", " +
665             // stackUnits + ", " +
666             // stackDepth + ", " + maxStackDepth );
667 
668             if (instr instanceof CLFlowControlInstruction) {
669                 CLFlowControlInstruction b = (CLFlowControlInstruction) instr;
670                 int jumpToIndex = b.pc() + b.jumpToOffset();
671                 CLInstruction instrAt = null;
672                 switch (opcode) {
673                 case JSR:
674                 case JSR_W:
675                 case RET:
676                     instr = null;
677                     break;
678                 case GOTO:
679                 case GOTO_W:
680                     instr = null;
681                 default:
682                     instrAt = instruction(jumpToIndex);
683                     if (instrAt != null) {
684                         branchTargets.push(instrAt, stackDepth);
685                     }
686                 }
687             } else {
688                 if ((opcode == ATHROW) ||
689                     ((opcode >= IRETURN) && (opcode <= RETURN))) {
690                     instr = null;
691                 }
692             }
693             if (instr != null) {
694                 c++;
695                 instr = (c >= mCode.size()) ? null : mCode.get(c);
696             }
697             if (instr == null) {
698                 CLBranchTarget bt = branchTargets.pop();
699                 if (bt != null) {
700                     instr = bt.target;
701                     stackDepth = bt.stackDepth;
702                     c = instructionIndex(instr.pc());
703                 }
704             }
705         }
706         return maxStackDepth;
707     }

713      *            index into the constant pool, the item at which is a

727         mCode.add(instr);

764     /**
765      * Construct and return a Code attribute given the list of bytes that make
766      * up the instructions and their operands, exception table, maximum depth of
767      * operand stack, and maximum number of local variables.
768      * 
769      * @param byteCode
770      *            list of bytes that make up the instructions and their
771      *            operands.
772      * @param exceptionTable
773      *            exception table.
774      * @param stackDepth
775      *            maximum depth of operand stack.
776      * @param maxLocals
777      *            maximum number of local variables.
778      * @return a Code attribute.
779      */
780 
781     private CLCodeAttribute codeAttribute(ArrayList<Integer> byteCode,
782             ArrayList<CLExceptionInfo> exceptionTable, int stackDepth,
783             int maxLocals) {
784         int codeLength = byteCode.size();
785         int attributeNameIndex = constantPool.constantUtf8Info(ATT_CODE);

922      * <p>
923      * This method instantiates a class file representation in memory, so this
924      * method <em>must</em> be called prior to methods that add information
925      * (fields, methods, instructions, etc.) to the class.
926      * 
927      * @param accessFlags
928      *            the access flags for the class or interface.
929      * @param thisClass
930      *            fully qualified name of the class or interface in internal
931      *            form.
932      * @param superClass
933      *            fully qualified name of the parent class in internal form.
934      * @param superInterfaces
935      *            list of direct super interfaces of this class or interface as
936      *            fully qualified names in internal form.
937      * @param isSynthetic
938      *            whether the class or interface is synthetic.
939      */
940 
941     public void addClass(ArrayList<String> accessFlags, String thisClass,
942                          String superClass, ArrayList<String> superInterfaces,
943                                             boolean isSynthetic) {
944         clFile       = new CLFile();
945         constantPool = new CLConstantPool();
946         interfaces   = new ArrayList<Integer>();
947         fields       = new ArrayList<CLFieldInfo>();
948         methods      = new ArrayList<CLMethodInfo>();
949         attributes   = new ArrayList<CLAttributeInfo>();
950         innerClasses = new ArrayList<CLInnerClassInfo>();
951 
952         errorHasOccurred = false;
953         clFile.magic = MAGIC;
954         clFile.majorVersion = MAJOR_VERSION;
955         clFile.minorVersion = MINOR_VERSION;
956         if (!validInternalForm(thisClass)) {
957             reportEmitterError("'%s' is not in internal form", thisClass);
958         }
959         if (!validInternalForm(superClass)) {
960             reportEmitterError("'%s' is not in internal form", superClass);
961         }
962         if (accessFlags != null) {
963             for (int i = 0; i < accessFlags.size(); i++) {
964                 clFile.accessFlags |= CLFile
965                         .accessFlagToInt(accessFlags.get(i));
966             }
967         }
968         name = thisClass;
969         clFile.thisClass = constantPool.constantClassInfo(thisClass);
970         clFile.superClass = constantPool.constantClassInfo(superClass);
971         for (int i = 0; superInterfaces != null && i < superInterfaces.size(); i++) {

973                 reportEmitterError("'%s' is not in internal form",
974                         superInterfaces.get(i));
975             }
976             interfaces.add(new Integer(constantPool
977                     .constantClassInfo(superInterfaces.get(i))));
978         }
979         if (isSynthetic) {
980             addClassAttribute(syntheticAttribute());
981         }
982     }

1452     * 
1453     * @param opcode
1454     *            opcode of the instruction.
1455     * @param target
1456     *            fully qualified name in internal form of the class to which
1457     *            the member belongs.
1458     * @param name
1459     *            name of the member.
1460     * @param type
1461     *            type descriptor of the member.
1462     */
1463
1464    public void addMemberAccessInstruction(int opcode, String target,
1465            String name, String type) {
1466        if (!validInternalForm(target)) {
1467            reportEmitterError("%s: '%s' is not in internal form",
1468                    eCurrentMethod, target);
1469        }
1470        CLInstruction instr = null;
1471        int index, stackUnits;
1472        switch (CLInstruction.instructionInfo[opcode].category) {
1473        case FIELD:
1474            if (!validTypeDescriptor(type)) {
1475                reportEmitterError(
1476                        "%s: '%s' is not a valid type descriptor for field",
1477                        eCurrentMethod, type);
1478            }
1479            index = constantPool.constantFieldRefInfo(target, name, type);
1480            stackUnits = typeStackResidue(type);
1481            if ((opcode == GETFIELD) || (opcode == PUTFIELD)) {
1482                // This is because target of this method is also
1483                // consumed from the operand stack
1484                stackUnits--;
1485            }
1486            instr = new CLFieldInstruction(opcode, mPC++, index, stackUnits);
1487            break;
1488        case METHOD1:
1489            if (!validMethodDescriptor(type)) {
1490                reportEmitterError(
1491                        "%s: '%s' is not a valid type descriptor for "
1492                                + "method", eCurrentMethod, type);
1493            }
1494            if (opcode == INVOKEINTERFACE) {
1495                index = constantPool.constantInterfaceMethodRefInfo(target,
1496                        name, type);
1497            } else {
1498                index = constantPool.constantMethodRefInfo(target, name, type);
1499            }
1500            stackUnits = methodStackResidue(type);
1501            if (opcode != INVOKESTATIC) {
1502                // This is because target of this method is also
1503                // consumed from the operand stack
1504                stackUnits--;
1505            }
1506            instr = new CLMethodInstruction(opcode, mPC++, index, stackUnits);
1507
1508            // INVOKEINTERFACE expects the number of arguments in
1509            // the method to be specified explicitly.
1510            if (opcode == INVOKEINTERFACE) {
1511                // We add 1 to account for "this"
1512                ((CLMethodInstruction) instr)
1513                        .setArgumentCount(argumentCount(type) + 1);
1514            }
1515            break;
1516        default:
1517            reportOpcodeError(opcode);
1518        }
1519        if (instr != null) {
1520            mPC += instr.operandCount();
1521            mCode.add(instr);
1522        }
1523    }
