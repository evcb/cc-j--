5   import java.util.Map;
6   import java.util.HashMap;
7   import java.util.Set;

9   /**
10   * A Context encapsulates the environment in which an AST is analyzed. It
11   * represents a scope; the scope of a variable is captured by its context. It's
12   * the symbol table.
13   * <p>
14   * Because scopes are lexically nested in Java (and so in j--), the environment
15   * can be seen as a stack of contexts, each of which is a mapping from names to
16   * their definitions ({@link IDefn}s). A Context keeps track of its (most 
17   * closely) surrounding context, its surrounding class context, and its 
18   * surrounding compilation unit context, and as well as a map from names to 
19   * definitions in the level of scope that the Context represents. Contexts are 
20   * created for the compilation unit (a {@link CompilationUnitContext}), a class
21   * (a {@link ClassContext}), each method (a {@link MethodContext}), and each 
22   * block (a LocalContext). If we were to add the for-statement to j--, we would
23   * necessarily create a (local) context.
24   * <p>
25   * From the outside, the structure looks like a tree strung over the AST. But
26   * from any location on the AST, that is from any point along a particular
27   * branch, it looks like a stack of context objects leading back to the root of
28   * the AST, that is, back to the JCompilationUnit object at the root.
29   * <p>
30   * Part of this structure is built during pre-analysis; pre-analysis reaches
31   * only into the type (for example a class) declaration for typing the members;
32   * pre-analysis does not reach into the method bodies. The rest of it is built
33   * during analysis.
34   */
35  
36  class Context {
37  
38      /** The surrounding context (scope). */
39      protected Context surroundingContext;
40  
41      /** The surrounding class context. */
42      protected ClassContext classContext;
43  
44      /**
45       * The compilation unit context (for the whole source program or file).
46       */
47      protected CompilationUnitContext compilationUnitContext;
48  
49      /**
50       * Map of (local variable, formal parameters, type) names to their
51       * definitions.
52       */
53      protected Map<String, IDefn> entries;
54  
55      /**
56       * Constructs a Context.
57       * 
58       * @param surrounding
59       *            the surrounding context (scope).
60       * @param classContext
61       *            the surrounding class context.
62       * @param compilationUnitContext
63       *            the compilation unit context (for the whole source program or
64       *            file).
65       */
66  
67      protected Context(Context surrounding, ClassContext classContext,
68              CompilationUnitContext compilationUnitContext) {
69          this.surroundingContext = surrounding;
70          this.classContext = classContext;
71          this.compilationUnitContext = compilationUnitContext;
72          this.entries = new HashMap<String, IDefn>();
73      }
74  
75      /**
76       * Adds an entry to the symbol table, binding a name to its definition in the
77       * current context.
78       * 
79       * @param line
80       *            the line number of the entry.
81       * @param name
82       *            the name being declared.
83       * @param definition
84       *            and its definition.
85       */
86  
87      public void addEntry(int line, String name, IDefn definition) {
88          if (entries.containsKey(name)) {
89              JAST.compilationUnit.reportSemanticError(line, "redefining name: "
90                      + name);
91          } else {
92              entries.put(name, definition);
93          }
94      }
95  
96      /**
97       * Returns the definition for a name in the environment. If it's not found in
98       * this context, we look for it in the surrounding context(s).
99       * 
100      * @param name
101      *            the name whose definition we're looking for.
102      * @return the definition; {@code null}, if not found.
103      */
104 
105     public IDefn lookup(String name) {
106         IDefn iDefn = (IDefn) entries.get(name);
107         return iDefn != null ? iDefn
108                 : surroundingContext != null ? surroundingContext.lookup(name)
109                         : null;
110     }
111 
112     /**
113      * Returns the definition for a type name in the environment. For now, we
114      * look for types only in the {@code CompilationUnitContext}.
115      * 
116      * @param name
117      *            the name of the type whose definition we're looking for.
118      * @return the definition; {@code null}, if not found.
119      */
120 
121     public Type lookupType(String name) {
122         TypeNameDefn defn = (TypeNameDefn) compilationUnitContext.lookup(name);
123         return defn == null ? null : defn.type();
124     }
125 
126     /**
127      * Adds the type to the environment.
128      * 
129      * @param line
130      *            line number of type declaration.
131      * @param type
132      *            the type we are declaring.
133      */
134 
135     public void addType(int line, Type type) {
136         IDefn iDefn = new TypeNameDefn(type);
137         compilationUnitContext.addEntry(line, type.simpleName(), iDefn);
138         if (!type.toString().equals(type.simpleName())) {
139             compilationUnitContext.addEntry(line, type.toString(), iDefn);
140         }
141     }
142 
143     /**
144      * Returns the type that defines this context (used principally for checking
145      * acessibility).
146      * 
147      * @return the type that defines this context.
148      */
149 
150     public Type definingType() {
151         return ((JTypeDecl) classContext.definition()).thisType();
152     }
153 
154     /**
155      * Returns the surrounding context (scope) in the stack of contexts.
156      * 
157      * @return the surrounding context.
158      */
159 
160     public Context surroundingContext() {
161         return surroundingContext;
162     }
163 
164     /**
165      * Returns the surrounding class context.
166      * 
167      * @return the surrounding class context.
168      */
169 
170     public ClassContext classContext() {
171         return classContext;
172     }
173 
174     /**
175      * Returns the surrounding compilation unit context. This is where imported
176      * types and other types defined in the compilation unit are declared.
177      * 
178      * @return the compilation unit context.
179      */
180 
181     public CompilationUnitContext compilationUnitContext() {
182         return compilationUnitContext;
183     }
184 
185     /**
186      * Returns the closest surrounding method context. Returns null if we're not
187      * within a method.
188      * 
189      * @return the method context.
190      */
191 
192     public MethodContext methodContext() {
193         Context context = this;
194         while (context != null && !(context instanceof MethodContext)) {
195             context = context.surroundingContext();
196         }
197         return (MethodContext) context;
198     }
199 
200     /**
201      * Returns the names declared in this context.
202      * 
203      * @return the set of declared names.
204      */
205 
206     public Set<String> names() {
207         return entries.keySet();
208     }
209 
210     /**
211      * Writes the contents of this context to STDOUT.
212      * 
213      * @param p
214      *            for pretty printing with indentation.
215      */
216 
217     public void writeToStdOut(PrettyPrinter p) {
218         // Nothing to write here
219     }
220 
221 }
222 
223 /**
224  * The compilation unit context is always the outermost context and is where
225  * imported types and locally defined types (classes) are declared.
226  */
227 
228 class CompilationUnitContext extends Context {
229 
230     /**
231      * Constructs a new compilation unit context. There are no surrounding
232      * contexts.
233      */
234 
235     public CompilationUnitContext() {
236         super(null, null, null);
237         compilationUnitContext = this;
238     }
239 
240     /**
241      * {@inheritDoc}
242      */
243 
244     public void writeToStdOut(PrettyPrinter p) {
245         p.println("<CompilationUnitContext>");
246         p.indentRight();
247         p.println("<Entries>");
248         if (entries != null) {
249             p.indentRight();
250             for (String key : names()) {
251                 p.println("<Entry>" + key + "</Entry>");
252             }
253             p.indentLeft();
254         }
255         p.println("</Entries>");
256         p.indentLeft();
257         p.println("</CompilationUnitContext>");
258     }
259 
260 }

262 /**
263  * Represents the context (scope, environment, symbol table) for a type, for 
264  * example a class, in j--. It also keeps track of its surrounding context(s)
265  * and the type whose context it represents.
266  */
267 
268 class ClassContext extends Context {
269 
270     /** AST node of the type that this class represents. */
271     private JAST definition;
272 
273     /**
274      * Constructs a class context.
275      * 
276      * @param definition
277      *            the AST node of the type that this class represents.
278      * @param surrounding
279      *            the surrounding context(s).
280      */
281 
282     public ClassContext(JAST definition, Context surrounding) {
283         super(surrounding, null, surrounding.compilationUnitContext());
284         classContext = this;
285         this.definition = definition;
286     }
287 
288     /**
289      * Returns the AST node of the type defined by this class.
290      * 
291      * @return the AST of the type defined by this class.
292      */
293 
294     public JAST definition() {
295         return definition;
296     }
297 
298 }

300 /**
301  * A local context is a context (scope) in which local variables (including
302  * formal parameters) can be declared. Local variables are allocated at fixed
303  * offsets from the base of the current method's stack frame; this is done
304  * during analysis. The definitions for local variables record these offsets. The
305  * offsets are used in code generation.
306  */
307 
308 class LocalContext extends Context {
309 
310     /** Next offset for a local variable. */
311     protected int offset;
312 
313     /**
314      * Constructs a local context. A local context is constructed for each block.
315      * 
316      * @param surrounding
317      *            the surrounding context.
318      */
319 
320     public LocalContext(Context surrounding) {
321         super(surrounding, surrounding.classContext(), surrounding
322                 .compilationUnitContext());
323         offset = (surrounding instanceof LocalContext) ? ((LocalContext) surrounding)
324                 .offset()
325                 : 0;
326     }
327 
328     /**
329      * The "next" offset. A simple getter. Not to be used for allocating new
330      * offsets (the {@code nextOffset} method is used for that).
331      * 
332      * @return the next available offset.
333      */
334 
335     public int offset() {
336         return offset;
337     }
338 
339     /**
340      * Allocates a new offset (for example, for a parameter or local variable).
341      * 
342      * @return the next allocated offset.
343      */
344 
345     public int nextOffset() {
346         return offset++;
347     }
348 
349     /**
350      * {@inheritDoc}
351      */
352 
353     public void writeToStdOut(PrettyPrinter p) {
354         p.println("<LocalContext>");
355         p.indentRight();
356         p.println("<Entries>");
357         if (entries != null) {
358             p.indentRight();
359             for (String key : names()) {
360                 IDefn defn = entries.get(key);
361                 if (defn instanceof LocalVariableDefn) {
362                     p.printf("<Entry name=\"%s\" " + "offset=\"%d\"/>\n", key,
363                             ((LocalVariableDefn) defn).offset());
364                 }
365             }
366             p.indentLeft();
367         }
368         p.println("</Entries>");
369         p.indentLeft();
370         p.println("</LocalContext>");
371     }
372 
373 }

375 /**
376  * A method context is where formal parameters are declared. Also, it's where we
377  * start computing the offsets for local variables (formal parameters included),
378  * which are allocated in the current stack frame (for a method invocation).
379  */
380 
381 class MethodContext extends LocalContext {
382 
383     /** Is this method static? */
384     private boolean isStatic;
385 
386     /** Return type of this method. */
387     private Type methodReturnType;
388 
389     /** Does (non-void) method have at least one return? */
390     private boolean hasReturnStatement = false;
391 
392     /**
393      * Constructs a method context.
394      * 
395      * @param surrounding
396      *            the surrounding (class) context.
397      * @param isStatic
398      *            is this method static?
399      * @param methodReturnType
400      *            return type of this method.
401      */
402 
403     public MethodContext(Context surrounding, boolean isStatic,
404             Type methodReturnType) {
405         super(surrounding);
406         this.isStatic = isStatic;
407         this.methodReturnType = methodReturnType;
408         offset = 0;
409     }
410 
411     /**
412      * Is this method static?
413      * 
414      * @return true or false.
415      */
416 
417     public boolean isStatic() {
418         return isStatic;
419     }
420 
421     /**
422      * Records fact that (non-void) method has at least one return.
423      */
424 
425     public void confirmMethodHasReturn() {
426         hasReturnStatement = true;
427     }
428 
429     /**
430      * Does this (non-void) method have at least one return?
431      * 
432      * @return true or false.
433      */
434 
435     public boolean methodHasReturn() {
436         return hasReturnStatement;
437     }
438 
439     /**
440      * Returns the return type of this method.
441      * 
442      * @return the return type of this method.
443      */
444 
445     public Type methodReturnType() {
446         return methodReturnType;
447     }
448 
449     /**
450      * {@inheritDoc}
451      */
452 
453     public void writeToStdOut(PrettyPrinter p) {
454         p.println("<MethodContext>");
455         p.indentRight();
456         super.writeToStdOut(p);
457         p.indentLeft();
458         p.println("</MethodContext>");
459     }
460 
461 }
