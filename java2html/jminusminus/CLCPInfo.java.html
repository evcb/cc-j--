9   /**
10   * Representation of cp_info structure (JVM Spec Section 4.5). Classes
11   * representing individual constant pool items inherit this class. Instances of
12   * these classes are created and populated into the constant pool table when a
13   * class is read using {@link CLAbsorber} or constructed using 
14   * {@link CLEmitter}.
15   */
16  
17  abstract class CLCPInfo {
18  
19      // The fields below represent the members of the cp_info
20      // structure and are thus inherited by the child classes of
21      // CLCPInfo. These classes define their own fields (if any)
22      // representing the members of the individual cp_info
23      // structures they represent.
24  
25      /** Index of this object into the constant pool. */
26      public int cpIndex;
27  
28      /** cp_info.tag item. */
29      public short tag;
30  
31      /**
32       * Writes the contents of this constant pool item to the specified output
33       * stream.
34       * 
35       * @param out
36       *            output stream.
37       * @throws IOException
38       *             if an error occurs while writing.
39       */
40  
41      public void write(CLOutputStream out) throws IOException {
42          out.writeByte(tag);
43      }
44  
45      /**
46       * Writes the content of this object to STDOUT in a format similar to that of
47       * javap.
48       * 
49       * @param p
50       *            for pretty printing with indentation.
51       */
52  
53      public void writeToStdOut(PrettyPrinter p) {
54          p.printf("%-10s", cpIndex);
55      }
56  
57  }

119 abstract class CLConstantMemberRefInfo extends CLCPInfo {

126 
127     /**

130      * @param classIndex
131      *            CONSTANT_Memberref_info.class_index item.
132      * @param nameAndTypeIndex
133      *            CONSTANT_Memberref_info.name_and_type_index item.
134      * @param tag
135      *            CONSTANT_Memberref_info.tag item.
136      */
137 
138     protected CLConstantMemberRefInfo(int classIndex, int nameAndTypeIndex,
139             short tag) {
140         super.tag = tag;
141         this.classIndex = classIndex;
142         this.nameAndTypeIndex = nameAndTypeIndex;
143     }
144 
145     /**
146      * {@inheritDoc}
147      */
148 
149     public void write(CLOutputStream out) throws IOException {
150         super.write(out);
151         out.writeShort(classIndex);
152         out.writeShort(nameAndTypeIndex);
153     }
154 
155     /**
156      * {@inheritDoc}
157      */
158 
159     public boolean equals(Object obj) {
160         if (obj instanceof CLConstantMemberRefInfo) {
161             CLConstantMemberRefInfo c = (CLConstantMemberRefInfo) obj;
162             if ((c.tag == tag) && (c.classIndex == classIndex)
163                     && (c.nameAndTypeIndex == nameAndTypeIndex)) {
164                 return true;
165             }
166         }
167         return false;
168     }

324 
325     /** The int number. */
326     public int i;
327 
328     /**
329      * Constructs a CLConstantIntegerInfo object.
330      * 
331      * @param i
332      *            the int number.
333      */
334 
335     public CLConstantIntegerInfo(int i) {
336         super.tag = CONSTANT_Integer;
337         this.i = i;
338     }
339 
340     /**
341      * Returns CONSTANT_Integer_info.bytes item.
342      * 
343      * @return CONSTANT_Integer_info.bytes item.
344      */
345 
346     public short[] bytes() {
347         short[] s = new short[4];
348         short mask = 0xFF;
349         int k = i;
350         for (int j = 0; j < 4; j++) {
351             s[3 - j] = (short) (k & mask);
352             k >>>= 8;
353         }
354         return s;
355     }
356 
357     /**
358      * {@inheritDoc}
359      */
360 
361     public void write(CLOutputStream out) throws IOException {
362         super.write(out);
363 
364         // out is cast to DataOutputStream to resolve the
365         // writeInt()
366         // ambiguity
367         ((DataOutputStream) out).writeInt(i);
368     }
369 
370     /**
371      * {@inheritDoc}
372      */
373 
374     public boolean equals(Object obj) {
375         if (obj instanceof CLConstantIntegerInfo) {
376             CLConstantIntegerInfo c = (CLConstantIntegerInfo) obj;
377             if (c.i == i) {
378                 return true;
379             }
380         }

382     }
383 
384     /**
