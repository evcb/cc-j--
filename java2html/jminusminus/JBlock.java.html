3   package jminusminus;
4   
5   import java.util.ArrayList;
6   
7   /**
8    * The AST node for a block, which delimits a nested level of scope.
9    *
10   * @see LocalContext
11   */
12  
13  class JBlock extends JStatement {
14  
15      /** List of statements forming the block body. */
16      private ArrayList<JStatement> statements;
17  
18      /**
19       * The new context (built in analyze()) represented by this block.
20       */
21      private LocalContext context;
22  
23      /**
24       * Constructs an AST node for a block given its line number, and the list of
25       * statements forming the block body.
26       * 
27       * @param line
28       *            line in which the block occurs in the source file.
29       * @param statements
30       *            list of statements forming the block body.
31       */
32  
33      public JBlock(int line, ArrayList<JStatement> statements) {
34          super(line);
35          this.statements = statements;
36      }
37  
38      /**
39       * Returns the list of statements comprising the block.
40       * 
41       * @return list of statements.
42       */
43  
44      public ArrayList<JStatement> statements() {
45          return statements;
46      }
47  
48      /**
49       * Analyzing a block consists of creating a new nested context for that
50       * block and analyzing each of its statements within that context.
51       * 
52       * @param context
53       *            context in which names are resolved.
54       * @return the analyzed (and possibly rewritten) AST subtree.
55       */
56  
57      public JBlock analyze(Context context) {
58          // { ... } defines a new level of scope.
59          this.context = new LocalContext(context);
60  
61          for (int i = 0; i < statements.size(); i++) {
62              statements.set(i, (JStatement) statements.get(i).analyze(
63                      this.context));
64          }
65          return this;
66      }
67  
68      /**
69       * Generating code for a block consists of generating code for each of its
70       * statements.
71       * 
72       * @param output
73       *            the code emitter (basically an abstraction for producing the
74       *            .class file).
75       */
76  
77      public void codegen(CLEmitter output) {
78          for (JStatement statement : statements) {
79              statement.codegen(output);
80          }
81      }
82  
83      /**
84       * {@inheritDoc}
85       */
86  
87      public void writeToStdOut(PrettyPrinter p) {
88          p.printf("<JBlock line=\"%d\">\n", line());
89          if (context != null) {
90              p.indentRight();
91              context.writeToStdOut(p);
92              p.indentLeft();
93          }
94          for (JStatement statement : statements) {
95              p.indentRight();
96              statement.writeToStdOut(p);
97              p.indentLeft();
98          }
99          p.printf("</JBlock>\n");
100     }
101 
102 }
