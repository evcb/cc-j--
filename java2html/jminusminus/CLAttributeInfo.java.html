10  /**
11   * The abstract base class for the representation of attribute_info structure 
12   * (JVM Spec Section 4.8). Classes representing individual attributes inherit
13   * this class. This file has representations for all attributes specified in
14   * JVM Spec Second Edition, including the ones that were added for JDK 1.5.
15   * <p>
16   * Attributes are used in the ClassFile ({@link CLFile}), field_info (
17   * {@link CLFieldInfo}), method_info ({@link CLMethodInfo}), and Code_attribute 
18   * ({@link CLCodeAttribute}) structures of the class file format. While there 
19   * are many kinds of attributes, only some are mandatory; these include:
20   * 
21   * <ul>
22   *   <li>InnerClasses_attribute (class attribute)</li>
23   *   <li>Synthetic_attribute (class, field, and method attribute)</li>
24   *   <li>Code_attribute (method attribute)</li>
25   *   <li>Exceptions_attribute (method attribute)</li>
26   * </ul>
27   * <br>
28   * {@link CLAbsorber} is capable of reading all attributes listed in this file. 
29   * The ones which it does not recognize, it simply skips them and reports a 
30   * warning to that extent.
31   * <p>
32   * {@link CLEmitter} implicitly adds the required attributes to the appropriate
33   * structure. The optional attributes have to be added explicitly using the
34   * {@link CLEmitter#addClassAttribute(CLAttributeInfo) addClassAttribute}, 
35   * {@link CLEmitter#addFieldAttribute(CLAttributeInfo) addFieldAttribute}, 
36   * {@link CLEmitter#addMethodAttribute(CLAttributeInfo) addMethodAttribute}, and
37   * {@link CLEmitter#addCodeAttribute(CLAttributeInfo) addCodeAttribute} methods in 
38   * {@code CLEmitter}.
39   */
40  
41  abstract class CLAttributeInfo {
42  
43      // The fields below represent the members of the
44      // attribute_info
45      // structure and are thus inherited by the child classes of
46      // CLAttributeInfo. These classes define their own fields
47      // (if any) representing the members of the individual
48      // attribute_info structures they represent.
49  
50      /** attribute_info.attribute_name_index item. */
51      public int attributeNameIndex;
52  
53      /** attribute_info.attribute_length item. */
54      public long attributeLength;
55  
56      /**
57       * Construct an CLAttributeInfo object.
58       * 
59       * @param attributeNameIndex
60       *            attribute_info.attribute_name_index item.
61       * @param attributeLength
62       *            attribute_info.attribute_length item.
63       */
64  
65      protected CLAttributeInfo(int attributeNameIndex, long attributeLength) {
66          this.attributeNameIndex = attributeNameIndex;
67          this.attributeLength = attributeLength;
68      }
69  
70      /**
71       * Write the contents of this attribute to the specified output stream.
72       * 
73       * @param out
74       *            output stream.
75       * @throws IOException
76       *             if an error occurs while writing.
77       */
78  
79      public void write(CLOutputStream out) throws IOException {
80          out.writeShort(attributeNameIndex);
81          out.writeInt((long) attributeLength);
82      }
83  
84      /**
85       * Write the contents of this attribute to STDOUT in a format similar to
86       * that of javap.
87       * 
88       * @param p
89       *            for pretty printing with indentation.
90       */
91  
92      public void writeToStdOut(PrettyPrinter p) {
93          p.printf("Attribute Name Index: %s\n", attributeNameIndex);
94          p.printf("Attribute Length: %s\n", attributeLength);
95      }
96  
97  }

224 class CLCodeAttribute extends CLAttributeInfo {
225 
226     /** Code_attribute.max_stack item. */
227     public int maxStack;
228 
229     /** Code_attribute.max_locals item. */
230     public int maxLocals;
231 
232     /** Code_attribute.code_length item. */
233     public long codeLength;
234 
235     /**
236      * Code_attribute.code item.
237      */
238     public ArrayList<Integer> code;
239 
240     /** Code_attribute.exception_table_length item. */
241     public int exceptionTableLength;
242 
243     /** Code_attribute.exception_table item. */
244     public ArrayList<CLExceptionInfo> exceptionTable;
245 
246     /** Code_attribute.attributes_count item. */
247     public int attributesCount;
248 
249     /** Code_attribute.attributes item. */
250     public ArrayList<CLAttributeInfo> attributes;
251 
252     /**
253      * Construct and return an integer from the four unsigned bytes specified.
254      * 
255      * @param a
256      *            unsigned byte.
257      * @param b
258      *            unsigned byte.
259      * @param c
260      *            unsigned byte.
261      * @param d
262      *            unsigned byte.
263      * @return an integer constructed from the four unsigned bytes specified.
264      */
265 
266     private int intValue(int a, int b, int c, int d) {
267         return (a << 24) | (b << 16) | (c << 8) | d;
268     }
269 
270     /**
271      * Construct a CLCodeAttribute object.
272      * 
273      * @param attributeNameIndex
274      *            Code_attribute.attribute_name_index item.
275      * @param attributeLength
276      *            Code_attribute.attribute_length item.
277      * @param maxStack
278      *            Code_attribute.max_stack item.
279      * @param maxLocals
280      *            Code_attribute.max_locals item.
281      * @param codeLength
282      *            Code_attribute.code_length item.
283      * @param code
284      *            Code_attribute.code item.
285      * @param exceptionTableLength
286      *            Code_attribute.exception_table_length item.
287      * @param exceptionTable
288      *            Code_attribute.exception_table item.
289      * @param attributesCount
290      *            Code_attribute.attributes_count item.
291      * @param attributes
292      *            Code_attribute.attributes item.
293      */
294 
295     public CLCodeAttribute(int attributeNameIndex, long attributeLength,
296             int maxStack, int maxLocals, long codeLength,
297             ArrayList<Integer> code, int exceptionTableLength,
298             ArrayList<CLExceptionInfo> exceptionTable, int attributesCount,
299             ArrayList<CLAttributeInfo> attributes) {
300         super(attributeNameIndex, attributeLength);
301         this.maxStack = maxStack;
302         this.maxLocals = maxLocals;
303         this.codeLength = codeLength;
304         this.code = code;
305         this.exceptionTableLength = exceptionTableLength;
306         this.exceptionTable = exceptionTable;
307         this.attributesCount = attributesCount;
308         this.attributes = attributes;
309     }
310 
311     /**
312      * {@inheritDoc}
313      */
314 
315     public void write(CLOutputStream out) throws IOException {
316         super.write(out);
317         out.writeShort(maxStack);
318         out.writeShort(maxLocals);
319         out.writeInt(codeLength);
320         for (int i = 0; i < code.size(); i++) {
321             out.writeByte(code.get(i));
322         }
323         out.writeShort(exceptionTableLength);
324         for (int i = 0; i < exceptionTable.size(); i++) {
325             exceptionTable.get(i).write(out);
326         }
327         out.writeShort(attributesCount);
328         for (int i = 0; i < attributes.size(); i++) {
329             attributes.get(i).write(out);
330         }
331     }
332 
333     /**
334      * {@inheritDoc}
335      */
336 
337     public void writeToStdOut(PrettyPrinter p) {
338         p.printf("Code {\n");
339         p.indentRight();
340         super.writeToStdOut(p);
341         p.printf("Max Stack: %s\n", maxStack);
342         p.printf("Max Locals: %s\n", maxLocals);
343         p.printf("Code Length: %s\n", codeLength);
344         p.printf("%-10s%-17s%s\n", "PC", "Opcode", "Operands");
345         p.printf("%-10s%-17s%s\n", "--", "------", "--------");
346         for (int i = 0; i < code.size(); i++) {
347             int pc = i;
348             int opcode = code.get(i);
349             String mnemonic = CLInstruction.instructionInfo[opcode].mnemonic;
350             int operandBytes = CLInstruction.instructionInfo[opcode].operandCount;
351             short operandByte1, operandByte2, operandByte3, operandByte4;
352             int pad, deflt;
353             switch (operandBytes) {
354             case 0:
355                 p.printf("%-10s%-17s\n", pc, mnemonic);
356                 break;
357             case 1:
358                 operandByte1 = code.get(++i).shortValue();
359                 p.printf("%-10s%-17s%-5s\n", pc, mnemonic, operandByte1);
360                 break;
361             case 2:
362                 operandByte1 = code.get(++i).shortValue();
363                 operandByte2 = code.get(++i).shortValue();
364                 p.printf("%-10s%-17s%-5s%-5s\n", pc, mnemonic, operandByte1,
365                         operandByte2);
366                 break;
367             case 3:
368                 operandByte1 = code.get(++i).shortValue();
369                 operandByte2 = code.get(++i).shortValue();
370                 operandByte3 = code.get(++i).shortValue();
371                 p.printf("%-10s%-17s%-5s%-5s%-5s\n", pc, mnemonic,
372                         operandByte1, operandByte2, operandByte3);
373                 break;
374             case 4:
375                 operandByte1 = code.get(++i).shortValue();
376                 operandByte2 = code.get(++i).shortValue();
377                 operandByte3 = code.get(++i).shortValue();
378                 operandByte4 = code.get(++i).shortValue();
379                 p.printf("%-10s%-17s%-5s%-5s%-5s%-5s\n", pc, mnemonic,
380                         operandByte1, operandByte2, operandByte3, operandByte4);
381                 break;
382             case DYNAMIC: // Variable length instructions
383                 if (opcode == TABLESWITCH) {
384                     int low, high;
385                     pad = 4 - ((i + 1) % 4);
386                     i = i + pad + 1;
387                     deflt = intValue(code.get(i++), code.get(i++), code
388                             .get(i++), code.get(i++));
389                     low = intValue(code.get(i++), code.get(i++), code.get(i++),
390                             code.get(i++));
391                     high = intValue(code.get(i++), code.get(i++),
392                             code.get(i++), code.get(i));
393                     p.printf("%-10s%s { // %s to %s \n", pc, mnemonic, low,
394                             high);
395                     for (int idx = low; idx <= high; idx++) {
396                         int offset = intValue(code.get(++i), code.get(++i),
397                                 code.get(++i), code.get(++i));
398                         p.printf("%-10s    %s:%s\n", "", idx, offset);
399                     }
400                     p.printf("%-10s    default: %s\n", "", deflt);
401                     p.printf("%-10s}\n", "");
402                 } else { // LOOKUPSWITCH
403                     int nPairs;
404                     pad = 4 - ((i + 1) % 4);
405                     i = i + pad + 1;
406                     deflt = intValue(code.get(i++), code.get(i++), code
407                             .get(i++), code.get(i++));
408                     nPairs = intValue(code.get(i++), code.get(i++), code
409                             .get(i++), code.get(i));
410                     p.printf("%-10s%s { \n", pc, mnemonic);
411                     for (int idx = 0; idx < nPairs; idx++) {
412                         int match = intValue(code.get(++i), code.get(++i), code
413                                 .get(++i), code.get(++i));
414                         int offset = intValue(code.get(++i), code.get(++i),
415                                 code.get(++i), code.get(++i));
416                         p.printf("%-10s    %s:%s\n", "", match, offset);
417                     }
418                     p.printf("%-10s    default: %s\n", "", deflt);
419                     p.printf("%-10s}\n", "");
420                 }
421             }
422         }
423         p.println();
424         p.printf("// Exception Table (%s Items)\n", exceptionTableLength);
425         p.printf("%s    %s    %s    %s\n", "Start PC", "End PC", "Handler PC",
426                 "Catch Type");
427         p.printf("%s    %s    %s    %s\n", "--------", "------", "----------",
428                 "----------");
429         for (int i = 0; i < exceptionTable.size(); i++) {
430             exceptionTable.get(i).writeToStdOut(p);
431         }
432         p.println();
433         p.printf("// Attributes (%s Items)\n", attributesCount);
434         for (int i = 0; i < attributes.size(); i++) {
435             CLAttributeInfo attributeInfo = attributes.get(i);
436             attributeInfo.writeToStdOut(p);
437         }
438         p.indentLeft();
439         p.printf("}\n");
440     }
441 
442 }
