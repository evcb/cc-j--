11  /**
12   * A lifetime interval, recording the interval of LIR code for which the
13   * corresponding virtual register contains a useful value.
14   */
15  
16  class NInterval implements Comparable<NInterval> {
17  
18      /** Control flow graph instance. */
19      private NControlFlowGraph cfg;
20  
21      /**
22       * The virtual register id corresponding to the index in the array list of
23       * NIntervals used by register allocation
24       */
25      public int vRegId;
26  
27      /** All live ranges for this virtual register */
28      public ArrayList<NRange> ranges;
29  
30      /**
31       * All use positions (in LIR) and their types for this virtual register
32       */
33      public TreeMap<Integer, InstructionType> usePositions;
34  
35      /**
36       * The NPhyicalRegister assigned to this interval. If an interval ends up
37       * needing more than one physical register it is split.
38       */
39      public NPhysicalRegister pRegister;
40  
41      /** Whether or not to spill. */
42      public boolean spill;
43  
44      /** From offset. */
45      public OffsetFrom offsetFrom;
46  
47      /** Offset. */
48      public int offset;
49  
50      /** Parent of this interval. */
51      public NInterval parent;
52  
53      /** Children of this interval. */
54      public ArrayList<NInterval> children;

56      /**
57       * Constructs a NInterval with the given virtual register ID for the given
58       * control flow graph.
59       * 
60       * @param virtualRegID
61       *            program counter.
62       * @param cfg
63       *            The control flow graph.
64       */
65  
66      public NInterval(int virtualRegID, NControlFlowGraph cfg) {
67          this.cfg = cfg;
68          this.ranges = new ArrayList<NRange>();
69          this.usePositions = new TreeMap<Integer, InstructionType>();
70          this.vRegId = virtualRegID;
71          this.children = new ArrayList<NInterval>();
72          this.parent = null;
73          spill = false;
74          offset = -1;
75      }
76  
77      /**
78       * This second constructor is used in instantiating children of a split
79       * interval.
80       * 
81       * @param virtualRegID
82       *            program counter.
83       * @param cfg
84       *            The control flow graph.
85       * @param childRanges
86       *            The instruction ranges for this child.
87       * @param parent
88       *            The parent interval.
89       */
90  
91      public NInterval(int virtualRegID, NControlFlowGraph cfg,
92              ArrayList<NRange> childRanges, NInterval parent) {
93          this.cfg = cfg;
94          this.ranges = childRanges;
95          this.usePositions = new TreeMap<Integer, InstructionType>();
96          this.vRegId = virtualRegID;
97          this.parent = parent;
98          this.children = new ArrayList<NInterval>();
99          spill = false;
100         offset = -1;
101     }

103     /**
104      * Adds a new range to the existing ranges. If the range overlaps then
105      * the old start position will be given the new start range position.
106      * 
107      * @param newNRange
108      *            the NRange to add
109      */
110 
111     public void addOrExtendNRange(NRange newNRange) {
112         if (!ranges.isEmpty()) {
113             if (newNRange.stop + 5 == ranges.get(0).start
114                     || newNRange.rangeOverlaps(ranges.get(0))) {
115                 ranges.get(0).start = newNRange.start;
116             } else {
117                 ranges.add(0, newNRange);
118             }
119         } else {
120             ranges.add(newNRange);
121         }
122     }
123 
124     /**
125      * Looks for the very first position where an intersection with another
126      * interval occurs.
127      * 
128      * NOTE: A.nextIntersection(B) equals B.nextIntersection(A)
129      * 
130      * @param otherInterval
131      *            the interval to compare against for intersection.
132      * @return the position where the intersection begins.
133      */
134 
135     public int nextIntersection(NInterval otherInterval) {
136         int a = -1, b = -2;
137         for (NRange r : this.ranges) {
138             if (otherInterval.isLiveAt(r.start)) {
139                 a = r.start;
140                 break;
141             }
142         }
143         for (NRange r : otherInterval.ranges) {
144             if (this.isLiveAt(r.start)) {
145                 b = r.start;
146                 break;
147             }
148         }
149         if (a >= 0 && b >= 0) {
150             return a <= b ? a : b;
151         } else {
152             return a > b ? a : b;
153         }
154     }
155 
156     /**
157      * Returns the next use position of this interval after the first range 
158      * start of the foreign interval. If there is no such use, then the first 
159      * use position is returned to preserve data flow (in case of loops).
160      * 
161      * @param currInterval
162      *            the interval with starting point after which we want to find
163      *            the next usage of this one.
164      * 
165      * @return the next use position.
166      * 
167      */
168 
169     public int nextUsageOverlapping(NInterval currInterval) {
170         int psi = currInterval.firstRangeStart();
171 
172         if (usePositions.ceilingKey(psi) != null) {
173             return usePositions.ceilingKey(psi);
174         } else if (!usePositions.isEmpty()) {
175             return usePositions.firstKey();
176         } else {
177             return Integer.MAX_VALUE;
178         }
179     }
180 
181     /**
182      * Returns the first use position in this interval.
183      * 
184      * @return the first use position.
185      */
186 
187     public int firstUsage() {
188         return usePositions.firstKey();
189     }
190 
191     /**
192      * Sets the start value of the very first range. Note: There will always be
193      * at least one range before this method is used by the
194      * {@link NRegisterAllocator#buildIntervals() buildIntervals} method.
195      * 
196      * @param newStart
197      *            the value to which the first range's start will be set.
198      */
199     public void newFirstRangeStart(int newStart) {
200         // Check
201         if (!ranges.isEmpty()) {
202             ranges.get(0).start = newStart;
203         }
204     }
205 
206     /**
207      * Registers a use (read or write).
208      * 
209      * @param index
210      *            the site of the use.
211      * @param type
212      *            the instruction type.
213      */
214 
215     public void addUsePosition(Integer index, InstructionType type) {
216         usePositions.put(index, type);
217     }
218 
219     /**
220      * Checks if this virtual register is alive at a given index.
221      * 
222      * @param atIndex
223      *            the index at which to see if this register is live.
224      * @return {@code true} if this virtual register is alive at the given 
225      *         index; {@code false} otherwise.
226      */
227 
228     public boolean isLiveAt(int atIndex) {
229         for (NRange r : ranges) {
230             if (atIndex >= r.start && atIndex <= r.stop) {
231                 return true;
232             }
233         }
234         return false;
235     }
236 
237     /**
238      * Returns the range in this interval in which the LIR instruction with the 
239      * given id is live; {@code null} otherwise. This will never return 
240      * {@code null} if called for an interval from the active list after it has 
241      * been set up by the allocation method.
242      * 
243      * @param id
244      *            LIR instruction id.
245      * @return range in which LIR instruction with given id is live, or null.
246      */
247 
248     private NRange liveRangeAt(int id) {
249         for (NRange r : ranges) {
250             if (id >= r.start && id <= r.stop) {
251                 return r;
252             }
253         }
254         return null;
255     }
256 
257     /**
258      * Writes the interval information to STDOUT.
259      * 
260      * @param p
261      *            for pretty printing with indentation.
262      */
263 
264     public void writeToStdOut(PrettyPrinter p) {
265         if (cfg.registers.get(vRegId) != null) {
266             String s = cfg.registers.get(vRegId).name() + ": ";
267             for (NRange r : ranges) {
268                 s += r.toString() + " ";
269             }
270             if (pRegister != null) {
271                 s += "-> " + pRegister.name();
272             } else {
273                 s += "-> None";
274             }
275             if (spill) {
276                 if (offsetFrom == OffsetFrom.FP) {
277                     s += " [frame:" + offset + "]";
278                 } else {
279                     s += " [stack:" + offset + "]";
280                 }
281             }
282             p.printf("%s\n", s);
283             for (NInterval child : this.children) {
284                 child.writeToStdOut(p);
285             }
286         } else if (this.isChild()) {
287             String s = "\tv" + this.vRegId + ": ";
288             for (NRange r : ranges) {
289                 s += r.toString() + " ";
290             }
291             if (pRegister != null) {
292                 s += "-> " + pRegister.name();
293             } else {
294                 s += "-> None";
295             }
296             if (offsetFrom == OffsetFrom.FP) {
297                 s += " [frame:" + offset + "]";
298             } else {
299                 s += " [stack:" + offset + "]";
300             }
301             p.printf("%s\n", s);
302             for (NInterval child : this.children) {
303                 child.writeToStdOut(p);
304             }
305         }
306     }

310      * 
311      * @return the start position.

315         if (ranges.isEmpty())
316             return -1;

321     /**
322      * Returns the stop position for the last range.

328         if (ranges.isEmpty())
329             return -1;
330         else
331             return ranges.get(ranges.size() - 1).stop;

344         return this.firstRangeStart() - other.firstRangeStart();

347     /**
348      * Two intervals are equal if they have the same virtual register ID.
349      * 
350      * @param other
351      *            the interval we are comparing ourself with.
352      * @return true if the two intervals are the same, false otherwise.

356         return (this.vRegId == other.vRegId);

359     /**
360      * Splits the current interval at the given index. Responsible for splitting
361      * a range if the index falls on one, moving remaining ranges over to child,
362      * and moving appropriate usePositions over to the child.
363      * 
364      * @param idx
365      *            the index at which this interval is to be split
366      * 
367      * @return the child interval which is to be sorted onto unhandled. If there
368      *         was no child created in the case of a pruning this interval is
369      *         returned.
370      */
371 
372     public NInterval splitAt(int idx) {
373 
374         ArrayList<NRange> childsRanges = new ArrayList<NRange>();
375         if (this.isLiveAt(idx)) { // means split falls on a range
376             // Assumptions: if a range is LIVE on an index, then there
377             // exist usePositions at or before the index
378             // within this same range.
379             NRange liveRange = this.liveRangeAt(idx);
380             int splitTo = idx;
381             splitTo = usePositions.ceilingKey(idx);
382             childsRanges.add((liveRange.splitRange(splitTo, idx - 5)));
383         }
384 
385         // The following two for loops take care of any untouched ranges
386         // which start after the split position and must be moved to the
387         // child interval.
388         for (NRange r : ranges) {
389             if (r.start > idx) {
390                 childsRanges.add(r);
391             }
392         }
393         for (NRange r : childsRanges) {
394             ranges.remove(r);
395         }
396 
397         NInterval child = new NInterval(cfg.maxIntervals++, cfg, childsRanges,
398                 this.getParent());
399         cfg.registers.add(null); // expand size of cfg.registers to
400         // avoid null pointer exception when printing.
401 
402         // transfer remaining use positions
403         while (this.usePositions.ceilingKey(idx) != null)
404             child.usePositions
405                     .put(this.usePositions.ceilingKey(idx), this.usePositions
406                             .remove(this.usePositions.ceilingKey(idx)));
407         this.getParent().children.add(child);
408         return child;
409     }

418         if (parent != null)

424     /**
425      * Returns the child interval at a given instruction index.
426      * 
427      * @param idx
428      *            The instruction index.
429      * @return the child interval.
430      */
431 
432     public NInterval childAt(int idx) {
433         for (NInterval child : children) {
434             if (child.isLiveAt(idx)) {
435                 return child;
436             }
437         }
438         return this;
439     }
440 
441     /**
442      * Returns the child of this interval which is live or ends before the given 
443      * basic block's end.
444      * 
445      * @param b
446      *            the basic block.
447      * @return the child of this interval which ends at or nearest (before) this
448      *         basic block's end (last lir instruction index).
449      */
450 
451     public NInterval childAtOrEndingBefore(NBasicBlock b) {
452         int idx = b.getLastLIRInstId();
453         for (NInterval child : children) {
454             if (child.isLiveAt(idx))
455                 return child;
456         }
457         NInterval tmp = this;
458         int highestEndingAllowed = b.getFirstLIRInstId();
459         for (NInterval child : children) {
460             // get the child which ends before idx but also ends closest to idx
461             if (child.lastNRangeStop() < idx
462                     && child.lastNRangeStop() > highestEndingAllowed) {
463                 tmp = child;
464                 highestEndingAllowed = tmp.lastNRangeStop();
465             }
466         }
467         return tmp;
468     }
469 
470     /**
471      * Returns the child of this interval which is live or starts after the given 
472      * basic block's start.
473      * 
474      * @param b
475      *            the basic block.
476      * @return the child of this interval which starts at or nearest (after)
477      *         this basic block's start (fist lir instruction index).
478      */
479 
480     public NInterval childAtOrStartingAfter(NBasicBlock b) {
481         int idx = b.getFirstLIRInstId();
482         for (NInterval child : children) {
483             if (child.isLiveAt(idx))
484                 return child;
485         }
486         NInterval tmp = this;
487         int lowestStartAllowed = b.getLastLIRInstId();// block's end
488         for (NInterval child : children) {
489             if (child.firstRangeStart() > idx
490                     && child.firstRangeStart() < lowestStartAllowed) {
491                 tmp = child;
492                 lowestStartAllowed = tmp.firstRangeStart();
493             }
494         }
495         return tmp;
496     }

498     /**
499      * Returns the basic block in which this interval's start position falls.
500      * 
501      * @return basic block in which this interval's start position falls.
502      */
503 
504     public int startsAtBlock() {
505         for (NBasicBlock b : this.cfg.basicBlocks) {
506             if (this.firstRangeStart() >= b.getFirstLIRInstId()
507                     && this.firstRangeStart() <= b.getLastLIRInstId())
508                 return b.id;
509         }
510         return -1; // this will never happen
511     }
512 
513     /**
514      * Returns the basic block in which this interval's end position falls.
515      * 
516      * @return the basic block number.
517      */
518 
519     public int endsAtBlock() {
520         for (NBasicBlock b : this.cfg.basicBlocks) {
521             if (this.lastNRangeStop() >= b.getFirstLIRInstId()
522                     && this.lastNRangeStop() <= b.getLastLIRInstId()) {
523                 return b.id;
524             }
525         }
526         return -1; // this will never happen
527     }
528 
529     /**
530      * Assigns an offset to this interval (if one hasn't been already assigned).
531      * Assigns that same offset to any (newly created) children.
532      */
533 
534     public void spill() {
535         this.spill = true;
536         if (this.offset == -1) {
537             this.offset = cfg.offset++;
538             this.offsetFrom = OffsetFrom.SP;
539         }
540         for (NInterval child : children) {
541             if (child.offset == -1) {
542                 child.offset = this.offset;
543                 child.offsetFrom = this.offsetFrom;
544             }
545         }
546     }

549     // for spills.

551     /**
552      * Is this interval a child interval?
553      * 
554      * @return {@code true} if this interval is a child interval; 
555      *         {@code false} otherwise.
556      */
557 
558     public boolean isChild() {
559         if (this.parent != null) {
560             return true;
561         } else {
562             return false;
563         }
564     }
565 
566     /**
567      * Is this interval a parent interval? (That is, does it have children?)
568      * 
569      * @return {@code true} if this interval is a parent interval; 
570      *         {@code false} otherwise.
571      */
572 
573     public boolean isParent() {
574         return !this.children.isEmpty();
575     }

589 /**
590  * A liveness range (for an interval).
591  */
592 
593 class NRange implements Comparable<NRange> {
594 
595     /** The range's start position. */
596     public int start;
597 
598     /** The range's stop position. */
599     public int stop;
600 
601     /**
602      * Constructs a liveness range extending from start to stop (positions in the
603      * code).
604      * 
605      * @param start
606      *            start position.
607      * @param stop
608      *            stop position.
609      */
610     public NRange(int start, int stop) {
611         this.start = start;
612         this.stop = stop;
613 
614     }
615 
616     /**
617      * Mutates current range to be only as long as the split point and returns
618      * the remainder as a new range.
619      * 
620      * @param newStart
621      *            the split location
622      * @param oldStop
623      *            the split location
624      * @return the new NRange which begins at the split position and runs to the
625      *         end of this previously whole range.
626      */
627 
628     public NRange splitRange(int newStart, int oldStop) {
629         NRange newRange = new NRange(newStart, stop);
630         this.stop = oldStop;
631 
632         return newRange;
633     }
634 
635     /**
636      * Does this liveness range overlap with another?
637      * 
638      * @param a
639      *            The other range.
640      * @return {@code true} if this liveness range overlaps;
641      *         {@code false} otherwise.
642      */
643 
644     public boolean rangeOverlaps(NRange a) {
645         return (a.start <= this.stop && this.start <= a.stop);
646     }
647 
648     /**
649      * One liveness range comes before another if its start position comes
650      * before the other's start position.
651      * 
652      * @param other
653      *            the other range.
654      * 
655      * @return comparison.
656      */
657 
658     public int compareTo(NRange other) {
659         return this.start - other.start;
660     }
661 
662     /**
663      * Returns the string representation of the range.
664      * 
665      * @return "[start,stop]"
666      */
667 
668     public String toString() {
669         return "[" + start + ", " + stop + "]";
670     }
