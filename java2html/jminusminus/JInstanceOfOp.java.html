3   package jminusminus;
4   
5   import static jminusminus.CLConstants.*;
6   
7   /**
8    * The AST node for an {@code instanceof} expression, having two
9    * arguments: an expression and a reference type.
10   */
11  
12  class JInstanceOfOp extends JExpression {
13  
14      /** The expression denoting the value to be tested. */
15      private JExpression expr;
16  
17      /** The reference type we are testing for. */
18      private Type typeSpec;
19  
20      /**
21       * Constructs an AST node for an {@code instanceof} expression given
22       * its line number, the relational expression, and reference
23       * type.
24       * 
25       * @param line
26       *            the line in which the instanceof expression
27       *            occurs in the source file.
28       * @param expr
29       *            the expression denoting the value to be
30       *            tested.
31       * @param typeSpec
32       *            the reference type we are testing for.
33       */
34  
35      public JInstanceOfOp(int line, JExpression expr, Type typeSpec) {
36          super(line);
37          this.expr = expr;
38          this.typeSpec = typeSpec;
39      }
40  
41      /**
42       * Analysis of an {@code instanceof} operation requires analyzing the
43       * expression to be tested, resolving the type we are testing for, and 
44       * determining if the test is legal, or if the answer can be determined 
45       * at compile time.
46       * 
47       * @param context
48       *            context in which names are resolved.
49       * @return the analyzed (and possibly rewritten) AST subtree.
50       */
51  
52      public JInstanceOfOp analyze(Context context) {
53          expr = (JExpression) expr.analyze(context);
54          typeSpec = typeSpec.resolve(context);
55          if (!typeSpec.isReference()) {
56              JAST.compilationUnit.reportSemanticError(line(),
57                  "Type argument to instanceof "
58                      + "operator must be a reference type");
59          } else if (!(expr.type() == Type.NULLTYPE
60              || expr.type() == Type.ANY || expr.type().isReference())) {
61              JAST.compilationUnit.reportSemanticError(line(),
62                  "operand to instanceof "
63                      + "operator must be a reference type");
64          } else if (expr.type().isReference()
65              && !typeSpec.isJavaAssignableFrom(expr.type())) {
66              JAST.compilationUnit.reportSemanticError(line(),
67                  "It is impossible for the expression "
68                      + "to be an instance of this type");
69          }
70          type = Type.BOOLEAN;
71          return this;
72      }
73  
74      /**
75       * Generates code for the type test.
76       * 
77       * @param output
78       *            the code emitter (basically an abstraction
79       *            for producing the .class file).
80       */
81  
82      public void codegen(CLEmitter output) {
83          expr.codegen(output);
84          output.addReferenceInstruction(INSTANCEOF, typeSpec
85              .toDescriptor());
86      }
87  
88      /**
89       * Short-circuiting branching for {@code instanceof}.
90       *
91       * @param output
92       *            code emitter.
93       * @param targetLabel
94       *            the label to which we should branch.
95       * @param onTrue
96       *            do we branch on true?
97       */
98  
99      public void codegen(CLEmitter output, String targetLabel, boolean onTrue) {
100         codegen(output);
101         if (onTrue) {
102             // Branch on true
103             output.addBranchInstruction(IFNE, targetLabel);
104         } else {
105             // Branch on false
106             output.addBranchInstruction(IFEQ, targetLabel);
107         }
108     }
109 
110     /**
111      * {@inheritDoc}
112      */
113 
114     public void writeToStdOut(PrettyPrinter p) {
115         p.printf("<JInstanceOfOp line=\"%d\" type=\"%s\">\n", line(),
116             ((type == null) ? "" : type.toString()));
117         p.indentRight();
118         p.printf("<RelationalExpression>\n");
119         p.indentRight();
120         expr.writeToStdOut(p);
121         p.indentLeft();
122         p.printf("</RelationalExpression>\n");
123         p.printf("<ReferenceType value=\"%s\"/>\n",
124             ((typeSpec == null) ? "" : typeSpec.toString()));
125         p.indentLeft();
126         p.printf("</JInstanceOfOp>\n");
127     }
128 }
