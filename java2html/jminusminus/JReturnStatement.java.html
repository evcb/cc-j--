3   package jminusminus;
4   
5   import static jminusminus.CLConstants.*;
6   
7   /**
8    * The AST node for a return-statement. If the enclosing method
9    * is non-void, then there is a value to return, so we keep track
10   * of the expression denoting that value and its type.
11   */
12  
13  class JReturnStatement extends JStatement {
14  
15      /** The returned expression. */
16      private JExpression expr;
17  
18      /**
19       * Constructs an AST node for a return-statement given its
20       * line number, and the expression that is returned.
21       * 
22       * @param line
23       *            line in which the return-statement appears
24       *            in the source file.
25       * @param expr
26       *            the returned expression.
27       */
28  
29      public JReturnStatement(int line, JExpression expr) {
30          super(line);
31          this.expr = expr;
32      }
33  
34      /**
35       * Analysis distinguishes between being in a constructor
36       * or in a regular method in checking return types. In the
37       * case of a return expression, analyze it and check types.
38       * Determine the (possibly void) return type.
39       * 
40       * @param context
41       *            context in which names are resolved.
42       * @return the analyzed (and possibly rewritten) AST subtree.
43       */
44  
45      public JStatement analyze(Context context) {
46          MethodContext methodContext = context.methodContext();
47  
48          // The methodContext can be null if return statement
49          // occurs in a block that is not within a method.
50          // For example, in the Java grammar, return statement, at least
51          // syntactically, can occur in a static block. But since
52          // j-- does not allow a block to occur outside of a
53          // method, we don't check for methodContext being null
54  
55          if (methodContext.methodReturnType() == Type.CONSTRUCTOR) {
56              if (expr != null) {
57                  // Can't return a value from a constructor
58                  JAST.compilationUnit.reportSemanticError(line(),
59                      "cannot return a value from a constructor");
60              }
61          } else {
62              // Must be a method
63              Type returnType = methodContext.methodReturnType();
64              methodContext.confirmMethodHasReturn();
65              if (expr != null) {
66                  if (returnType == Type.VOID) {
67                      // Can't return a value from void method
68                      JAST.compilationUnit.reportSemanticError(line(),
69                          "cannot return a value from a void method");
70                  } else {
71                      // There's a (non-void) return expression.
72                      // Its type must match the return type of the method
73                      expr = expr.analyze(context);
74                      expr.type().mustMatchExpected(line(), returnType);
75                  }
76              } else {
77                  // The method better have void as return type
78                  if (returnType != Type.VOID) {
79                      JAST.compilationUnit.reportSemanticError(line(),
80                          "missing return value");
81                  }
82              }
83          }
84          return this;
85      }
86  
87      /**
88       * Generates code for the return statement. In the case of
89       * void method types, generate a simple (void) return. In the
90       * case of a return expression, generate code to load that
91       * onto the stack and then generate the appropriate return
92       * instruction.
93       * 
94       * @param output
95       *            the code emitter (basically an abstraction
96       *            for producing the .class file).
97       */
98  
99      public void codegen(CLEmitter output) {
100         if (expr == null) {
101             output.addNoArgInstruction(RETURN);
102         } else {
103             expr.codegen(output);
104             if (expr.type() == Type.INT
105                 || expr.type() == Type.BOOLEAN
106                 || expr.type() == Type.CHAR) {
107                 output.addNoArgInstruction(IRETURN);
108             } else {
109                 output.addNoArgInstruction(ARETURN);
110             }
111         }
112     }
113 
114     /**
115      * {@inheritDoc}
116      */
117 
118     public void writeToStdOut(PrettyPrinter p) {
119         if (expr != null) {
120             p.printf("<JReturnStatement line=\"%d\">\n", line());
121             p.indentRight();
122             expr.writeToStdOut(p);
123             p.indentLeft();
124             p.printf("</JReturnStatement>\n");
125         } else {
126             p.printf("<JReturnStatement line=\"%d\"/>\n", line());
127         }
128     }
129 }
