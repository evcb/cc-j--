12  /**
13   * CLAbsorber is for reading a Java class into an in-memory {@link CLFile}
14   * representation and printing it out to STDOUT in a format similar to that of
15   * javap.
16   */
17  
18  public class CLAbsorber {
19  
20      /** CLFile representation of the class that is read. */
21      private CLFile classFile;
22  
23      /** Whether or not an error occurred in reading the class. */
24      private boolean errorHasOccurred;
25  
26      /** Name of the class that is read. */
27      private String className;
28  
29      /**
30       * Print the specified warning to STDERR.
31       * 
32       * @param message
33       *            warning.
34       * @param args
35       *            related values.
36       */
37  
38      private void reportWarning(String message, Object... args) {
39          System.err.printf("CLAbsorber Warning: " + message + "\n", args);
40      }
41  
42      /**
43       * Print the specified error message to STDERR and set the error flag to
44       * true.
45       * 
46       * @param message
47       *            error message.
48       * @param args
49       *            related values.
50       */
51  
52      private void reportError(String message, Object... args) {
53          System.err.printf("CLAbsorber Error: " + message + "\n", args);
54          errorHasOccurred = true;
55      }

57      /**
58       * Read the constant pool information from the specified stream, and return
59       * the information as a CLConstantPool object.
60       * 
61       * @param in
62       *            input stream.
63       * @return the constant pool.
64       */
65  
66      private CLConstantPool readConstantPool(CLInputStream in) {
67          CLConstantPool cp = new CLConstantPool();
68          try {
69              for (int i = 1; i < classFile.constantPoolCount; i++) {
70                  int tag = in.readUnsignedByte();
71                  switch (tag) {
72                  case CONSTANT_Class:
73                      cp.addCPItem(new CLConstantClassInfo(in
74                                      .readUnsignedShort()));
75                      break;
76                  case CONSTANT_Fieldref:
77                      cp.addCPItem(new CLConstantFieldRefInfo(in
78                              .readUnsignedShort(), in.readUnsignedShort()));
79                      break;
80                  case CONSTANT_Methodref:
81                      cp.addCPItem(new CLConstantMethodRefInfo(in
82                              .readUnsignedShort(), in.readUnsignedShort()));
83                      break;
84                  case CONSTANT_InterfaceMethodref:
85                      cp.addCPItem(new CLConstantInterfaceMethodRefInfo(in
86                              .readUnsignedShort(), in.readUnsignedShort()));
87                      break;
88                  case CONSTANT_String:
89                      cp.addCPItem(new CLConstantStringInfo(in
90                              .readUnsignedShort()));
91                      break;
92                  case CONSTANT_Integer:
93                      cp.addCPItem(new CLConstantIntegerInfo(in.readInt()));
94                      break;
95                  case CONSTANT_Float:
96                      cp.addCPItem(new CLConstantFloatInfo(in.readFloat()));
97                      break;
98                  case CONSTANT_Long:
99                      cp.addCPItem(new CLConstantLongInfo(in.readLong()));
100                     i++;
101                     break;
102                 case CONSTANT_Double:
103                     cp.addCPItem(new CLConstantDoubleInfo(in.readDouble()));
104                     i++;
105                     break;
106                 case CONSTANT_NameAndType:
107                     cp.addCPItem(new CLConstantNameAndTypeInfo(in
108                             .readUnsignedShort(), in.readUnsignedShort()));
109                     break;
110                 case CONSTANT_Utf8:
111                     int length = in.readUnsignedShort();
112                     byte[] b = new byte[length];
113                     in.read(b);
114                     cp.addCPItem(new CLConstantUtf8Info(b));
115                     break;
116                 default:
117                     reportError("Unknown cp_info tag '%d'", tag);
118                     return cp;
119                 }
120             }
121         } catch (IOException e) {
122             reportError("Error reading constant pool from file %s", className);
123         }
124         return cp;
125     }

127     /**
128      * Read the fields from the specified stream, and return them as a list.
129      * 
130      * @param in
131      *            input stream.
132      * @param fieldsCount
133      *            number of fields.
134      * @return list of fields.
135      */
136 
137     private ArrayList<CLFieldInfo> readFields(CLInputStream in, int fieldsCount) {
138         ArrayList<CLFieldInfo> fields = new ArrayList<CLFieldInfo>();
139         try {
140             for (int i = 0; i < fieldsCount; i++) {
141                 int accessFlags     = in.readUnsignedShort();
142                 int nameIndex       = in.readUnsignedShort();
143                 int descriptorIndex = in.readUnsignedShort();
144                 int attributesCount = in.readUnsignedShort();
145                 fields.add(new CLFieldInfo(accessFlags, 
146                                            nameIndex,
147                                            descriptorIndex, 
148                                            attributesCount, 
149                                            readAttributes(in, attributesCount))
150                           );
151             }
152         } catch (IOException e) {
153             reportError("Error reading fields from file %s", className);
154         }
155         return fields;
156     }

190     /**
191      * Read the attributes from the specified stream, and return them as a list
192      * 
193      * @param in
194      *            input stream.
195      * @param attributeCount
196      *            number of attributes.
197      * @return list of attributes.
198      */
199 
200     private ArrayList<CLAttributeInfo> readAttributes(CLInputStream in,
201             int attributesCount) {
202         ArrayList<CLAttributeInfo> attributes = new ArrayList<CLAttributeInfo>();
203         try {
204             CLConstantPool cp = classFile.constantPool;
205             for (int i = 0; i < attributesCount; i++) {
206                 int attributeNameIndex = in.readUnsignedShort();
207                 long attributeLength   = in.readUnsignedInt();
208                 CLAttributeInfo attributeInfo = null;
209 
210                 String attributeName = new String(((CLConstantUtf8Info) cp
211                         .cpItem(attributeNameIndex)).b);
212                 if (attributeName.equals(ATT_CONSTANT_VALUE)) {
213                     attributeInfo = readConstantValueAttribute(in,
214                             attributeNameIndex, attributeLength);
215                 } else if (attributeName.equals(ATT_CODE)) {
216                     attributeInfo = readCodeAttribute(in, attributeNameIndex,
217                             attributeLength);
218                 } else if (attributeName.equals(ATT_EXCEPTIONS)) {
219                     attributeInfo = readExceptionsAttribute(in,
220                             attributeNameIndex, attributeLength);
221                 } else if (attributeName.equals(ATT_INNER_CLASSES)) {
222                     attributeInfo = readInnerClassesAttribute(in,
223                             attributeNameIndex, attributeLength);
224                 } else if (attributeName.equals(ATT_ENCLOSING_METHOD)) {
225                     attributeInfo = readEnclosingMethodAttribute(in,
226                             attributeNameIndex, attributeLength);
227                 } else if (attributeName.equals(ATT_SYNTHETIC)) {
228                     attributeInfo = readSyntheticAttribute(in,
229                             attributeNameIndex, attributeLength);
230                 } else if (attributeName.equals(ATT_SIGNATURE)) {
231                     attributeInfo = readSignatureAttribute(in,
232                             attributeNameIndex, attributeLength);
233                 } else if (attributeName.equals(ATT_SOURCE_FILE)) {
234                     attributeInfo = readSourceFileAttribute(in,
235                             attributeNameIndex, attributeLength);
236                 } else if (attributeName.equals(ATT_SOURCE_DEBUG_EXTENSION)) {
237                     attributeInfo = readSourceDebugExtensionAttribute(in,
238                             attributeNameIndex, attributeLength);
239                 } else if (attributeName.equals(ATT_LINE_NUMBER_TABLE)) {
240                     attributeInfo = readLineNumberTableAttribute(in,
241                             attributeNameIndex, attributeLength);
242                 } else if (attributeName.equals(ATT_LOCAL_VARIABLE_TABLE)) {
243                     attributeInfo = readLocalVariableTableAttribute(in,
244                             attributeNameIndex, attributeLength);
245                 } else if (attributeName.equals(ATT_LOCAL_VARIABLE_TYPE_TABLE)) {
246                     attributeInfo = readLocalVariableTypeTableAttribute(in,
247                             attributeNameIndex, attributeLength);
248                 } else if (attributeName.equals(ATT_DEPRECATED)) {
249                     attributeInfo = readDeprecatedAttribute(in,
250                             attributeNameIndex, attributeLength);
251                 } else if (attributeName
252                         .equals(ATT_RUNTIME_VISIBLE_ANNOTATIONS)) {
253                     attributeInfo = readRuntimeVisibleAnnotationsAttribute(in,
254                             attributeNameIndex, attributeLength);
255                 } else if (attributeName
256                         .equals(ATT_RUNTIME_INVISIBLE_ANNOTATIONS)) {
257                     attributeInfo = readRuntimeInvisibleAnnotationsAttribute(
258                             in, attributeNameIndex, attributeLength);
259                 } else if (attributeName
260                         .equals(ATT_RUNTIME_VISIBLE_PARAMETER_ANNOTATIONS)) {
261                     attributeInfo = readRuntimeVisibleParameterAnnotationsAttribute(
262                             in, attributeNameIndex, attributeLength);
263                 } else if (attributeName
264                         .equals(ATT_RUNTIME_INVISIBLE_PARAMETER_ANNOTATIONS)) {
265                     attributeInfo = readRuntimeInvisibleParameterAnnotationsAttribute(
266                             in, attributeNameIndex, attributeLength);
267                 } else if (attributeName.equals(ATT_ANNOTATION_DEFAULT)) {
268                     attributeInfo = readAnnotationDefaultAttribute(in,
269                             attributeNameIndex, attributeLength);
270                 } else {
271                     reportWarning("Unknown attribute '%s'", attributeName,
272                             className);
273                     for (long j = 0; j < attributeLength; j++) {
274                         in.readUnsignedByte();
275                     }
276                 }
277                 if (attributeInfo != null) {
278                     attributes.add(attributeInfo);
279                 }
280             }
281         } catch (IOException e) {
282             reportError("Error reading attributes from file %s", className);
283         }
284         return attributes;
285     }

314     /**
315      * Read a Code attribute from the specified input stream, and return it.
316      * 
317      * @param in
318      *            input stream.
319      * @param attributeNameIndex
320      *            constant pool index of the attribute name.
321      * @param attributeLength
322      *            length of attribute.
323      * @return a Code attribute.
324      */
325 
326     private CLCodeAttribute readCodeAttribute(CLInputStream in,
327                                               int attributeNameIndex,
328                                               long attributeLength) {
329         CLCodeAttribute attribute = null;
330         try {
331             int maxStack  = in.readUnsignedShort();
332             int maxLocals = in.readUnsignedShort();
333             ArrayList<Integer> code = new ArrayList<Integer>();
334             long codeLength = in.readUnsignedInt();
335             for (long l = 0; l < codeLength; l++) {
336                 code.add(in.readUnsignedByte());
337             }
338             int exceptionTableLength = in.readUnsignedShort();
339             ArrayList<CLExceptionInfo> exceptionTable = new ArrayList<CLExceptionInfo>();
340             for (int l = 0; l < exceptionTableLength; l++) {
341                 int startPC   = in.readUnsignedShort();
342                 int endPC     = in.readUnsignedShort();
343                 int handlerPC = in.readUnsignedShort();
344                 int catchType = in.readUnsignedShort();
345                 exceptionTable.add(new CLExceptionInfo(startPC, endPC,
346                                                        handlerPC, catchType));
347             }
348             int codeAttrAttributesCount = in.readUnsignedShort();
349             ArrayList<CLAttributeInfo> codeAttrAttributes = readAttributes(in,
350                     codeAttrAttributesCount);
351             attribute = new CLCodeAttribute(attributeNameIndex, attributeLength,
352                                             maxStack, maxLocals, 
353                                             codeLength, code,
354                                             exceptionTableLength, exceptionTable,
355                                             codeAttrAttributesCount, codeAttrAttributes);
356         } catch (IOException e) {
357             reportError("Error reading Code_attribute from file %s", className);
358         }
359         return attribute;
360     }

859     /**
860      * Read an ElementValue from the specified input stream, and return it.
861      * 
862      * @param in
863      *            input stream.
864      * @return an ElemenvValue.
865      */
866 
867     private CLElementValue readElementValue(CLInputStream in) {
868         CLElementValue elementValue = null;
869         try {
870             int tag = in.readUnsignedByte();
871             switch (tag) {
872             case ELT_B:
873             case ELT_C:
874             case ELT_D:
875             case ELT_F:
876             case ELT_I:
877             case ELT_J:
878             case ELT_S:
879             case ELT_Z:
880             case ELT_s:
881                 elementValue = new CLElementValue((short) tag, in
882                         .readUnsignedShort());
883                 break;
884             case ELT_e:
885                 elementValue = new CLElementValue(in.readUnsignedShort(), in
886                         .readUnsignedShort());
887                 break;
888             case ELT_c:
889                 elementValue = new CLElementValue(in.readUnsignedShort());
890                 break;
891             case ELT_ANNOTATION:
892                 elementValue = new CLElementValue(readAnnotation(in));
893                 break;
894             case ELT_ARRAY:
895                 int numValues = in.readUnsignedShort();
896                 ArrayList<CLElementValue> values = new ArrayList<CLElementValue>();
897                 for (int i = 0; i < numValues; i++) {
898                     values.add(readElementValue(in));
899                 }
900                 elementValue = new CLElementValue(numValues, values);
901             }
902         } catch (IOException e) {
903             reportError("Error reading AnnotationDefault_attribute "
904                     + "from file %s", className);
905         }
906         return elementValue;
907     }

910      * Read an Annotation from the specified input stream, and return it.
911      * 
912      * @param in
913      *            input stream.
914      * @return an Annotation.
915      */
916 
917     private CLAnnotation readAnnotation(CLInputStream in) {
918         CLAnnotation annotation = null;
919         try {
920             int typeIndex = in.readUnsignedShort();
921             int numElementValuePairs = in.readUnsignedShort();
922             ArrayList<CLElementValuePair> elementValuePairs = new ArrayList<CLElementValuePair>();
923             for (int i = 0; i < numElementValuePairs; i++) {
924                 int elementNameIndex = in.readUnsignedShort();
925                 CLElementValue value = readElementValue(in);
926                 elementValuePairs.add(new CLElementValuePair(elementNameIndex,
927                         value));
928             }
929             annotation = new CLAnnotation(typeIndex, numElementValuePairs,
930                     elementValuePairs);
931         } catch (IOException e) {
932             reportError("Error reading Annotation from file %s", className);
933         }
934         return annotation;

937     /**
938      * Construct a CLAbsorber object given the (fully-qualified) name of the
939      * class file to read.
940      * 
941      * @param className
942      *            fully qualified name of the input class file.
943      */
944 
945     public CLAbsorber(String className) {
946         try {
947             this.className = className;
948             CLPath classPath = new CLPath();
949             CLInputStream in = classPath.loadClass(className);
950             errorHasOccurred = false;
951             if (in == null) {
952                 reportError("Error loading %s", className);
953                 return;
954             }
955             classFile = new CLFile();
956 
957             // Read magic number (0xCAFEBABE)
958             long magic = in.readUnsignedInt();
959             if (magic != MAGIC) {
960                 reportWarning("%s has an invalid magic number", className);
961                 return;
962             }
963             classFile.magic = magic;
964 
965             // Read minor version, major version
966             classFile.minorVersion = in.readUnsignedShort();
967             classFile.majorVersion = in.readUnsignedShort();
968 
969             // Read constant pool count, constant pool
970             classFile.constantPoolCount = in.readUnsignedShort();
971             classFile.constantPool = readConstantPool(in);
972             if (errorHasOccurred()) {
973                 return;
974             }
975 
976             // Read access flags for the class
977             classFile.accessFlags = in.readUnsignedShort();
978 
979             // Read this class' constant pool index
980             classFile.thisClass = in.readUnsignedShort();
981 
982             // Read super class' constant pool index
983             classFile.superClass = in.readUnsignedShort();
984 
985             // Read interfaces (implemented by the class being
986             // read) count and interfaces (constant pool
987             // indices)
988             int interfacesCount = in.readUnsignedShort();
989             ArrayList<Integer> interfaces = new ArrayList<Integer>();
990             classFile.interfacesCount = interfacesCount;
991             for (int i = 0; i < interfacesCount; i++) {
992                 interfaces.add(in.readUnsignedShort());
993             }
994             classFile.interfaces = interfaces;
995 
996             // Read fields count and fields
997             classFile.fieldsCount = in.readUnsignedShort();
998             classFile.fields = readFields(in, classFile.fieldsCount);
999             if (errorHasOccurred()) {
1000                return;
1001            }
1002
1003            // Read methods count and methods
1004            classFile.methodsCount = in.readUnsignedShort();
1005            classFile.methods = readMethods(in, classFile.methodsCount);
1006            if (errorHasOccurred()) {
1007                return;
1008            }
1009
1010            // Read class attributes
1011            classFile.attributesCount = in.readUnsignedShort();
1012            classFile.attributes = readAttributes(in, classFile.attributesCount);
1013        } catch (EOFException e) {
1014            reportError("Unexpected end of file %s", className);
1015        } catch (IOException e) {
1016            reportError("Error reading file %s", className);
1017        }
1018    }

1041    /**
1042     * Driver for CLAbsorber. It accepts the (fully-qualified) name of a class
1043     * file as command-line argument and dumps its (ClassFile) structure --
1044     * {@code CLFile} in our representation -- to STDOUT in a format similar to 
1045     * that of javap.
1046     *
1047     * @param args
1048     *            the command-line arguments.
1049     */
1050
1051    public static void main(String[] args) {
1052        String classFile = "";
1053        if (args.length == 1) {
1054            classFile = args[0];
1055        } else {
1056            String usage = "Usage: java jminusminus.CLAbsorber <class name>\n"
1057                    + "Where the class name must be fully qualified; "
1058                    + "eg, java/util/ArrayList";
1059            System.out.println(usage);
1060            System.exit(0);
1061        }
1062        CLAbsorber r = new CLAbsorber(classFile);
1063        if (!r.errorHasOccurred()) {
1064            CLFile c = r.classFile();
1065            c.writeToStdOut();
1066        }
1067    }

1071/**
1072 * Inherits from java.io.DataInputStream and provides an extra function for
1073 * reading unsigned int from the input stream, which is required for reading
1074 * Java class files.
1075 */
1076
1077class CLInputStream extends DataInputStream {
1078
1079    /**
1080     * Constructs a CLInputStream object from the specified input stream.
1081     * 
1082     * @param in
1083     *            input stream.
1084     */
1085
1086    public CLInputStream(InputStream in) {
1087        super(in);
1088    }
1089
1090    /**
1091     * Reads four input bytes and returns a {@code long} value in the range 
1092     * 0 through 4294967295. Let a, b, c, d be the four bytes. 
1093     * The value returned is:
1094     * 
1095     * <pre>
1096     *   ( b[ 0 ] &amp; 0xFF ) &lt;&lt; 24 ) | 
1097     *   ( ( b[ 1 ] &amp; 0xFF ) &lt;&lt; 16 ) | 
1098     *   ( ( b[ 2 ] &amp; 0xFF ) &lt;&lt; 8 ) | 
1099     *   ( b[ 3 ] &amp; 0xFF )
1100     * </pre>
1101     * 
1102     * @return the unsigned 32-bit value.
1103     * @exception EOFException
1104     *                if this stream reaches the end before reading all the
1105     *                bytes.
1106     * @exception IOException
1107     *                if an I/O error occurs.
1108     */
1109
1110    public final long readUnsignedInt() throws IOException {
1111        byte[] b = new byte[4];
1112        long mask = 0xFF, l;
1113        in.read(b);
1114        l = ((b[0] & mask) << 24) | ((b[1] & mask) << 16)
1115                | ((b[2] & mask) << 8) | (b[3] & mask);
1116        return l;
1117    }
