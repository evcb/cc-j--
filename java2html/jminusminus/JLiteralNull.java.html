3   package jminusminus;
4   
5   import static jminusminus.CLConstants.*;
6   
7   /**
8    * The AST node for the {@code null} literal.
9    */
10  
11  class JLiteralNull extends JExpression {
12  
13      /**
14       * Constructs an AST node for the {@code null} literal given its line 
15       * number.
16       * 
17       * @param line
18       *            line in which the literal occurs in the source file.
19       */
20  
21      public JLiteralNull(int line) {
22          super(line);
23      }
24  
25      /**
26       * Analyzing the null literal is trivial.
27       * 
28       * @param context
29       *            context in which names are resolved (ignored here).
30       * @return the analyzed (and possibly rewritten) AST subtree.
31       */
32  
33      public JExpression analyze(Context context) {
34          type = Type.NULLTYPE;
35          return this;
36      }
37  
38      /**
39       * Generating code for a null literal means generating code to push it onto
40       * the stack.
41       * 
42       * @param output
43       *            the code emitter (basically an abstraction for producing the
44       *            .class file).
45       */
46  
47      public void codegen(CLEmitter output) {
48          output.addNoArgInstruction(ACONST_NULL);
49      }
50  
51      /**
52       * {@inheritDoc}
53       */
54  
55      public void writeToStdOut(PrettyPrinter p) {
56          p.printf("<JLiteralNull line=\"%d\" type=\"%s\"/>\n", line(),
57                  ((type == null) ? "" : type.toString()));
58      }
59  
60  }
