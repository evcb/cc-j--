14  /**
15   * Representation of a JVM instruction. It stores the opcode and mnenomic of an
16   * instruction, its operand count (DYNAMIC if the instruction has variable
17   * number operands), pc (location counter), stack units (words produced - words
18   * consumed from the operand stack), and local variable index (IRRELEVANT if the
19   * instruction does not operate on local variables).
20   *
21   * @see CLConstants
22   */

371     /**
372      * Returns the byte from i at position byteNum.
373      * 
374      * @param i
375      *            number whose individual byte is required.
376      * @param byteNum
377      *            the byte to return; 1 (lower) - 4 (higher) instructions.
378      * @return the byte at the specified position.
379      */
380 
381     protected int byteAt(int i, int byteNum) {
382         int j = 0, mask = 0xFF;
383         switch (byteNum) {
384         case 1: // lower order
385             j = i & mask;
386             break;
387         case 2:
388             j = (i >> 8) & mask;
389             break;

1061     * @param matchLabelPairs
1062     *            key match table.

1074        this.matchLabelPairs = matchLabelPairs;
1075        pad = 4 - ((pc + 1) % 4);
1076        if (pad == 4) {
1077            pad = 0;
1078        }
1079        operandCount = pad + 8 + 8 * numPairs;
1080    }

1082    /**
1083     * Resolves the jump labels to the corresponding offset values using the
1084     * given label to pc mapping. If unable to resolve a label, the offset is
1085     * set such that the next instruction will be executed.
1086     * 
1087     * @param labelToPC
1088     *            label to pc mapping.
1089     * @return {@code true} if all labels were resolved successfully; 
1090     *         {@code false} otherwise.
1091     */
1092
1093    public boolean resolveLabels(Hashtable<String, Integer> labelToPC) {
1094        boolean allLabelsResolved = true;
1095        if (instructionInfo[opcode].category == FLOW_CONTROL1) {
1096            if (labelToPC.containsKey(jumpToLabel)) {
1097                jumpToOffset = labelToPC.get(jumpToLabel) - pc;
1098            } else {
1099                jumpToOffset = operandCount;
1100                allLabelsResolved = false;
1101            }
1102        } else if (opcode == LOOKUPSWITCH) {
1103            if (labelToPC.containsKey(defaultLabel)) {
1104                defaultOffset = labelToPC.get(defaultLabel) - pc;
1105            } else {
1106                defaultOffset = operandCount;
1107                allLabelsResolved = false;
1108            }
1109            matchOffsetPairs = new TreeMap<Integer, Integer>();
1110            Set<Entry<Integer, String>> matches = matchLabelPairs.entrySet();
1111            Iterator<Entry<Integer, String>> iter = matches.iterator();
1112            while (iter.hasNext()) {
1113                Entry<Integer, String> entry = iter.next();
1114                int match = entry.getKey();
1115                String label = entry.getValue();
1116                if (labelToPC.containsKey(label)) {
1117                    matchOffsetPairs.put(match, labelToPC.get(label) - pc);
1118                } else {
1119                    matchOffsetPairs.put(match, operandCount);
1120                    allLabelsResolved = false;
1121                }
1122            }
1123        } else if (opcode == TABLESWITCH) {
1124            if (labelToPC.containsKey(defaultLabel)) {
1125                defaultOffset = labelToPC.get(defaultLabel) - pc;
1126            } else {
1127                defaultOffset = operandCount;
1128                allLabelsResolved = false;
1129            }
1130            offsets = new ArrayList<Integer>();
1131            for (int i = 0; i < labels.size(); i++) {
1132                if (labelToPC.containsKey(labels.get(i))) {
1133                    offsets.add(labelToPC.get(labels.get(i)) - pc);
1134                } else {
1135                    offsets.add(operandCount);
1136                    allLabelsResolved = false;
1137                }
1138            }
1139        }
1140        return allLabelsResolved;
1141    }

1143    /**
1144     * Returns the pc of instruction to jump to.
1145     * 
1146     * @return pc to jump to.
1147     */
1148
1149    public int jumpToOffset() {
1150        return jumpToOffset;
1151    }

1153    /**
1154     * {@inheritDoc}
1155     */
1156
1157    public ArrayList<Integer> toBytes() {
1158        ArrayList<Integer> bytes = new ArrayList<Integer>();
1159        bytes.add(opcode);
1160        switch (opcode) {
1161        case RET:
1162            if (isWidened) {
1163                bytes.add(byteAt(index, 2));
1164                bytes.add(byteAt(index, 1));
1165            } else {
1166                bytes.add(byteAt(index, 1));
1167            }
1168            break;
1169        case TABLESWITCH:
1170            for (int i = 0; i < pad; i++) {
1171                bytes.add(0);
1172            }
1173            bytes.add(byteAt(defaultOffset, 4));
1174            bytes.add(byteAt(defaultOffset, 3));
1175            bytes.add(byteAt(defaultOffset, 2));
1176            bytes.add(byteAt(defaultOffset, 1));
1177            bytes.add(byteAt(low, 4));
1178            bytes.add(byteAt(low, 3));
1179            bytes.add(byteAt(low, 2));
1180            bytes.add(byteAt(low, 1));
1181            bytes.add(byteAt(high, 4));
1182            bytes.add(byteAt(high, 3));
1183            bytes.add(byteAt(high, 2));
1184            bytes.add(byteAt(high, 1));
1185            for (int i = 0; i < offsets.size(); i++) {
1186                int jumpOffset = offsets.get(i);
1187                bytes.add(byteAt(jumpOffset, 4));
1188                bytes.add(byteAt(jumpOffset, 3));
1189                bytes.add(byteAt(jumpOffset, 2));
1190                bytes.add(byteAt(jumpOffset, 1));
1191            }
1192            break;
1193        case LOOKUPSWITCH:
1194            for (int i = 0; i < pad; i++) {
1195                bytes.add(0);
1196            }
1197            bytes.add(byteAt(defaultOffset, 4));
1198            bytes.add(byteAt(defaultOffset, 3));
1199            bytes.add(byteAt(defaultOffset, 2));
1200            bytes.add(byteAt(defaultOffset, 1));
1201            bytes.add(byteAt(numPairs, 4));
1202            bytes.add(byteAt(numPairs, 3));
1203            bytes.add(byteAt(numPairs, 2));
1204            bytes.add(byteAt(numPairs, 1));
1205            Set<Entry<Integer, Integer>> matches = matchOffsetPairs.entrySet();
1206            Iterator<Entry<Integer, Integer>> iter = matches.iterator();
1207            while (iter.hasNext()) {
1208                Entry<Integer, Integer> entry = iter.next();
1209                int match = entry.getKey();
1210                int offset = entry.getValue();
1211                bytes.add(byteAt(match, 4));
1212                bytes.add(byteAt(match, 3));
1213                bytes.add(byteAt(match, 2));
1214                bytes.add(byteAt(match, 1));
1215                bytes.add(byteAt(offset, 4));
1216                bytes.add(byteAt(offset, 3));
1217                bytes.add(byteAt(offset, 2));
1218                bytes.add(byteAt(offset, 1));
1219            }
1220            break;
1221        case GOTO_W:
1222        case JSR_W:
1223            bytes.add(byteAt(jumpToOffset, 4));
1224            bytes.add(byteAt(jumpToOffset, 3));
1225            bytes.add(byteAt(jumpToOffset, 2));
1226            bytes.add(byteAt(jumpToOffset, 1));
1227            break;
1228        default:
1229            bytes.add(byteAt(jumpToOffset, 2));
1230            bytes.add(byteAt(jumpToOffset, 1));
1231        }
1232        return bytes;
1233    }
