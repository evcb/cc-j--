8   /**
9    * A class declaration has a list of modifiers, a name, a super class and a
10   * class block; it distinguishes between instance fields and static (class)
11   * fields for initialization, and it defines a type. It also introduces its own
12   * (class) context.
13   *
14   * @see ClassContext
15   */
16  
17  class JClassDeclaration extends JAST implements JTypeDecl {
18  
19      /** Class modifiers. */
20      private ArrayList<String> mods;
21  
22      /** Class name. */
23      private String name;
24  
25      /** Class block. */
26      private ArrayList<JMember> classBlock;
27  
28      /** Super class type. */
29      private Type superType;
30  
31      /** This class type. */
32      private Type thisType;
33  
34      /** Context for this class. */
35      private ClassContext context;
36  
37      /** Whether this class has an explicit constructor. */
38      private boolean hasExplicitConstructor;
39  
40      /** Instance fields of this class. */
41      private ArrayList<JFieldDeclaration> instanceFieldInitializations;
42  
43      /** Static (class) fields of this class. */
44      private ArrayList<JFieldDeclaration> staticFieldInitializations;
45  
46      /**
47       * Constructs an AST node for a class declaration given the line number, list
48       * of class modifiers, name of the class, its super class type, and the
49       * class block.
50       * 
51       * @param line
52       *            line in which the class declaration occurs in the source file.
53       * @param mods
54       *            class modifiers.
55       * @param name
56       *            class name.
57       * @param superType
58       *            super class type.
59       * @param classBlock
60       *            class block.
61       */
62  
63      public JClassDeclaration(int line, ArrayList<String> mods, String name,
64              Type superType, ArrayList<JMember> classBlock) {
65          super(line);
66          this.mods = mods;
67          this.name = name;
68          this.superType = superType;
69          this.classBlock = classBlock;
70          hasExplicitConstructor = false;
71          instanceFieldInitializations = new ArrayList<JFieldDeclaration>();
72          staticFieldInitializations = new ArrayList<JFieldDeclaration>();
73      }

76       * Returns the class name.

78       * @return the class name.

83      }

86       * Returns the class' super class type.

92          return superType;

116     /**
117      * Declares this class in the parent (compilation unit) context.
118      * 
119      * @param context
120      *            the parent (compilation unit) context.
121      */
122 
123     public void declareThisType(Context context) {
124         String qualifiedName = JAST.compilationUnit.packageName() == "" ? name
125                 : JAST.compilationUnit.packageName() + "/" + name;
126         CLEmitter partial = new CLEmitter(false);
127         partial.addClass(mods, qualifiedName, Type.OBJECT.jvmName(), null,
128                 false); // Object for superClass, just for now
129         thisType = Type.typeFor(partial.toClass());
130         context.addType(line, thisType);
131     }
132 
133     /**
134      * Pre-analyzes the members of this declaration in the parent context.
135      * Pre-analysis extends to the member headers (including method headers) but
136      * not into the bodies.
137      * 
138      * @param context
139      *            the parent (compilation unit) context.
140      */
141 
142     public void preAnalyze(Context context) {
143         // Construct a class context
144         this.context = new ClassContext(this, context);
145 
146         // Resolve superclass
147         superType = superType.resolve(this.context);
148 
149         // Creating a partial class in memory can result in a
150         // java.lang.VerifyError if the semantics below are
151         // violated, so we can't defer these checks to analyze()
152         thisType.checkAccess(line, superType);
153         if (superType.isFinal()) {
154             JAST.compilationUnit.reportSemanticError(line,
155                     "Cannot extend a final type: %s", superType.toString());
156         }
157 
158         // Create the (partial) class
159         CLEmitter partial = new CLEmitter(false);
160 
161         // Add the class header to the partial class
162         String qualifiedName = JAST.compilationUnit.packageName() == "" ? name
163                 : JAST.compilationUnit.packageName() + "/" + name;
164         partial.addClass(mods, qualifiedName, superType.jvmName(), null, false);
165 
166         // Pre-analyze the members and add them to the partial
167         // class
168         for (JMember member : classBlock) {
169             member.preAnalyze(this.context, partial);
170             if (member instanceof JConstructorDeclaration
171                     && ((JConstructorDeclaration) member).params.size() == 0) {
172                 hasExplicitConstructor = true;
173             }
174         }
175 
176         // Add the implicit empty constructor?
177         if (!hasExplicitConstructor) {
178             codegenPartialImplicitConstructor(partial);
179         }
180 
181         // Get the Class rep for the (partial) class and make it
182         // the
183         // representation for this type
184         Type id = this.context.lookupType(name);
185         if (id != null && !JAST.compilationUnit.errorHasOccurred()) {
186             id.setClassRep(partial.toClass());
187         }
188     }
189 
190     /**
191      * Performs semantic analysis on the class and all of its members within the
192      * given context. Analysis includes field initializations and the method
193      * bodies.
194      * 
195      * @param context
196      *            the parent (compilation unit) context. Ignored here.
197      * @return the analyzed (and possibly rewritten) AST subtree.
198      */
199 
200     public JAST analyze(Context context) {
201         // Analyze all members
202         for (JMember member : classBlock) {
203             ((JAST) member).analyze(this.context);
204         }
205 
206         // Copy declared fields for purposes of initialization.
207         for (JMember member : classBlock) {
208             if (member instanceof JFieldDeclaration) {
209                 JFieldDeclaration fieldDecl = (JFieldDeclaration) member;
210                 if (fieldDecl.mods().contains("static")) {
211                     staticFieldInitializations.add(fieldDecl);
212                 } else {
213                     instanceFieldInitializations.add(fieldDecl);
214                 }
215             }
216         }
217 
218         // Finally, ensure that a non-abstract class has
219         // no abstract methods.
220         if (!thisType.isAbstract() && thisType.abstractMethods().size() > 0) {
221             String methods = "";
222             for (Method method : thisType.abstractMethods()) {
223                 methods += "\n" + method;
224             }
225             JAST.compilationUnit.reportSemanticError(line,
226                     "Class must be declared abstract since it defines "
227                             + "the following abstract methods: %s", methods);
228 
229         }
230         return this;
231     }

233     /**
234      * Generates code for the class declaration.
235      * 
236      * @param output
237      *            the code emitter (basically an abstraction for producing the
238      *            .class file).
239      */
240 
241     public void codegen(CLEmitter output) {
242         // The class header
243         String qualifiedName = JAST.compilationUnit.packageName() == "" ? name
244                 : JAST.compilationUnit.packageName() + "/" + name;
245         output.addClass(mods, qualifiedName, superType.jvmName(), null, false);
246 
247         // The implicit empty constructor?
248         if (!hasExplicitConstructor) {
249             codegenImplicitConstructor(output);
250         }
251 
252         // The members
253         for (JMember member : classBlock) {
254             ((JAST) member).codegen(output);
255         }
256 
257         // Generate a class initialization method?
258         if (staticFieldInitializations.size() > 0) {
259             codegenClassInit(output);
260         }
261     }

263     /**
264      * {@inheritDoc}
265      */
266 
267     public void writeToStdOut(PrettyPrinter p) {
268         p.printf("<JClassDeclaration line=\"%d\" name=\"%s\""
269                 + " super=\"%s\">\n", line(), name, superType.toString());
270         p.indentRight();
271         if (context != null) {
272             context.writeToStdOut(p);
273         }
274         if (mods != null) {
275             p.println("<Modifiers>");
276             p.indentRight();
277             for (String mod : mods) {
278                 p.printf("<Modifier name=\"%s\"/>\n", mod);
279             }
280             p.indentLeft();
281             p.println("</Modifiers>");
282         }
283         if (classBlock != null) {
284             p.println("<ClassBlock>");
285             p.indentRight();
286             for (JMember member : classBlock) {
287                 ((JAST) member).writeToStdOut(p);
288             }
289             p.indentLeft();
290             p.println("</ClassBlock>");
291         }
292         p.indentLeft();
293         p.println("</JClassDeclaration>");
294     }

296     /**
297      * Generates code for an implicit empty constructor. (Necessary only if there
298      * is not already an explicit one.)
299      * 
300      * @param partial
301      *            the code emitter (basically an abstraction for producing a
302      *            Java class).
303      */
304 
305     private void codegenPartialImplicitConstructor(CLEmitter partial) {
306         // Invoke super constructor
307         ArrayList<String> mods = new ArrayList<String>();
308         mods.add("public");
309         partial.addMethod(mods, "<init>", "()V", null, false);
310         partial.addNoArgInstruction(ALOAD_0);
311         partial.addMemberAccessInstruction(INVOKESPECIAL, superType.jvmName(),
312                 "<init>", "()V");
313 
314         // Return
315         partial.addNoArgInstruction(RETURN);
316     }
317 
318     /**
319      * Generates code for an implicit empty constructor. (Necessary only if there
320      * is not already an explicit one.
321      * 
322      * @param output
323      *            the code emitter (basically an abstraction for producing the
324      *            .class file).
325      */
326 
327     private void codegenImplicitConstructor(CLEmitter output) {
328         // Invoke super constructor
329         ArrayList<String> mods = new ArrayList<String>();
330         mods.add("public");
331         output.addMethod(mods, "<init>", "()V", null, false);
332         output.addNoArgInstruction(ALOAD_0);
333         output.addMemberAccessInstruction(INVOKESPECIAL, superType.jvmName(),
334                 "<init>", "()V");
335 
336         // If there are instance field initializations, generate
337         // code for them
338         for (JFieldDeclaration instanceField : instanceFieldInitializations) {
339             instanceField.codegenInitializations(output);
340         }
341 
342         // Return
343         output.addNoArgInstruction(RETURN);
344     }
345 
346     /**
347      * Generates code for class initialization, in j-- this means static field
348      * initializations.
349      * 
350      * @param output
351      *            the code emitter (basically an abstraction for producing the
352      *            .class file).
353      */
354 
355     private void codegenClassInit(CLEmitter output) {
356         ArrayList<String> mods = new ArrayList<String>();
357         mods.add("public");
358         mods.add("static");
359         output.addMethod(mods, "<clinit>", "()V", null, false);
360 
361         // If there are instance initializations, generate code
362         // for them
363         for (JFieldDeclaration staticField : staticFieldInitializations) {
364             staticField.codegenInitializations(output);
365         }
366 
367         // Return
368         output.addNoArgInstruction(RETURN);
369     }
