14  /**
15   * A tuple representation of a JVM instruction.
16   */
17  
18  class NTuple {
19  
20      /** Program counter of the instruction. */
21      public int pc;
22  
23      /** Opcode of the instruction. */
24      public int opcode;
25  
26      /** Operands of the instructions. */
27      public ArrayList<Short> operands;
28  
29      /** String representation (mnemonic) of the instruction. */
30      public String mnemonic;
31  
32      /** Is this tuple the leader of the block containing it. */
33      public boolean isLeader;
34  
35      /**
36       * Construct a tuple representing the JVM instruction with the given program
37       * counter, opcode, and operand list.
38       * 
39       * @param pc
40       *            program counter.
41       * @param opcode
42       *            opcode of the instruction.
43       * @param operands
44       *            list of operands of the instruction.
45       */
46  
47      public NTuple(int pc, int opcode, ArrayList<Short> operands) {
48          this.pc = pc;
49          this.opcode = opcode;
50          this.operands = operands;
51          this.mnemonic = CLInstruction.instructionInfo[opcode].mnemonic;
52          this.isLeader = false;
53      }
54  
55      /**
56       * Write the information pertaining to this tuple to STDOUT.
57       * 
58       * @param p
59       *            for pretty printing with indentation.
60       */
61  
62      public void writeToStdOut(PrettyPrinter p) {
63          p.printf("%s: %s", pc, mnemonic);
64          for (short s : operands) {
65              p.printf(" %s", s);
66          }
67          p.printf("\n");
68      }
69  
70  }

218     /**
219      * Write the HIR instructions in this block to STDOUT.
220      * 
221      * @param p
222      *            for pretty printing with indentation.
223      */
224 
225     public void writeHirToStdOut(PrettyPrinter p) {
226         String s = id() + (isLoopHead ? " [LH]" : "")
227                 + (isLoopTail ? " [LT]" : "");
228         if (tuples.size() > 0) {
229             s += " [" + tuples.get(0).pc + ", "
230                     + tuples.get(tuples.size() - 1).pc + "]";
231         }
232         if (dom != null) {
233             s += " dom: " + dom.id();
234         }
235         if (predecessors.size() > 0) {
236             s += " pred: ";
237             for (NBasicBlock block : predecessors) {
238                 s += block.id() + " ";
239             }
240         }
241         if (successors.size() > 0) {
242             s += " succ: ";
243             for (NBasicBlock block : successors) {
244                 s += block.id() + " ";
245             }
246         }
247         p.printf(s + "\n");
248         s = "Locals: ";
249         if (locals != null) {
250             for (int i = 0; i < locals.length; i++) {
251                 if (!(cfg.hirMap.get(locals[i]) instanceof NHIRLocal)) {
252                     s += cfg.hirMap.get(locals[i]).id() + " ";
253                 }
254             }
255         }
256         p.printf("%s\n", s);
257         for (int ins : hir) {
258             if (cfg.hirMap.get(ins) instanceof NHIRPhiFunction) {
259                 p.printf("%s: %s\n", ((NHIRPhiFunction) cfg.hirMap.get(ins))
260                         .id(), ((NHIRPhiFunction) cfg.hirMap.get(ins)));
261             }
262         }
263         for (int ins : hir) {
264             if (!(cfg.hirMap.get(ins) instanceof NHIRPhiFunction)) {
265                 p.printf("%s\n", cfg.hirMap.get(ins));
266             }
267         }
268         p.printf("\n");
269     }

309     /**
310      * Iterates through the lir array of this block, returning an
311      * NLIRInstruction with the specified id.
312      * 
313      * @param id
314      *            the id to look for.
315      * 
316      * @return NLIRInstruction with the specified id, null if none matched.
317      */
318     public NLIRInstruction getInstruction(int id) {
319         for (NLIRInstruction i : this.lir) {
320             if (i.id == id) {
321                 return i;
322             }
323         }
324         return null;
325     }

327     /**
328      * Checks to see if there is an LIRInstruction with this id in the block's
329      * lir.
330      * 
331      * @param id
332      *            the id to check for.
333      * @return {@code true} if there is a LIRInstruction with this id;
334      *         {@code false} otherwise.
335      */
336     public boolean idIsFree(int id) {
337         if (this.getInstruction(id) != null) {
338             return false;
339         } else {
340             return true;
341         }
342     }

344     /**
345      * Inserts an NLIRInstruction to the appropriate place in this block's lir
346      * array based on its id -- preserving order by id.
347      * 
348      * @param inst
349      *            the NLIRInstruction to be inserted.
350      */
351     public void insertLIRInst(NLIRInstruction inst) {
352         int idx = -1;
353         for (int i = 0; i < this.lir.size(); i++) {
354             if (this.lir.get(i).id < inst.id) {
355                 idx = i;
356             }
357         }
358         if (++idx == this.lir.size()) {
359             this.lir.add(inst);
360         } else {
361             this.lir.add(idx, inst);
362         }
363     }

444     /**
445      * Constructs a NControlFlowGraph object for a method given the constant
446      * pool for the class containing the method and the object containing
447      * information about the method.
448      * 
449      * @param cp
450      *            constant pool for the class containing the method.
451      * @param m
452      *            contains information about the method.
453      */
454 
455     public NControlFlowGraph(CLConstantPool cp, CLMethodInfo m) {
456         this.cp = cp;
457         this.m = m;
458         name = new String(((CLConstantUtf8Info) cp.cpItem(m.nameIndex)).b);
459         desc = new String(((CLConstantUtf8Info) cp.cpItem(m.descriptorIndex)).b);
460         basicBlocks = new ArrayList<NBasicBlock>();
461         pcToBasicBlock = new HashMap<Integer, NBasicBlock>();
462         ArrayList<Integer> code = getByteCode();
463         ArrayList<NTuple> tuples = bytecodeToTuples(code);
464         if (tuples.size() == 0) {
465             return;
466         }
467         NTuple[] tupleAt = new NTuple[code.size()];
468         for (NTuple tuple : tuples) {
469             tupleAt[tuple.pc] = tuple;
470         }
471 
472         // Identify the leaders
473         findLeaders(tuples, tupleAt);
474         
475         // Form blocks
476         buildBB(tuples);
477 
478         // Connect up the blocks for this method, that is, build
479         // its control flow graph.
480         basicBlocks.get(0).successors.add(basicBlocks.get(1));
481         basicBlocks.get(1).predecessors.add(basicBlocks.get(0));
482         NBasicBlock[] blockAt = new NBasicBlock[code.size()];
483         for (NBasicBlock block : basicBlocks) {
484             if (block.tuples.size() == 0) {
485                 continue;
486             }
487             blockAt[block.tuples.get(0).pc] = block;
488         }
489         for (int j = 0; j < basicBlocks.size(); j++) {
490             NBasicBlock block = basicBlocks.get(j);
491             if (block.tuples.size() == 0) {
492                 continue;
493             }
494             NTuple tuple = block.tuples.get(block.tuples.size() - 1);
495             short operandByte1, operandByte2, operandByte3, operandByte4;
496             int offset;
497             NBasicBlock target;
498             switch (tuple.opcode) {
499             case IFEQ:
500             case IFNE:
501             case IFLT:
502             case IFGE:
503             case IFGT:
504             case IFLE:
505             case IF_ICMPEQ:
506             case IF_ICMPNE:
507             case IF_ICMPLT:
508             case IF_ICMPGE:
509             case IF_ICMPGT:
510             case IF_ICMPLE:
511             case IF_ACMPEQ:
512             case IF_ACMPNE:
513             case IFNULL:
514             case IFNONNULL:
515                 operandByte1 = tuple.operands.get(0);
516                 operandByte2 = tuple.operands.get(1);
517                 offset = shortValue(operandByte1, operandByte2);
518                 target = blockAt[tuple.pc + offset];
519                 if (j < basicBlocks.size() - 1) {
520                     block.successors.add(basicBlocks.get(j + 1));
521                     basicBlocks.get(j + 1).predecessors.add(block);
522                 }
523                 block.successors.add(target);
524                 target.predecessors.add(block);
525                 break;
526             case GOTO:
527             case JSR:
528                 operandByte1 = tuple.operands.get(0);
529                 operandByte2 = tuple.operands.get(1);
530                 offset = shortValue(operandByte1, operandByte2);
531                 target = blockAt[tuple.pc + offset];
532                 block.successors.add(target);
533                 target.predecessors.add(block);
534                 break;
535             case GOTO_W:
536             case JSR_W:
537                 operandByte1 = tuple.operands.get(0);
538                 operandByte2 = tuple.operands.get(1);
539                 operandByte3 = tuple.operands.get(2);
540                 operandByte4 = tuple.operands.get(3);
541                 offset = intValue(operandByte1, operandByte2, operandByte3,
542                         operandByte4);
543                 target = blockAt[tuple.pc + offset];
544                 block.successors.add(target);
545                 target.predecessors.add(block);
546                 break;
547             case IRETURN:
548             case LRETURN:
549             case FRETURN:
550             case DRETURN:
551             case ARETURN:
552             case RETURN:
553             case RET:
554             case ATHROW:
555                 break;
556             case TABLESWITCH: // TBD
557                 break;
558             case LOOKUPSWITCH: // TBD
559                 break;
560             default:
561                 if (j < basicBlocks.size() - 1) {
562                     block.successors.add(basicBlocks.get(j + 1));
563                     basicBlocks.get(j + 1).predecessors.add(block);
564                 }
565             }
566         }
567 
568         // Calculate the ref count and number of forward branches
569         // to each block in the this cfg.
570         for (NBasicBlock block : basicBlocks) {
571             block.ref = block.predecessors.size();
572             block.fwdBranches = block.predecessors.size() - block.bwdBranches;
573         }
574     }
575 
576     /**
577      * Finds the leaders for this control flow graph.
578      *
579      * @param tuples
580      *            list of tuples representing the JVM instructions.
581      * @param tupleAt
582      *            array of tuples at each program counter number.
583      */
584 
585     private void findLeaders(ArrayList<NTuple> tuples, NTuple[] tupleAt) {
586         tuples.get(0).isLeader = true;
587         for (int j = 1; j < tuples.size(); j++) {
588             NTuple tuple = tuples.get(j);
589             boolean jumpInstruction = true;
590             short operandByte1, operandByte2, operandByte3, operandByte4;
591             int offset;
592             switch (tuple.opcode) {
593             case IFEQ:
594             case IFNE:
595             case IFLT:
596             case IFGE:
597             case IFGT:
598             case IFLE:
599             case IF_ICMPEQ:
600             case IF_ICMPNE:
601             case IF_ICMPLT:
602             case IF_ICMPGE:
603             case IF_ICMPGT:
604             case IF_ICMPLE:
605             case IF_ACMPEQ:
606             case IF_ACMPNE:
607             case GOTO:
608             case JSR:
609             case IFNULL:
610             case IFNONNULL:
611                 operandByte1 = tuple.operands.get(0);
612                 operandByte2 = tuple.operands.get(1);
613                 offset = shortValue(operandByte1, operandByte2);
614                 tupleAt[tuple.pc + offset].isLeader = true;
615                 break;
616             case GOTO_W:
617             case JSR_W:
618                 operandByte1 = tuple.operands.get(0);
619                 operandByte2 = tuple.operands.get(1);
620                 operandByte3 = tuple.operands.get(2);
621                 operandByte4 = tuple.operands.get(3);
622                 offset = intValue(operandByte1, operandByte2, operandByte3,
623                         operandByte4);
624                 tupleAt[tuple.pc + offset].isLeader = true;
625                 break;
626             case IRETURN:
627             case LRETURN:
628             case FRETURN:
629             case DRETURN:
630             case ARETURN:
631             case RETURN:
632             case RET:
633             case ATHROW:
634                 break;
635             case TABLESWITCH: // TBD
636                 break;
637             case LOOKUPSWITCH: // TBD
638                 break;
639             default:
640                 jumpInstruction = false;
641             }
642             if (jumpInstruction) {
643                 if (j < tuples.size() - 1) {
644                     tuples.get(j + 1).isLeader = true;
645                 }
646             }
647         }
648 
649     }

651     /**
652      * Builds the basic blocks for this control flow graph.
653      *
654      * @param tuples
655      *            the list of tuple representing the JVM instructions.
656      */
657     private void buildBB(ArrayList<NTuple> tuples) {
658         blockId = 0;
659         NBasicBlock block = new NBasicBlock(this, blockId++);
660         for (NTuple tuple : tuples) {
661             if (tuple.isLeader) {
662                 basicBlocks.add(block);
663                 block = new NBasicBlock(this, blockId++);
664                 if (!pcToBasicBlock.containsKey(tuple.pc)) {
665                     pcToBasicBlock.put(tuple.pc, block);
666                 }
667             }
668             block.tuples.add(tuple);
669         }
670         basicBlocks.add(block);
671     }

673     /**
674      * Implements loop detection algorithm to figure out if the specified block
675      * is a loop head or a loop tail. Also calculates the number of backward
676      * branches to the block.
677      * 
678      * @param block
679      *            a block.
680      * @param pred
681      *            block's predecessor or null.
682      */
683 
684     public void detectLoops(NBasicBlock block, NBasicBlock pred) {
685         if (!block.visited) {
686             block.visited = true;
687             block.active = true;
688             for (NBasicBlock succ : block.successors) {
689                 detectLoops(succ, block);
690             }
691             block.active = false;
692         } else if (block.active) {
693             block.isLoopHead = true;
694             pred.isLoopTail = true;
695             block.bwdBranches++;
696             block.loopIndex = NControlFlowGraph.loopIndex++;
697         }
698     }

700     /**
701      * Removes blocks that cannot be reached from the begin block (B0). Also
702      * removes these blocks from the predecessor lists.
703      */
704 
705     public void removeUnreachableBlocks() {
706         // Create a list of blocks that cannot be reached.
707         ArrayList<NBasicBlock> toRemove = new ArrayList<NBasicBlock>();
708         for (NBasicBlock block : basicBlocks) {
709             if (!block.visited) {
710                 toRemove.add(block);
711             }
712         }
713 
714         // From the predecessor list for each blocks, remove
715         // the ones that are in toRemove list.
716         for (NBasicBlock block : basicBlocks) {
717             for (NBasicBlock pred : toRemove) {
718                 block.predecessors.remove(pred);
719             }
720         }
721 
722         // From the list of all blocks, remove the ones that
723         // are in toRemove list.
724         for (NBasicBlock block : toRemove) {
725             basicBlocks.remove(block);
726         }
727     }

729     /**
730      * Computes the dominator of each block in this cfg recursively given the
731      * starting block and its predecessor.
732      * 
733      * @param block
734      *            starting block.
735      * @param pred
736      *            block's predecessor.
737      */
738 
739     public void computeDominators(NBasicBlock block, NBasicBlock pred) {
740         if (block.ref > 0) {
741             block.ref--;
742         }
743         if (block.dom == null) {
744             block.dom = pred;
745         } else {
746             block.dom = commonDom(block.dom, pred);
747         }
748         if (block.ref == block.bwdBranches) {
749             for (NBasicBlock s : block.successors) {
750                 computeDominators(s, block);
751             }
752         }
753     }

755     /**
756      * Converts tuples in each block to their high-level (HIR) representations.
757      */
758 
759     public void tuplesToHir() {
760         clearBlockVisitations();
761         hirId = 0;
762         loopIndex = 0;
763         hirMap = new TreeMap<Integer, NHIRInstruction>();
764         int numLocals = numLocals();
765         int[] locals = new int[numLocals];
766         ArrayList<String> argTypes = argumentTypes(desc);
767         NBasicBlock beginBlock = basicBlocks.get(0);
768         for (int i = 0; i < locals.length; i++) {
769             NHIRInstruction ins = null;
770             if (i < argTypes.size()) {
771                 String lType = argTypes.get(i);
772                 ins = new NHIRLoadLocal(beginBlock, hirId++, i,
773                         shortType(lType), lType);
774                 beginBlock.hir.add(ins.id);
775             } else {
776                 ins = new NHIRLocal(beginBlock, hirId++, i, "", "");
777             }
778             beginBlock.cfg.hirMap.put(ins.id, ins);
779             locals[i] = ins.id;
780         }
781         beginBlock.locals = locals;
782         Stack<Integer> operandStack = new Stack<Integer>();
783         Queue<NBasicBlock> q = new LinkedList<NBasicBlock>();
784         beginBlock.visited = true;
785         q.add(beginBlock);
786         while (!q.isEmpty()) {
787             NBasicBlock block = q.remove();
788             for (NBasicBlock succ : block.successors) {
789                 if (!succ.visited) {
790                     succ.visited = true;
791                     q.add(succ);
792                 }
793             }
794 
795             // Convert tuples in block to HIR instructions.
796             if (block.predecessors.size() == 1) {
797                 block.locals = block.predecessors.get(0).locals.clone();
798             } else if (block.predecessors.size() > 1) {
799                 if (block.isLoopHead) {
800                     for (NBasicBlock pred : block.predecessors) {
801                         if (pred.locals != null) {
802                             block.locals = pred.locals.clone();
803                             break;
804                         }
805                     }
806                     for (int i = 0; i < block.locals.length; i++) {
807                         ArrayList<Integer> args = new ArrayList<Integer>();
808                         NHIRPhiFunction phi = new NHIRPhiFunction(block,
809                                 hirId++, args, i);
810                         block.hir.add(phi.id);
811                         block.cfg.hirMap.put(phi.id, phi);
812                         phi.arguments.add(block.locals[i]);
813                         for (int j = 1; j < block.predecessors.size(); j++) {
814                             phi.arguments.add(phi.id);
815                         }
816                         block.locals[i] = phi.id;
817                         phi.inferType();
818                     }
819                 } else {
820                     block.locals = block.predecessors.get(0).locals.clone();
821                     for (int i = 1; i < block.predecessors.size(); i++) {
822                         NBasicBlock pred = block.predecessors.get(i);
823                         mergeLocals(block, pred);
824                     }
825                 }
826             }
827             for (NTuple tuple : block.tuples) {
828                 CLInsInfo insInfo = CLInstruction.instructionInfo[tuple.opcode];
829                 int localVariableIndex = insInfo.localVariableIndex;
830                 NHIRInstruction ins = null;
831                 short operandByte1 = 0, operandByte2 = 0, operandByte3 = 0, offset = 0;
832                 int operand1 = 0, operand2 = 0, operand3 = 0;
833                 switch (insInfo.opcode) {
834                 case MULTIANEWARRAY: {
835                     operandByte1 = tuple.operands.get(0);
836                     operandByte2 = tuple.operands.get(1);
837                     operandByte3 = tuple.operands.get(2);
838                     int index = shortValue(operandByte1, operandByte2);
839                     int classIndex = ((CLConstantClassInfo) cp.cpItem(index)).nameIndex;
840                     String type = new String(((CLConstantUtf8Info) cp
841                             .cpItem(classIndex)).b);
842                     ins = new NHIRNewArray(block, hirId++, insInfo.opcode,
843                             (int) operandByte3, shortType(type), type);
844                     block.cfg.hirMap.put(ins.id, ins);
845                     block.hir.add(ins.id);
846                     operandStack.push(ins.id);
847                     break;
848                 }
849                 case AALOAD: {
850                     operand2 = operandStack.pop();
851                     operand1 = operandStack.pop();
852 
853                     // Compute base address.
854                     NHIRInstruction ins1 = new NHIRIntConstant(block, hirId++,
855                             12);
856                     NHIRInstruction ins2 = new NHIRArithmetic(block, hirId++,
857                             IADD, operand1, ins1.id);
858                     block.cfg.hirMap.put(ins1.id, ins1);
859                     block.hir.add(ins1.id);
860                     block.cfg.hirMap.put(ins2.id, ins2);
861                     block.hir.add(ins2.id);
862 
863                     // Compute index.
864                     NHIRInstruction ins3 = new NHIRIntConstant(block, hirId++,
865                             4);
866                     NHIRInstruction ins4 = new NHIRArithmetic(block, hirId++,
867                             IMUL, operand2, ins3.id);
868                     block.cfg.hirMap.put(ins3.id, ins3);
869                     block.hir.add(ins3.id);
870                     block.cfg.hirMap.put(ins4.id, ins4);
871                     block.hir.add(ins4.id);
872 
873                     ins = new NHIRALoad(block, hirId++, insInfo.opcode,
874                             ins2.id, ins4.id, "L", "L");
875                     block.cfg.hirMap.put(ins.id, ins);
876                     block.hir.add(ins.id);
877                     operandStack.push(ins.id);
878                     break;
879                 }
880                 case IALOAD: {
881                     operand2 = operandStack.pop();
882                     operand1 = operandStack.pop();
883 
884                     // Compute base address.
885                     NHIRInstruction ins1 = new NHIRIntConstant(block, hirId++,
886                             12);
887                     NHIRInstruction ins2 = new NHIRArithmetic(block, hirId++,
888                             IADD, operand1, ins1.id);
889                     block.cfg.hirMap.put(ins1.id, ins1);
890                     block.hir.add(ins1.id);
891                     block.cfg.hirMap.put(ins2.id, ins2);
892                     block.hir.add(ins2.id);
893 
894                     // Compute index.
895                     NHIRInstruction ins3 = new NHIRIntConstant(block, hirId++,
896                             4);
897                     NHIRInstruction ins4 = new NHIRArithmetic(block, hirId++,
898                             IMUL, operand2, ins3.id);
899                     block.cfg.hirMap.put(ins3.id, ins3);
900                     block.hir.add(ins3.id);
901                     block.cfg.hirMap.put(ins4.id, ins4);
902                     block.hir.add(ins4.id);
903 
904                     ins = new NHIRALoad(block, hirId++, insInfo.opcode,
905                             ins2.id, ins4.id, "I", "I");
906                     block.cfg.hirMap.put(ins.id, ins);
907                     block.hir.add(ins.id);
908                     operandStack.push(ins.id);
909                     break;
910                 }
911                 case IASTORE: {
912                     operand3 = operandStack.pop();
913                     operand2 = operandStack.pop();
914                     operand1 = operandStack.pop();
915 
916                     // Compute base address.
917                     NHIRInstruction ins1 = new NHIRIntConstant(block, hirId++,
918                             12);
919                     NHIRInstruction ins2 = new NHIRArithmetic(block, hirId++,
920                             IADD, operand1, ins1.id);
921                     block.cfg.hirMap.put(ins1.id, ins1);
922                     block.hir.add(ins1.id);
923                     block.cfg.hirMap.put(ins2.id, ins2);
924                     block.hir.add(ins2.id);
925 
926                     // Compute index.
927                     NHIRInstruction ins3 = new NHIRIntConstant(block, hirId++,
928                             4);
929                     NHIRInstruction ins4 = new NHIRArithmetic(block, hirId++,
930                             IMUL, operand2, ins3.id);
931                     block.cfg.hirMap.put(ins3.id, ins3);
932                     block.hir.add(ins3.id);
933                     block.cfg.hirMap.put(ins4.id, ins4);
934                     block.hir.add(ins4.id);
935 
936                     ins = new NHIRAStore(block, hirId++, insInfo.opcode,
937                             ins2.id, ins4.id, operand3, "I", "I");
938                     block.cfg.hirMap.put(ins.id, ins);
939                     block.hir.add(ins.id);
940                     break;
941                 }
942                 case ICONST_0:
943                 case ICONST_1:
944                 case ICONST_2:
945                 case ICONST_3:
946                 case ICONST_4:
947                 case ICONST_5: {
948                     ins = new NHIRIntConstant(block, hirId++, tuple.opcode - 3);
949                     block.cfg.hirMap.put(ins.id, ins);
950                     block.hir.add(ins.id);
951                     operandStack.push(ins.id);
952                     break;
953                 }
954                 case ILOAD: {
955                     operandByte1 = tuple.operands.get(0);
956                     localVariableIndex = operandByte1;
957                     operandStack.push(block.locals[localVariableIndex]);
958                     break;
959                 }
960                 case ILOAD_0:
961                 case ILOAD_1:
962                 case ILOAD_2:
963                 case ILOAD_3:
964                 case ALOAD_0:
965                 case ALOAD_1:
966                 case ALOAD_2:
967                 case ALOAD_3: {
968                     operandStack.push(block.locals[localVariableIndex]);
969                     break;
970                 }
971                 case ISTORE: {
972                     operandByte1 = tuple.operands.get(0);
973                     localVariableIndex = operandByte1;
974                     block.locals[localVariableIndex] = operandStack.pop();
975                     break;
976                 }
977                 case ISTORE_0:
978                 case ISTORE_1:
979                 case ISTORE_2:
980                 case ISTORE_3:
981                 case ASTORE_0:
982                 case ASTORE_1:
983                 case ASTORE_2:
984                 case ASTORE_3: {
985                     block.locals[localVariableIndex] = operandStack.pop();
986                     break;
987                 }
988                 case BIPUSH: {
989                     operandByte1 = tuple.operands.get(0);
990                     ins = new NHIRIntConstant(block, hirId++, operandByte1);
991                     block.cfg.hirMap.put(ins.id, ins);
992                     block.hir.add(ins.id);
993                     operandStack.push(ins.id);
994                     break;
995                 }
996                 case SIPUSH: {
997                     operandByte1 = tuple.operands.get(0);
998                     operandByte2 = tuple.operands.get(1);
999                     ins = new NHIRIntConstant(block, hirId++, shortValue(
1000                            operandByte1, operandByte2));
1001                    block.cfg.hirMap.put(ins.id, ins);
1002                    block.hir.add(ins.id);
1003                    operandStack.push(ins.id);
1004                    break;
1005                }
1006                case LDC: {
1007                    operandByte1 = tuple.operands.get(0);
1008
1009                    // Only allowing ldc of string constants for
1010                    // now.
1011                    int stringIndex = ((CLConstantStringInfo) cp
1012                            .cpItem(operandByte1)).stringIndex;
1013                    String s = new String(((CLConstantUtf8Info) cp
1014                            .cpItem(stringIndex)).b);
1015                    ins = new NHIRStringConstant(block, hirId++, s);
1016                    block.cfg.hirMap.put(ins.id, ins);
1017                    block.hir.add(ins.id);
1018                    operandStack.push(ins.id);
1019                    break;
1020                }
1021                case IADD:
1022                case ISUB:
1023                case IMUL: {
1024                    operand2 = operandStack.pop();
1025                    operand1 = operandStack.pop();
1026                    ins = new NHIRArithmetic(block, hirId++, insInfo.opcode,
1027                            operand1, operand2);
1028                    block.cfg.hirMap.put(ins.id, ins);
1029                    block.hir.add(ins.id);
1030                    operandStack.push(ins.id);
1031                    break;
1032                }
1033                case IINC: {
1034                    operandByte1 = tuple.operands.get(0);
1035                    operandByte2 = tuple.operands.get(1);
1036                    operand1 = block.locals[operandByte1];
1037                    NHIRInstruction ins1 = new NHIRIntConstant(block, hirId++,
1038                            (byte) operandByte2);
1039                    ins = new NHIRArithmetic(block, hirId++, IADD, operand1,
1040                            ins1.id);
1041                    block.locals[operandByte1] = ins.id;
1042                    block.hir.add(ins1.id);
1043                    block.cfg.hirMap.put(ins1.id, ins1);
1044                    block.hir.add(ins.id);
1045                    block.cfg.hirMap.put(ins.id, ins);
1046                    break;
1047                }
1048                case IF_ICMPNE:
1049                case IF_ICMPGT:
1050                case IF_ICMPLE: {
1051                    operandByte1 = tuple.operands.get(0);
1052                    operandByte2 = tuple.operands.get(1);
1053                    offset = shortValue(operandByte1, operandByte2);
1054                    int rhs = operandStack.pop();
1055                    int lhs = operandStack.pop();
1056                    NBasicBlock trueDestination = pcToBasicBlock.get(tuple.pc
1057                            + offset);
1058                    NBasicBlock falseDestination = pcToBasicBlock
1059                            .get(tuple.pc + 3);
1060                    ins = new NHIRConditionalJump(block, hirId++, lhs, rhs,
1061                            insInfo.opcode, trueDestination, falseDestination);
1062                    block.cfg.hirMap.put(ins.id, ins);
1063                    block.hir.add(ins.id);
1064                    break;
1065                }
1066                case GOTO: {
1067                    operandByte1 = tuple.operands.get(0);
1068                    operandByte2 = tuple.operands.get(1);
1069                    offset = shortValue(operandByte1, operandByte2);
1070                    NBasicBlock destination = pcToBasicBlock.get(tuple.pc
1071                            + offset);
1072                    ins = new NHIRGoto(block, hirId++, destination);
1073                    block.cfg.hirMap.put(ins.id, ins);
1074                    block.hir.add(ins.id);
1075                    break;
1076                }
1077                case GETSTATIC:
1078                case PUTSTATIC: {
1079                    operandByte1 = tuple.operands.get(0);
1080                    operandByte2 = tuple.operands.get(1);
1081                    int index = shortValue(operandByte1, operandByte2);
1082                    int classIndex = ((CLConstantFieldRefInfo) cp.cpItem(index)).classIndex;
1083                    int nameAndTypeIndex = ((CLConstantFieldRefInfo) cp
1084                            .cpItem(index)).nameAndTypeIndex;
1085                    int nameIndex = ((CLConstantClassInfo) cp
1086                            .cpItem(classIndex)).nameIndex;
1087                    String target = new String(((CLConstantUtf8Info) cp
1088                            .cpItem(nameIndex)).b);
1089                    int fieldNameIndex = ((CLConstantNameAndTypeInfo) cp
1090                            .cpItem(nameAndTypeIndex)).nameIndex;
1091                    int fieldDescIndex = ((CLConstantNameAndTypeInfo) cp
1092                            .cpItem(nameAndTypeIndex)).descriptorIndex;
1093                    String name = new String(((CLConstantUtf8Info) cp
1094                            .cpItem(fieldNameIndex)).b);
1095                    String desc = new String(((CLConstantUtf8Info) cp
1096                            .cpItem(fieldDescIndex)).b);
1097                    if (insInfo.opcode == PUTSTATIC) {
1098                        ins = new NHIRPutField(block, hirId++, insInfo.opcode,
1099                                target, name, shortType(desc), desc,
1100                                operandStack.pop());
1101                    } else {
1102                        ins = new NHIRGetField(block, hirId++, insInfo.opcode,
1103                                target, name, shortType(desc), desc);
1104                        operandStack.push(ins.id);
1105                    }
1106                    block.cfg.hirMap.put(ins.id, ins);
1107                    block.hir.add(ins.id);
1108                    break;
1109                }
1110                case INVOKESPECIAL:
1111                case INVOKESTATIC: {
1112                    operandByte1 = tuple.operands.get(0);
1113                    operandByte2 = tuple.operands.get(1);
1114                    int index = shortValue(operandByte1, operandByte2);
1115                    int classIndex = ((CLConstantMethodRefInfo) cp
1116                            .cpItem(index)).classIndex;
1117                    int nameAndTypeIndex = ((CLConstantMethodRefInfo) cp
1118                            .cpItem(index)).nameAndTypeIndex;
1119                    int nameIndex = ((CLConstantClassInfo) cp
1120                            .cpItem(classIndex)).nameIndex;
1121                    String target = new String(((CLConstantUtf8Info) cp
1122                            .cpItem(nameIndex)).b);
1123                    int methodNameIndex = ((CLConstantNameAndTypeInfo) cp
1124                            .cpItem(nameAndTypeIndex)).nameIndex;
1125                    int methodDescIndex = ((CLConstantNameAndTypeInfo) cp
1126                            .cpItem(nameAndTypeIndex)).descriptorIndex;
1127                    String name = new String(((CLConstantUtf8Info) cp
1128                            .cpItem(methodNameIndex)).b);
1129                    String desc = new String(((CLConstantUtf8Info) cp
1130                            .cpItem(methodDescIndex)).b);
1131                    ArrayList<Integer> args = new ArrayList<Integer>();
1132                    int numArgs = argumentCount(desc);
1133                    for (int i = 0; i < numArgs; i++) {
1134                        int arg = operandStack.pop();
1135                        args.add(0, arg);
1136                    }
1137                    String returnType = returnType(desc);
1138                    ins = new NHIRInvoke(block, hirId++, insInfo.opcode,
1139                            target, name, args, shortType(returnType),
1140                            returnType);
1141                    if (!returnType.equals("V")) {
1142                        operandStack.push(ins.id);
1143                    }
1144                    block.cfg.hirMap.put(ins.id, ins);
1145                    block.hir.add(ins.id);
1146                    break;
1147                }
1148                case IRETURN:
1149                case ARETURN: {
1150                    ins = new NHIRReturn(block, hirId++, insInfo.opcode,
1151                            operandStack.pop());
1152                    block.cfg.hirMap.put(ins.id, ins);
1153                    block.hir.add(ins.id);
1154                    break;
1155                }
1156                case RETURN: {
1157                    ins = new NHIRReturn(block, hirId++, insInfo.opcode, -1);
1158                    block.cfg.hirMap.put(ins.id, ins);
1159                    block.hir.add(ins.id);
1160                    break;
1161                }
1162                }
1163            }
1164        }
1165    }

1175    /**
1176     * Eliminates redundant phi functions of the form x = (y, x, x, ..., x) with
1177     * y.
1178     */
1179
1180    public void eliminateRedundantPhiFunctions() {
1181        for (int ins : hirMap.keySet()) {
1182            NHIRInstruction hir = hirMap.get(ins);
1183            if (hir instanceof NHIRPhiFunction) {
1184                NHIRPhiFunction phi = (NHIRPhiFunction) hir;
1185                int firstArg = phi.arguments.get(0);
1186                boolean match = true;
1187                NBasicBlock block = phi.block;
1188                if (!block.isLoopHead) {
1189                    continue;
1190                }
1191                for (int i = 1; i < phi.arguments.size(); i++) {
1192                    if (phi.arguments.get(i) != block.predecessors.get(i).locals[phi.local]) {
1193                        match = false;
1194                        phi.arguments.set(i,
1195                                block.predecessors.get(i).locals[phi.local]);
1196                    }
1197                }
1198                if (match && firstArg != phi.id) {
1199                    hirMap.put(phi.id, hirMap.get(firstArg));
1200                    phi.block.hir.remove((Integer) phi.id);
1201                }
1202            }
1203        }
1204    }

1206    /**
1207     * Converts the hir instructions in this cfg to lir instructions.
1208     */
1209
1210    public void hirToLir() {
1211        lirId = 0;
1212        regId = 32;
1213        offset = 0;
1214        registers = new ArrayList<NRegister>();
1215        data = new ArrayList<String>();
1216        for (int i = 0; i < 32; i++) {
1217            registers.add(null);
1218        }
1219        pRegisters = new ArrayList<NPhysicalRegister>();
1220        for (int ins : hirMap.keySet()) {
1221            hirMap.get(ins).toLir();
1222        }
1223
1224        // We now know how many virtual registers are needed, so
1225        // we can initialize bitset fields in each block that are
1226        // needed for interval calculation.
1227        int size = registers.size();
1228        for (NBasicBlock block : basicBlocks) {
1229            block.liveDef = new BitSet(size);
1230            block.liveUse = new BitSet(size);
1231            block.liveIn  = new BitSet(size);
1232            block.liveOut = new BitSet(size);
1233        }
1234    }

1236    /**
1237     * Resolves the phi functions in this cfg, i.e., for each x = phi(x1, x2,
1238     * ..., xn) generate an (LIR) move xi, x instruction at the end of the
1239     * predecessor i of thte block defining the phi function; if the instruction
1240     * there is a branch, add the instruction prior to the branch.
1241     */
1242
1243    public void resolvePhiFunctions() {
1244        for (int ins1 : hirMap.keySet()) {
1245            NHIRInstruction hir = hirMap.get(ins1);
1246            if (hir instanceof NHIRPhiFunction) {
1247                NHIRPhiFunction phi = (NHIRPhiFunction) hir;
1248                NBasicBlock block = phi.block;
1249                for (int i = 0; i < phi.arguments.size(); i++) {
1250                    NHIRInstruction arg = hirMap.get(phi.arguments.get(i));
1251                    if (arg.sType.equals("")) {
1252                        continue;
1253                    }
1254                    NBasicBlock targetBlock = block.predecessors.get(i);
1255                    NLIRMove move = new NLIRMove(arg.block, lirId++, arg.lir,
1256                            phi.lir);
1257                    int len = targetBlock.hir.size();
1258                    if (isHIRJmp(hirMap.get(targetBlock.hir.get(len - 1)))) {
1259                        targetBlock.lir.add(len - 1, move);
1260                    } else {
1261                        targetBlock.lir.add(move);
1262                    }
1263                }
1264            }
1265        }
1266    }

1307     * Assigns new ids to the LIR instructions in this cfg.
1308     */

1310    public void renumberLirInstructions() {
1311        int nextId = 0;
1312        for (NBasicBlock block : basicBlocks) {
1313            ArrayList<NLIRInstruction> newLir = new ArrayList<NLIRInstruction>();
1314            for (NLIRInstruction lir : block.lir) {
1315                if (lir instanceof NLIRLoadLocal
1316                        && ((NLIRLoadLocal) lir).local < 4) {
1317                    // Ignore first four formals.
1318                    continue;

1322                             //   don't use it
1323                newLir.add(lir);
1324            }
1325            block.lir = newLir;
1326        }

1423     *            predecessor of b.
1424     * 
1425     * @return common dominator of the given block and its predecessor.

1427
1428    private NBasicBlock commonDom(NBasicBlock b, NBasicBlock pred) {
1429        NBasicBlock dom = b;
1430        clearBlockVisitations();
1431        while (dom != null) {
1432            dom.visited = true;
1433            dom = dom.dom;
1434        }
1435        dom = pred;
1436        while (!dom.visited) {
1437            dom = dom.dom;
1438        }
1439        return dom;

1457     * Merge the locals in block b with the locals in block a.
1458     * 
1459     * @param a
1460     *            block to merge into.
1461     * @param b
1462     *            block to merge from.
1463     */
1464
1465    private void mergeLocals(NBasicBlock a, NBasicBlock b) {
1466        for (int i = 0; i < a.locals.length; i++) {
1467            if (a.cfg.hirMap.get(a.locals[i]).equals(
1468                    b.cfg.hirMap.get(b.locals[i]))) {
1469                continue;
1470            } else {
1471                ArrayList<Integer> args = new ArrayList<Integer>();
1472                args.add(a.locals[i]);
1473                args.add(b.locals[i]);
1474                NHIRInstruction ins = new NHIRPhiFunction(a,
1475                        NControlFlowGraph.hirId++, args, i);
1476                a.locals[i] = ins.id;
1477                a.hir.add(ins.id);
1478                a.cfg.hirMap.put(ins.id, ins);
1479                ((NHIRPhiFunction) ins).inferType();
1480            }
1481        }
1482    }

1486     * representations.

1493
1494    private ArrayList<NTuple> bytecodeToTuples(ArrayList<Integer> code) {
1495        ArrayList<NTuple> tuples = new ArrayList<NTuple>();
1496        for (int i = 0; i < code.size(); i++) {
1497            int pc = i;
1498            int opcode = code.get(i);
1499            int operandBytes = CLInstruction.instructionInfo[opcode].operandCount;
1500            short operandByte1, operandByte2, operandByte3, operandByte4;
1501            int pad, deflt;
1502            ArrayList<Short> operands = new ArrayList<Short>();
1503            switch (operandBytes) {
1504            case 0:
1505                break;
1506            case 1:
1507                operandByte1 = code.get(++i).shortValue();
1508                operands.add(operandByte1);
1509                break;
1510            case 2:
1511                operandByte1 = code.get(++i).shortValue();
1512                operandByte2 = code.get(++i).shortValue();
1513                operands.add(operandByte1);
1514                operands.add(operandByte2);
1515                break;
1516            case 3:
1517                operandByte1 = code.get(++i).shortValue();
1518                operandByte2 = code.get(++i).shortValue();
1519                operandByte3 = code.get(++i).shortValue();
1520                operands.add(operandByte1);
1521                operands.add(operandByte2);
1522                operands.add(operandByte3);
1523                break;
1524            case 4:
1525                operandByte1 = code.get(++i).shortValue();
1526                operandByte2 = code.get(++i).shortValue();
1527                operandByte3 = code.get(++i).shortValue();
1528                operandByte4 = code.get(++i).shortValue();
1529                operands.add(operandByte1);
1530                operands.add(operandByte2);
1531                operands.add(operandByte3);
1532                operands.add(operandByte4);
1533                break;
1534            case DYNAMIC: // TBD
1535                break;
1536            }
1537            tuples.add(new NTuple(pc, opcode, operands));
1538        }
1539        return tuples;
