6   import java.io.FileReader;

10  import java.util.Hashtable;

12  import static jminusminus.TokenKind.*;

14  /**
15   * A lexical analyzer for j--, that has no backtracking mechanism.
16   * <p>
17   * When you add a new token to the scanner, you must also add an entry in the
18   * {@link TokenKind} enum in {@code TokenInfo.java} specifying the kind and 
19   * image of the new token.
20   * <p>
21   * See Appendix C.2.1 of the textbook or the
22   * <a href="https://docs.oracle.com/javase/specs/jls/se8/html/jls-3.html">Java Language Specifications</a>
23   * for the full lexical grammar.
24   */
25  
26  class Scanner {
27  
28      /** End of file character. */
29      public final static char EOFCH = CharReader.EOFCH;
30  
31      /** Keywords in j--. */
32      private Hashtable<String, TokenKind> reserved;
33  
34      /** Source characters. */
35      private CharReader input;
36  
37      /** Next unscanned character. */
38      private char ch;
39  
40      /** Whether a scanner error has been found. */
41      private boolean isInError;
42  
43      /** Source file name. */
44      private String fileName;
45  
46      /** Line number of current token. */
47      private int line;
48  
49      /**
50       * Constructs a Scanner object.
51       * 
52       * @param fileName
53       *            the name of the file containing the source.
54       * @exception FileNotFoundException
55       *                when the named file cannot be found.
56       */
57  
58      public Scanner(String fileName) throws FileNotFoundException {
59          this.input = new CharReader(fileName);
60          this.fileName = fileName;
61          isInError = false;
62  
63          // Keywords in j--
64          reserved = new Hashtable<String, TokenKind>();
65          reserved.put(ABSTRACT.image(), ABSTRACT);
66          reserved.put(BOOLEAN.image(), BOOLEAN);
67          reserved.put(CHAR.image(), CHAR);
68          reserved.put(CLASS.image(), CLASS);
69          reserved.put(ELSE.image(), ELSE);
70          reserved.put(EXTENDS.image(), EXTENDS);
71          reserved.put(FALSE.image(), FALSE);
72          reserved.put(IF.image(), IF);
73          reserved.put(IMPORT.image(), IMPORT);
74          reserved.put(INSTANCEOF.image(), INSTANCEOF);
75          reserved.put(INT.image(), INT);
76          reserved.put(NEW.image(), NEW);
77          reserved.put(NULL.image(), NULL);
78          reserved.put(PACKAGE.image(), PACKAGE);
79          reserved.put(PRIVATE.image(), PRIVATE);
80          reserved.put(PROTECTED.image(), PROTECTED);
81          reserved.put(PUBLIC.image(), PUBLIC);
82          reserved.put(RETURN.image(), RETURN);
83          reserved.put(STATIC.image(), STATIC);
84          reserved.put(SUPER.image(), SUPER);
85          reserved.put(THIS.image(), THIS);
86          reserved.put(TRUE.image(), TRUE);
87          reserved.put(VOID.image(), VOID);
88          reserved.put(WHILE.image(), WHILE);
89  
90          // Prime the pump.
91          nextCh();
92      }
93  
94      /**
95       * Scans the next token from input.
96       * 
97       * @return the next scanned token.
98       */
99  
100     public TokenInfo getNextToken() {
101         StringBuffer buffer;
102         boolean moreWhiteSpace = true;
103         while (moreWhiteSpace) {
104             while (isWhitespace(ch)) {
105                 nextCh();
106             }
107             if (ch == '/') {
108                 nextCh();
109                 if (ch == '/') {
110                     // CharReader maps all new lines to '\n'
111                     while (ch != '\n' && ch != EOFCH) {
112                         nextCh();
113                     }
114                 } else {
115                     reportScannerError("Operator / is not supported in j--.");
116                 }
117             } else {
118                 moreWhiteSpace = false;
119             }
120         }
121         line = input.line();
122         switch (ch) {
123         case '(':
124             nextCh();
125             return new TokenInfo(LPAREN, line);
126         case ')':
127             nextCh();
128             return new TokenInfo(RPAREN, line);
129         case '{':
130             nextCh();
131             return new TokenInfo(LCURLY, line);
132         case '}':
133             nextCh();
134             return new TokenInfo(RCURLY, line);
135         case '[':
136             nextCh();
137             return new TokenInfo(LBRACK, line);
138         case ']':
139             nextCh();
140             return new TokenInfo(RBRACK, line);
141         case ';':
142             nextCh();
143             return new TokenInfo(SEMI, line);
144         case ',':
145             nextCh();
146             return new TokenInfo(COMMA, line);
147         case '=':
148             nextCh();
149             if (ch == '=') {
150                 nextCh();
151                 return new TokenInfo(EQUAL, line);
152             } else {
153                 return new TokenInfo(ASSIGN, line);
154             }
155         case '!':
156             nextCh();
157             return new TokenInfo(LNOT, line);
158         case '*':
159             nextCh();
160             return new TokenInfo(STAR, line);
161         case '+':
162             nextCh();
163             if (ch == '=') {
164                 nextCh();
165                 return new TokenInfo(PLUS_ASSIGN, line);
166             } else if (ch == '+') {
167                 nextCh();
168                 return new TokenInfo(INC, line);
169             } else {
170                 return new TokenInfo(PLUS, line);
171             }
172         case '-':
173             nextCh();
174             if (ch == '-') {
175                 nextCh();
176                 return new TokenInfo(DEC, line);
177             } else {
178                 return new TokenInfo(MINUS, line);
179             }
180         case '&':
181             nextCh();
182             if (ch == '&') {
183                 nextCh();
184                 return new TokenInfo(LAND, line);
185             } else {
186                 reportScannerError("Operator & is not supported in j--.");
187                 return getNextToken();
188             }
189         case '>':
190             nextCh();
191             return new TokenInfo(GT, line);
192         case '<':
193             nextCh();
194             if (ch == '=') {
195                 nextCh();
196                 return new TokenInfo(LE, line);
197             } else {
198                 reportScannerError("Operator < is not supported in j--.");
199                 return getNextToken();
200             }
201         case '\'':
202             buffer = new StringBuffer();
203             buffer.append('\'');
204             nextCh();
205             if (ch == '\\') {
206                 nextCh();
207                 buffer.append(escape());
208             } else {
209                 buffer.append(ch);
210                 nextCh();
211             }
212             if (ch == '\'') {
213                 buffer.append('\'');
214                 nextCh();
215                 return new TokenInfo(CHAR_LITERAL, buffer.toString(), line);
216             } else {
217                 // Expected a ' ; report error and try to
218                 // recover.
219                 reportScannerError(ch
220                         + " found by scanner where closing ' was expected.");
221                 while (ch != '\'' && ch != ';' && ch != '\n') {
222                     nextCh();
223                 }
224                 return new TokenInfo(CHAR_LITERAL, buffer.toString(), line);
225             }
226         case '"':
227             buffer = new StringBuffer();
228             buffer.append("\"");
229             nextCh();
230             while (ch != '"' && ch != '\n' && ch != EOFCH) {
231                 if (ch == '\\') {
232                     nextCh();
233                     buffer.append(escape());
234                 } else {
235                     buffer.append(ch);
236                     nextCh();
237                 }
238             }
239             if (ch == '\n') {
240                 reportScannerError("Unexpected end of line found in String");
241             } else if (ch == EOFCH) {
242                 reportScannerError("Unexpected end of file found in String");
243             } else {
244                 // Scan the closing "
245                 nextCh();
246                 buffer.append("\"");
247             }
248             return new TokenInfo(STRING_LITERAL, buffer.toString(), line);
249         case '.':
250             nextCh();
251             return new TokenInfo(DOT, line);
252         case EOFCH:
253             return new TokenInfo(EOF, line);
254         case '0':
255             // Handle only simple decimal integers for now.
256             nextCh();
257             return new TokenInfo(INT_LITERAL, "0", line);
258         case '1':
259         case '2':
260         case '3':
261         case '4':
262         case '5':
263         case '6':
264         case '7':
265         case '8':
266         case '9':
267             buffer = new StringBuffer();
268             while (isDigit(ch)) {
269                 buffer.append(ch);
270                 nextCh();
271             }
272             return new TokenInfo(INT_LITERAL, buffer.toString(), line);
273         default:
274             if (isIdentifierStart(ch)) {
275                 buffer = new StringBuffer();
276                 while (isIdentifierPart(ch)) {
277                     buffer.append(ch);
278                     nextCh();
279                 }
280                 String identifier = buffer.toString();
281                 if (reserved.containsKey(identifier)) {
282                     return new TokenInfo(reserved.get(identifier), line);
283                 } else {
284                     return new TokenInfo(IDENTIFIER, identifier, line);
285                 }
286             } else {
287                 reportScannerError("Unidentified input token: '%c'", ch);
288                 nextCh();
289                 return getNextToken();
290             }
291         }
292     }
293 
294     /**
295      * Scans and returns an escaped character.
296      * 
297      * @return escaped character.
298      */
299 
300     private String escape() {
301         switch (ch) {
302         case 'b':
303             nextCh();
304             return "\\b";
305         case 't':
306             nextCh();
307             return "\\t";
308         case 'n':
309             nextCh();
310             return "\\n";
311         case 'f':
312             nextCh();
313             return "\\f";
314         case 'r':
315             nextCh();
316             return "\\r";
317         case '"':
318             nextCh();
319             return "\"";
320         case '\'':
321             nextCh();
322             return "\\'";
323         case '\\':
324             nextCh();
325             return "\\\\";
326         default:
327             reportScannerError("Badly formed escape: \\%c", ch);
328             nextCh();
329             return "";
330         }
331     }
332 
333     /**
334      * Advances ch to the next character from input, and updates the line 
335      * number.
336      */
337 
338     private void nextCh() {
339         line = input.line();
340         try {
341             ch = input.nextChar();
342         } catch (Exception e) {
343             reportScannerError("Unable to read characters from input");
344         }
345     }
346 
347     /**
348      * Reports a lexcial error and records the fact that an error has occured.
349      * This fact can be ascertained from the Scanner by sending it an
350      * errorHasOccurred message.
351      * 
352      * @param message
353      *            message identifying the error.
354      * @param args
355      *            related values.
356      */
357 
358     private void reportScannerError(String message, Object... args) {
359         isInError = true;
360         System.err.printf("%s:%d: ", fileName, line);
361         System.err.printf(message, args);
362         System.err.println();
363     }
364 
365     /**
366      * Returns true if the specified character is a digit (0-9); false otherwise.
367      * 
368      * @param c
369      *            character.
370      * @return true or false.
371      */
372 
373     private boolean isDigit(char c) {
374         return (c >= '0' && c <= '9');
375     }
376 
377     /**
378      * Returns true if the specified character is a whitespace; false otherwise.
379      * 
380      * @param c
381      *            character.
382      * @return true or false.
383      */
384 
385     private boolean isWhitespace(char c) {
386         return (c == ' ' || c == '\t' || c == '\n' || c == '\f');
387     }
388 
389     /**
390      * Returns true if the specified character can start an identifier name;
391      * false otherwise.
392      * 
393      * @param c
394      *            character.
395      * @return true or false.
396      */
397 
398     private boolean isIdentifierStart(char c) {
399         return (c >= 'a' && c <= 'z' || c >= 'A' && c <= 'Z' || c == '_' || c == '$');
400     }
401 
402     /**
403      * Returns true if the specified character can be part of an identifier name;
404      * false otherwise.
405      * 
406      * @param c
407      *            character.
408      * @return true or false.
409      */
410 
411     private boolean isIdentifierPart(char c) {
412         return (isIdentifierStart(c) || isDigit(c));
413     }
414 
415     /**
416      * Has an error occurred up to now in lexical analysis?
417      * 
418      * @return {@code true} if an error occurred and {@code false} otherwise.
419      */
420 
421     public boolean errorHasOccurred() {
422         return isInError;
423     }
424 
425     /**
426      * Returns the name of the source file.
427      * 
428      * @return name of the source file.
429      */
430 
431     public String fileName() {
432         return fileName;
433     }

437 /**
438  * A buffered character reader. Abstracts out differences between platforms,
439  * mapping all new lines to '\n'. Also, keeps track of line numbers where the
440  * first line is numbered 1.
441  */
442 
443 class CharReader {
444 
445     /** A representation of the end of file as a character. */
446     public final static char EOFCH = (char) -1;
447 
448     /** The underlying reader records line numbers. */
449     private LineNumberReader lineNumberReader;
450 
451     /** Name of the file that is being read. */
452     private String fileName;
453 
454     /**
455      * Constructs a CharReader from a file name.
456      * 
457      * @param fileName
458      *            the name of the input file.
459      * @exception FileNotFoundException
460      *                if the file is not found.
461      */
462 
463     public CharReader(String fileName) throws FileNotFoundException {
464         lineNumberReader = new LineNumberReader(new FileReader(fileName));
465         this.fileName = fileName;
466     }
467 
468     /**
469      * Scans the next character.
470      * 
471      * @return the character scanned.
472      * @exception IOException
473      *                if an I/O error occurs.
474      */
475 
476     public char nextChar() throws IOException {
477         return (char) lineNumberReader.read();
478     }
479 
480     /**
481      * Returns the current line number in the source file, starting at 1.
482      * 
483      * @return the current line number.
484      */
485 
486     public int line() {
487         // LineNumberReader counts lines from 0.
488         return lineNumberReader.getLineNumber() + 1;
489     }
490 
491     /**
492      * Returns the file name.
493      * 
494      * @return the file name.
495      */
496 
497     public String fileName() {
498         return fileName;
499     }
500 
501     /**
502      * Closes the file.
503      * 
504      * @exception IOException
505      *                if an I/O error occurs.
506      */
507 
508     public void close() throws IOException {
509         lineNumberReader.close();
510     }
